def _safe_subagent_path_component(value: str | None, default: str) -> str:
    raw = (value or default).strip() or default
    safe = re.sub(r"[^A-Za-z0-9_.-]+", "-", raw).strip(".-")
    return safe or default


def _is_workspace_keep_placeholder(path: str, data: bytes) -> bool:
    return path.endswith(".keep") and len(data) == 0


def _workspace_path_from_stable_rw(path: str) -> str:
    if path == STABLE_CORTEX_RW:
        return "/rw/"
    if not path.startswith(STABLE_CORTEX_RW + "/"):
        raise ValueError(f"LogicalFS patch path must be under {STABLE_CORTEX_RW}: {path!r}")
    return "/rw/" + path.removeprefix(STABLE_CORTEX_RW + "/")

class MountNamespaceLogicalFS:
    """Cortex Workspace adapter over the generic LogicalFS substrate."""

    def __init__(
        self,
        workspace: Workspace,
        *,
        max_sync_bytes: int | None = None,
        max_wall_timeout: int | None = None,
        token_factory: Callable[[], str] | None = None,
        cortex_api_url: str = "http://localhost:19996",
        provider: LocalLogicalFSProvider | None = None,
    ) -> None:
        self._workspace = workspace
        self._max_sync_bytes = max_sync_bytes
        self._token_factory = token_factory
        self._cortex_api_url = cortex_api_url
        self._provider = provider or LocalLogicalFSProvider()
        self.capabilities = _detect_logicalfs_capabilities()

    @staticmethod
    def _snapshot_path(logical_path: str) -> tuple[str, bool]:
        path = "/" + logical_path.strip("/")
        if path == "/ro":
            return "", False
        if path == "/rw":
            return "", True
        if path.startswith("/ro/"):
            return path.removeprefix("/ro/"), False
        if path.startswith("/rw/"):
            return path.removeprefix("/rw/"), True
        raise ValueError(f"snapshot path must be under /ro or /rw: {logical_path!r}")

    async def _add_file_if_exists(
        self,
        files: dict[tuple[bool, str], LogicalFSFile],
        logical_path: str,
    ) -> None:
        try:
            data = await self._workspace.read_bytes(logical_path)
        except (KeyError, FileNotFoundError):
            return
        relpath, writable = self._snapshot_path(logical_path)
        if relpath and not _is_workspace_keep_placeholder(relpath, data):
            files[(writable, relpath)] = LogicalFSFile(path=relpath, data=data, writable=writable)

    async def _add_tree(
        self,
        files: dict[tuple[bool, str], LogicalFSFile],
        logical_prefix: str,
    ) -> None:
        prefix = "/" + logical_prefix.strip("/") + "/"
        base_rel, writable = self._snapshot_path(prefix)
        for relpath, data in await self._workspace.read_tree_bytes(
            prefix,
            concurrency=_MATERIALIZE_CONCURRENCY,
        ):
            if not relpath:
                continue
            target = f"{base_rel.rstrip('/')}/{relpath}".strip("/")
            if target and not _is_workspace_keep_placeholder(target, data):
                files[(writable, target)] = LogicalFSFile(path=target, data=data, writable=writable)

    async def _add_directory_files(
        self,
        files: dict[tuple[bool, str], LogicalFSFile],
        logical_prefix: str,
    ) -> None:
        """Add direct files under a directory without recursively syncing children."""

        prefix = "/" + logical_prefix.strip("/") + "/"
        try:
            entries = await self._workspace.list_dir(prefix)
        except (KeyError, FileNotFoundError):
            return
        for entry in entries:
            if entry.is_dir:
                continue
            await self._add_file_if_exists(files, entry.path)

    async def _add_rw_working_set(
        self,
        files: dict[tuple[bool, str], LogicalFSFile],
        *,
        extra_env: Mapping[str, str] | None,
    ) -> None:
        """Materialize the bounded writable working set for shell execution.

        The writable tree can contain large generated artifacts. Shell startup
        must not become proportional to all historical RW files, so only direct
        root files and explicit shared/current-subagent work directories are
        mounted by default. Any new file the command writes under /rw is still
        persisted on release.
        """

        await self._add_directory_files(files, "/rw/")
        for logical_prefix in (
            "/rw/public/",
            "/rw/system/",
        ):
            await self._add_tree(files, logical_prefix)

        subagent_id = _safe_subagent_path_component((extra_env or {}).get("NOVAIC_SUBAGENT_ID"), "main")
        await self._add_tree(files, f"/rw/subagents/{subagent_id}/")

## novaic-cortex/novaic_cortex/sandbox.py
"""Cortex shell execution boundary.

The public ``Sandbox`` facade composes two explicit substrates:

* ``MountNamespaceLogicalFS`` owns Cortex filesystem semantics and the stable
  private ``/cortex`` mount namespace.
* A sandboxd SDK client owns remote process execution.

There is deliberately no alternate local path adapter. If a host cannot provide
the sandboxd service boundary, shell execution fails explicitly instead of
silently running with different path semantics.
"""

from __future__ import annotations

import tempfile
from pathlib import Path
from typing import Callable, Protocol

from sandbox_sdk import SandboxExecResult, SandboxExecSpec

from novaic_cortex.logical_fs import (
    STABLE_CORTEX_RO,
    STABLE_CORTEX_RW,
    LogicalFSCapabilities,
    MountNamespaceLogicalFS,
)
from novaic_cortex.types import ShellResult
from novaic_cortex.workspace import Workspace

_EPHEMERAL_SANDBOX_MARKER = "novaic-cortex-sandbox-"


class SandboxExecutor(Protocol):
    async def execute(self, spec: SandboxExecSpec) -> SandboxExecResult:
        ...

def _reject_ephemeral_sandbox_paths(command: str) -> ShellResult | None:
    """Reject commands that reuse temp backing paths leaked from older shell output."""
    if _EPHEMERAL_SANDBOX_MARKER not in command:
        return None
    return ShellResult(
        stdout="",
        stderr=(
            "Cortex shell paths under novaic-cortex-sandbox-* are ephemeral. "
            f"Use {STABLE_CORTEX_RO}, {STABLE_CORTEX_RW}, $RO, or $RW instead."
        ),
        exit_code=-2,
        files_changed=[],
    )


class ShellExecutionOrchestrator:
    """Acquire LogicalFS view, run a process, release view, assemble ShellResult."""

    def __init__(
        self,
        logical_fs: MountNamespaceLogicalFS,
        sandbox_executor: SandboxExecutor | None,
        *,
        max_wall_timeout: int | None = None,
    ) -> None:
        self.logical_fs = logical_fs
        self.sandbox_executor = sandbox_executor
        self._max_wall_timeout = max_wall_timeout

    async def exec(
        self,
        command: str,
        timeout: int = 30,
        cwd: str | None = None,
        extra_env: dict[str, str] | None = None,
    ) -> ShellResult:
        rejected = _reject_ephemeral_sandbox_paths(command)
        if rejected is not None:
            return rejected

        if self._max_wall_timeout is not None:
            timeout = min(timeout, self._max_wall_timeout)

        if self.sandbox_executor is None:
            return ShellResult(
                stdout="",
                stderr=(
                    "Sandbox executor is not configured. Cortex shell execution "
                    "must go through sandboxd via sandbox_sdk."
                ),
                exit_code=-2,
                files_changed=[],
            )

        with tempfile.TemporaryDirectory(prefix="novaic-cortex-logicalfs-") as tmp:
            root = Path(tmp)
            try:
                view = await self.logical_fs.acquire_view(root, cwd=cwd, extra_env=extra_env)
            except ValueError as exc:
                return ShellResult(
                    stdout="",
                    stderr=str(exc),
                    exit_code=-2,
                    files_changed=[],
                )
            if not view.cwd.is_dir():
                stderr = self.logical_fs.sanitize_output(
                    f"cwd is not a directory: {view.cwd}",
                    view,
                )
                return ShellResult(stdout="", stderr=stderr, exit_code=-2, files_changed=[])

            try:
                mount_plan = self.logical_fs.process_mount_plan(view)
            except RuntimeError as exc:
                return ShellResult(stdout="", stderr=str(exc), exit_code=-2, files_changed=[])
            process_result = await self.sandbox_executor.execute(
                SandboxExecSpec(
                    command=command,
                    cwd=str(self.logical_fs.process_cwd(view)),
                    env=view.env,
                    timeout=timeout,
                    display_command=command,
                    mount=mount_plan,
                )
            )

            try:
                from novaic_cortex.observability import log_cortex

                dur_ms = int(process_result.duration_s * 1000)
                log_cortex(
                    "sandbox.exec",
                    cmd=command[:200],
                    exit_code=process_result.exit_code,
                    duration_ms=dur_ms,
                    duration_s=round(process_result.duration_s, 3),
                    fs_provider=view.capabilities.provider,
                )
            except ImportError:
                pass

            files_changed = await self.logical_fs.release_view(view)

            stdout = process_result.stdout.decode("utf-8", errors="replace")
            stderr = process_result.stderr.decode("utf-8", errors="replace")
            stdout = self.logical_fs.sanitize_output(stdout, view)
            stderr = self.logical_fs.sanitize_output(stderr, view)

            return ShellResult(
                stdout=stdout,
                stderr=stderr,
                exit_code=process_result.exit_code,
                files_changed=files_changed,
            )


class Sandbox:
    """Public shell facade backed by explicit LogicalFS and sandboxd SDK client."""

    def __init__(
        self,
        workspace: Workspace,
        *,
        max_sync_bytes: int | None = None,
        max_wall_timeout: int | None = None,
        token_factory: Callable[[], str] | None = None,
        cortex_api_url: str = "http://localhost:19996",
        process_runner: SandboxExecutor | None = None,
    ) -> None:
        self.logical_fs = MountNamespaceLogicalFS(
            workspace,
            max_sync_bytes=max_sync_bytes,
            token_factory=token_factory,
            cortex_api_url=cortex_api_url,
        )
        self.process_runner = process_runner
        self._orchestrator = ShellExecutionOrchestrator(
            self.logical_fs,
            self.process_runner,
            max_wall_timeout=max_wall_timeout,
        )

    @property
    def logical_fs_capabilities(self) -> LogicalFSCapabilities:
        return self.logical_fs.capabilities

    async def exec(
        self,
        command: str,
        timeout: int = 30,
        cwd: str | None = None,
        extra_env: dict[str, str] | None = None,
    ) -> ShellResult:
        return await self._orchestrator.exec(
            command,
            timeout=timeout,
            cwd=cwd,
            extra_env=extra_env,
        )

## novaic-cortex/novaic_cortex/blob_payload.py
"""Blob raw-byte payload storage for Cortex.

Cortex owns work trace semantics. Blob Service owns large raw bytes. This
module is the narrow adapter between the two: it stores JSON-serialized tool
payloads in Blob Service and returns ``blob://cortex-payload/...`` refs.
"""

from __future__ import annotations

import hashlib
import json
import os
from dataclasses import dataclass
from typing import Any, Protocol
from urllib.parse import quote

from common.http.clients import internal_async_client


DEFAULT_PAYLOAD_BLOB_THRESHOLD_BYTES = 64 * 1024


class PayloadBlobClient(Protocol):
    async def put_payload(
        self,
        *,
        tenant_id: str,
        scope_path: str,
        source_payload_ref: str,
        payload: Any,
        payload_bytes: bytes,
    ) -> str:
        ...

    async def get_payload(self, *, tenant_id: str, blob_ref: str) -> Any:
        ...


@dataclass(frozen=True)
class BlobPayloadPolicy:
    threshold_bytes: int = DEFAULT_PAYLOAD_BLOB_THRESHOLD_BYTES

    @classmethod
    def from_env(cls) -> "BlobPayloadPolicy":
        raw = os.getenv("CORTEX_PAYLOAD_BLOB_THRESHOLD_BYTES", "").strip()
        if not raw:
            return cls()
        try:
            value = int(raw)
        except ValueError:
            value = DEFAULT_PAYLOAD_BLOB_THRESHOLD_BYTES
        return cls(threshold_bytes=max(0, value))

    def should_externalize(self, payload_size: int) -> bool:
        return payload_size > self.threshold_bytes


class HttpBlobPayloadClient:
    def __init__(self, base_url: str, *, timeout: float = 10.0) -> None:
        self._base_url = base_url.rstrip("/")
        self._timeout = timeout

    @classmethod
    def from_env(cls) -> "HttpBlobPayloadClient | None":
        base_url = os.getenv("CORTEX_BLOB_SERVICE_URL", "").strip()
        if not base_url:
            return None
        return cls(base_url)

    async def put_payload(
        self,
        *,
        tenant_id: str,
        scope_path: str,
        source_payload_ref: str,
        payload: Any,
        payload_bytes: bytes,
    ) -> str:
        blob_id = _payload_blob_id(tenant_id, scope_path, source_payload_ref)
        digest = hashlib.sha256(payload_bytes).hexdigest()
        create_body = {
            "tenant_id": tenant_id,
            "namespace": "cortex-payload",
            "mime_type": "application/json",
            "blob_id": blob_id,
            "expected_size": len(payload_bytes),
            "expected_sha256": digest,
            "created_by": "cortex",
            "source_ref": source_payload_ref,
        }
        async with internal_async_client("cortex", timeout=self._timeout) as client:
            create_resp = await client.post(f"{self._base_url}/v1/blobs/uploads", json=create_body)
            create_resp.raise_for_status()
            session_id = create_resp.json().get("session_id")
            if not isinstance(session_id, str) or not session_id:
                raise ValueError("Blob Service did not return an upload session_id")

            parts: list[int] = []
            if payload_bytes:
                part_resp = await client.put(
                    f"{self._base_url}/v1/blobs/uploads/{quote(session_id)}/parts/1",
                    content=payload_bytes,
                    headers={"X-Tenant-ID": tenant_id},
                )
                part_resp.raise_for_status()
                parts = [1]

            complete_resp = await client.post(
                f"{self._base_url}/v1/blobs/uploads/{quote(session_id)}/complete",
                json={"parts": parts, "final_sha256": digest},
                headers={"X-Tenant-ID": tenant_id},
            )
            complete_resp.raise_for_status()
            data = complete_resp.json().get("blob") or {}
        blob_ref = data.get("blob_ref")
        if not isinstance(blob_ref, str) or not blob_ref.startswith("blob://cortex-payload/"):
            raise ValueError("Blob Service did not return a cortex-payload BlobRef")
        return blob_ref

    async def get_payload(self, *, tenant_id: str, blob_ref: str) -> Any:
        namespace, blob_id = _parse_blob_ref(blob_ref)
        async with internal_async_client("cortex", timeout=self._timeout) as client:
            resp = await client.get(
                f"{self._base_url}/v1/blobs/{quote(namespace)}/{quote(blob_id)}",
                headers={"X-Tenant-ID": tenant_id},
            )
            resp.raise_for_status()
            raw = resp.content
        return json.loads(raw.decode("utf-8"))


def encode_payload_bytes(payload: Any) -> bytes:
    return json.dumps(payload, ensure_ascii=False, default=str).encode("utf-8")


def _payload_blob_id(tenant_id: str, scope_path: str, source_payload_ref: str) -> str:
    digest = hashlib.sha256(f"{tenant_id}:{scope_path}:{source_payload_ref}".encode("utf-8")).hexdigest()
    return f"cpx-{digest[:40]}"


def _parse_blob_ref(blob_ref: str) -> tuple[str, str]:
    prefix = "blob://"
    if not isinstance(blob_ref, str) or not blob_ref.startswith(prefix):
        raise ValueError("payload blob_ref must start with blob://")
    rest = blob_ref[len(prefix):]
    parts = rest.split("/")
    if len(parts) != 2:
        raise ValueError("payload blob_ref must be blob://{namespace}/{blob_id}")
    return parts[0], parts[1]

"""Workspace — /ro + /rw ACL + tree-structured scopes.

Two zones:
  - /ro/ = Cortex-managed (scopes, config, skills, knowledge).
           Agent can READ but NOT write. All scope operations (create, step,
           archive) are Cortex system operations and use _sys_* methods.
  - /rw/ = Agent's free space (scratch files, agent-managed data).
           Agent can read AND write freely via public write/append_line/etc.

Scope tree:
  - Active scopes live at /ro/active/{scope_id}/
  - Archived scopes live at /ro/scopes/{scope_id}/
  - Child scopes nest inside parent's steps/ as directories
  - LIFO: children close before parents → root archive moves tree atomically

class PayloadReadError(RuntimeError):

def _payload_filename(payload_ref: str) -> str:
    digest = hashlib.sha256(payload_ref.encode("utf-8")).hexdigest()
    return f"{digest}.json"


def _scope_id_from_scope_dir(name: str) -> str:
    if "_scope_" in name:
        return name.split("_scope_", 1)[1]
    return name


def _payload_scope_ids(scope_path: str) -> tuple[str, str]:
    parts = [part for part in scope_path.strip("/").split("/") if part]
    if len(parts) < 3 or parts[0] != "ro" or parts[1] not in {"active", "scopes"}:
        raise ValueError(f"cannot derive payload scope ids from {scope_path!r}")

    root_scope_id = parts[2]
    scope_id = root_scope_id
    for index, part in enumerate(parts):
        if part == "steps" and index + 1 < len(parts):
            scope_id = _scope_id_from_scope_dir(parts[index + 1])
    return root_scope_id, scope_id

class Workspace:
    def __init__(
        self,
        file_authority: Any,
        agent_id: str,
        hooks: CortexHooks | None = None,
        payload_blob_client: PayloadBlobClient | None = None,
        payload_blob_policy: BlobPayloadPolicy | None = None,
        clock: Callable[[], float] | None = None,
        operational_store: Any | None = None,
    ) -> None:
        validate_agent_id(agent_id)
        self._files = file_authority
        self._agent_id = agent_id
        self._hooks = hooks
        self._payload_blob_client = payload_blob_client
        self._payload_blob_policy = payload_blob_policy or BlobPayloadPolicy()
        self._clock = clock or time.time
        self._operational_store = operational_store

    def _now(self) -> float:
        return float(self._clock())

    async def _emit(self, hook_list_name: str, *args: object) -> None:
        if self._hooks is None:
            return
        cbs = getattr(self._hooks, hook_list_name, None)
        if not cbs:
            return
        for cb in cbs:
            try:
                if inspect.iscoroutinefunction(cb):
                    await cb(*args)
                else:
                    cb(*args)
            except Exception as e:
                warnings.warn(
                    f"Cortex hook {hook_list_name!r} failed: {e}",
                    UserWarning,
                    stacklevel=2,
                )

    @property
    def agent_id(self) -> str:
        return self._agent_id

    @property
    def operational_store(self) -> Any | None:
        return self._operational_store

    def _require_operational_store(self) -> Any:
        try:
            operational_store = self._operational_store
        except AttributeError:
            operational_store = None
        if operational_store is None:
            raise RuntimeError("Workspace operational_store is required for scope lifecycle transitions")
        return operational_store

    def _update_payload_manifest_error(
        self,
        *,
        payload_ref: str,
        status: str,
        code: str,
        message: str,
    ) -> None:
        operational_store = self._require_operational_store()
        manifest = operational_store.get_payload_manifest(payload_ref=payload_ref)
        if manifest is None:
            return
        operational_store.put_payload_manifest(
            payload_ref=manifest["payload_ref"],
            source_payload_ref=manifest.get("source_payload_ref"),
            root_scope_id=manifest["root_scope_id"],
            scope_id=manifest.get("scope_id"),
            step_ref=manifest.get("step_ref"),
            blob_ref=manifest.get("blob_ref"),
            mime_type=manifest["mime_type"],
            size_bytes=int(manifest["size_bytes"]),
            sha256=manifest["sha256"],
            status=status,
            retention_class=manifest["retention_class"],
            error={"code": code, "message": message},
        )

    def _payload_read_error(
        self,
        *,
        payload_ref: str,
        code: str,
        status: str,
        message: str,
    ) -> "PayloadReadError":
        self._update_payload_manifest_error(
            payload_ref=payload_ref,
            status=status,
            code=code,
            message=message,
        )
        return PayloadReadError(code=code, payload_ref=payload_ref, message=message)

    def _validate_write(self, logical: str) -> None:
        """Agent ACL: only /rw/ is writable by the agent."""
        if not logical.strip().startswith("/rw/"):
            raise PermissionError(f"agent cannot write {logical!r} (only /rw/)")

    # ── System I/O (Cortex-managed, bypasses agent ACL) ──

    async def _sys_write(self, path: str, content: str) -> None:
        """Cortex system write — no agent ACL check."""
        await self._files.write_text(path, content)

    async def _sys_write_json(self, path: str, obj: dict) -> None:
        await self._files.write_json(path, obj)

    async def _sys_append_line(self, path: str, line: str) -> None:
        await self._files.append_line(path, line)

    # ── Agent I/O (subject to /rw/ ACL) ──

    async def read(self, path: str) -> str:
        return await self._files.read_text(path)

    async def read_bytes(self, path: str) -> bytes:
        return await self._files.read_bytes(path)

    async def read_tree_bytes(

        LogicalFS: callers work in logical workspace paths, while Workspace
        keeps object-store keys and persistence adapters hidden below this

class RecordingRunner:
    def __init__(
        self,
        stdout: bytes = b"runner",
        on_execute: Callable[[SandboxExecSpec], None] | None = None,
    ) -> None:
        self.stdout = stdout
        self.on_execute = on_execute
        self.seen: list[SandboxExecSpec] = []
        self.seen_capability_bins: list[list[str]] = []

    async def execute(self, spec: SandboxExecSpec) -> SandboxExecResult:
        self.seen.append(spec)
        if spec.mount is not None:
            bin_dir = Path(spec.mount.source_root) / "bin"

            self.on_execute(spec)
        return SandboxExecResult(
            stdout=self.stdout,
            stderr=b"",
            exit_code=0,
            duration_s=0.01,
        )

def _server_for_device_and_blob(*, screenshot_bytes: bytes = b"", file_bytes: bytes = b""):
    state: dict[str, object] = {
        "uploads": [],
        "parts": [],
        "device_bodies": [],
    }

    class Handler(BaseHTTPRequestHandler):
        def log_message(self, *_args) -> None:
            return

        def _read_json(self) -> dict:
            length = int(self.headers.get("content-length", "0") or "0")
            raw = self.rfile.read(length) if length else b""
            if not raw:
                return {}
            return json.loads(raw.decode("utf-8"))

        def _write_json(self, payload: dict) -> None:
            raw = json.dumps(payload).encode("utf-8")
            self.send_response(200)
            self.send_header("Content-Type", "application/json")
            self.send_header("Content-Length", str(len(raw)))
            self.end_headers()
            self.wfile.write(raw)

        def do_POST(self) -> None:
            body = self._read_json()
            if self.path.endswith("/hd/desktop/screenshot"):
                state["device_bodies"].append(body)
                self._write_json(
                    {
                        "success": True,
                        "format": "jpeg",
                        "width": 640,
                        "height": 360,
                        "screenshot": base64.b64encode(screenshot_bytes).decode("ascii"),
                    }
                )
                return
            if self.path.endswith("/hd/file/pull"):
                state["device_bodies"].append(body)
                self._write_json(
                    {
                        "success": True,
                        "data": base64.b64encode(file_bytes).decode("ascii"),
                        "is_base64": True,
                        "size": len(file_bytes),
                    }
                )
                return
            if self.path == "/v1/blobs/uploads":
                upload = body
                state["uploads"].append(upload)
                self._write_json({"session_id": "upload-" + "a" * 32, "status": "created", "parts": []})
                return
            if self.path.endswith("/complete"):
                parts = state["parts"]
                data = b"".join(part["data"] for part in parts) if parts else b""
                upload = state["uploads"][-1]
                blob_id = upload["blob_id"]
                self._write_json(
                    {
                        "blob": {
                            "blob_ref": f"blob://runtime-artifact/{blob_id}",
                            "namespace": "runtime-artifact",
                            "blob_id": blob_id,
                            "mime_type": upload["mime_type"],
                            "size": len(data),
                            "sha256": hashlib.sha256(data).hexdigest(),
                        }
                    }
                )
                return
            self.send_error(404)

        def do_PUT(self) -> None:
            length = int(self.headers.get("content-length", "0") or "0")
            data = self.rfile.read(length) if length else b""
            state["parts"].append({"path": self.path, "data": data})
            self._write_json({"part_number": 1, "status": "receiving"})

    server = ThreadingHTTPServer(("127.0.0.1", 0), Handler)
    thread = threading.Thread(target=server.serve_forever, daemon=True)
    thread.start()
    return server, thread, state
