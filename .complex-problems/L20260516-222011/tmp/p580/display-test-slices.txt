    65	def test_display_fetches_blob_service_image_as_mcp_content(monkeypatch):
    66	    calls = []
    67	
    68	    class FakeClient:
    69	        def __init__(self, *args, **kwargs):
    70	            pass
    71	
    72	        def __enter__(self):
    73	            return self
    74	
    75	        def __exit__(self, *_exc):
    76	            return False
    77	
    78	        def get(self, url, headers):
    79	            calls.append({"url": url, "headers": headers})
    80	            return _FakeResponse(content=b"\x89PNG\r\n", content_type="image/png")
    81	
    82	    monkeypatch.setattr(tool_handlers, "internal_sync_client", lambda *a, **kw: FakeClient())
    83

    90	    item = result["_mcp_content"][0]
    91	    assert item["type"] == "image"
    92	    assert item["mimeType"] == "image/png"
    93	    assert base64.b64decode(item["data"]) == b"\x89PNG\r\n"
    94	
    95	    wrapped = _ok("scope-1", "round-1", "display:0", "display", result)
    96	    public = json.loads(wrapped["content"])
    97	    public_item = public["_mcp_content"][0]

    99	    assert "data" not in public_item
   100	    assert item["data"] not in wrapped["content"]
