    18	def _tool_output_size_label(size: Any) -> str:
    19	    if not isinstance(size, int) or size < 0:
    20	        return ""
    21	    value = float(size)
    22	    for unit in ("B", "KB", "MB", "GB"):
    23	        if value < 1024:
    24	            return f"{int(value)}B" if unit == "B" else f"{value:.1f}{unit}"
    25	        value /= 1024
    26	    return f"{value:.1f}TB"

    29	def _tool_output_artifact_line(artifact: Dict[str, Any]) -> str:
    30	    uri = str(artifact.get("uri") or "").strip()
    31	    modality = str(artifact.get("modality") or "resource").strip() or "resource"
    32	    name = str(artifact.get("name") or "").strip()
    33	    mime_type = str(artifact.get("mime_type") or "").strip()
    34	    summary = str(artifact.get("summary") or "").strip()
    35	    size = _tool_output_size_label(artifact.get("size_bytes"))
    36	    label = name or uri or "artifact"
    37	    details = ", ".join(part for part in (modality, mime_type, size) if part)
    38	    prefix = f"[Artifact: {label}"
    39	    if details:
    40	        prefix += f" ({details})"
    41	    prefix += "]"
    42	
    43	    access = artifact.get("access") if isinstance(artifact.get("access"), dict) else {}
    44	    display = access.get("display") if isinstance(access.get("display"), dict) else {}
    45	    display_args = display.get("args") if isinstance(display.get("args"), dict) else {}
    46	    display_url = display_args.get("file_url") or uri
    47	
    48	    parts = [prefix]
    49	    if summary:
    50	        parts.append(summary)
    51	    if display_url:
    52	        parts.append(f'Access: display(file_url="{display_url}")')
    53	    return " — ".join(parts)

    56	def _parse_tool_output_v1(raw: Dict[str, Any]) -> Dict[str, Any]:
    57	    text_parts: list[str] = []
    58	    text = raw.get("text")
    59	    if text:
    60	        text_parts.append(str(text))
    61	
    62	    artifacts = raw.get("artifacts") if isinstance(raw.get("artifacts"), list) else []
    63	    for artifact in artifacts:
    64	        if isinstance(artifact, dict):
    65	            text_parts.append(_tool_output_artifact_line(artifact))
    66	
    67	    events = raw.get("events") if isinstance(raw.get("events"), list) else []
    68	    for event in events:
    69	        if not isinstance(event, dict):
    70	            continue
    71	        event_type = str(event.get("type") or "").strip()
    72	        summary = str(event.get("summary") or "").strip()
    73	        if event_type or summary:
    74	            label = f"[Event: {event_type or 'event'}]"
    75	            text_parts.append(f"{label} {summary}".strip())
    76	
    77	    error = raw.get("error")
    78	    if error and not raw.get("ok", True):
    79	        text_parts.append(f"[Error] {error}")
    80	
    81	    return {
    82	        "text": "\n".join(text_parts),
    83	        "files_created": [],
    84	        "display_files": [],
    85	        "artifacts": artifacts,
    86	        "events": events,

    90	def parse_tool_result(raw: Any) -> Dict[str, Any]:
    91	    """
    92	    解析工具原始结果为三要素格式。
    93	
    94	    支持的输入格式：
    95	    1. 统一格式: {text, files_created, display_files}
    96	    2. MCP 格式: {content: [{type, text/data}]}
    97	    3. 纯文本: str
    98	
    99	    Returns:
   100	        {"text": "...", "files_created": [...], "display_files": [...]}
   101	    """
   102	    if raw is None:
   103	        return {"text": "", "files_created": [], "display_files": []}
   104	
   105	    if isinstance(raw, str):
   106	        stripped = raw.strip()
   107	        if stripped.startswith("{") or stripped.startswith("["):
   108	            try:
   109	                decoded = json.loads(stripped)
   110	            except json.JSONDecodeError:
   111	                decoded = None
   112	            if isinstance(decoded, (dict, list)):
   113	                return parse_tool_result(decoded)
   114	        return {"text": raw, "files_created": [], "display_files": []}
   115	
   116	    if not isinstance(raw, dict):
   117	        return {"text": str(raw), "files_created": [], "display_files": []}
   118	
   119	    if raw.get("version") == _TOOL_STEP_PAYLOAD_VERSION:
   120	        return parse_tool_result(raw.get("llm_content", ""))
   121	
   122	    if raw.get("version") == "tool-output.v1":
   123	        return _parse_tool_output_v1(raw)
   124	
   125	    # 1. 统一格式（三要素）
   126	    if "display_files" in raw or "files_created" in raw:
   127	        text_parts = []
   128	        if raw.get("text"):
   129	            text_parts.append(str(raw["text"]))
   130	        # 把 files_created 的 URL 写入文本
   131	        files_created = raw.get("files_created") or []
   132	        if files_created and isinstance(files_created, list):
   133	            urls = [f.get("url") for f in files_created if isinstance(f, dict) and f.get("url")]
   134	            if urls:
   135	                text_parts.append(f"file_url: {urls[0]}" if len(urls) == 1 else f"file_urls: {', '.join(urls)}")
   136	        return {
   137	            "text": "\n".join(text_parts),
   138	            "files_created": files_created,
   139	            "display_files": raw.get("display_files") or [],
   140	        }
   141	
   142	    # 2. MCP 格式 content 数组
   143	    content = raw.get("content") or raw.get("_mcp_content")
   144	    if isinstance(content, list):
   145	        text_parts = []
   146	        display_files = []
   147	        for item in content:
   148	            if not isinstance(item, dict):
   149	                continue
   150	            t = item.get("type", "")
   151	            if t == "text":
   152	                text_parts.append(item.get("text", ""))
   153	            elif t == "image":
   154	                # 支持 URL 形式或直接的 data(base64) 形式
   155	                if item.get("url"):
   156	                    display_files.append({
   157	                        "url": item["url"],
   158	                        "filename": "",
   159	                        "modality": "image",
   160	                    })
   161	                elif item.get("data"):
   162	                    mime = item.get("mimeType", "image/png")
   163	                    display_files.append({
   164	                        "url": f"data:{mime};base64,{item['data']}",
   165	                        "filename": "",
   166	                        "modality": "image",
   167	                    })
   168	            elif t == "image_ref":
   169	                ref_url = item.get("file_url") or item.get("url")
   170	                if ref_url:
   171	                    display_files.append({
   172	                        "url": ref_url,
   173	                        "filename": item.get("filename", ""),
   174	                        "modality": "image",
   175	                        "mime_type": item.get("mimeType") or item.get("mime_type", ""),
   176	                        "size": item.get("size", 0),
   177	                    })
   178	            elif t == "resource":
   179	                if item.get("url"):
   180	                    display_files.append({
   181	                        "url": item["url"],
   182	                        "filename": "",
   183	                        "modality": "resource",
   184	                    })
   185	        return {
   186	            "text": "\n".join(text_parts),
   187	            "files_created": [],
   188	            "display_files": display_files,
   189	        }
   190	
   191	    # 4. 其他 dict：提取 text 字段或序列化
   192	    if raw.get("text"):
   193	        return {"text": str(raw["text"]), "files_created": [], "display_files": []}
   194	
   195	    # 5. Unknown dict fallback: diagnostic text only, bounded so unexpected
   196	    # payload shapes cannot flood LLM context.
   197	    def safe_default(obj):
   198	        return f"<Unserializable {type(obj).__name__}>"
   199	    try:
   200	        serialized = json.dumps(raw, ensure_ascii=False, default=safe_default)
   201	    except Exception as e:
   202	        serialized = f"[StepResultProjection Parse Error] Failed to serialize object: {str(e)}"
   203	    if len(serialized) > _UNKNOWN_DICT_TEXT_LIMIT:
   204	        serialized = (
   205	            serialized[:_UNKNOWN_DICT_TEXT_LIMIT]
   206	            + f"\n\n... [Truncated unknown tool result JSON length > {_UNKNOWN_DICT_TEXT_LIMIT}]"
   207	        )
   208	    serialized = "[Unknown tool result JSON]\n" + serialized
   209	    return {"text": serialized, "files_created": [], "display_files": []}

   212	def _format_mcp_content(
   213	    result_data: Dict[str, Any],
   214	    *,
   215	    include_display: bool,
   216	    max_text_len: Optional[int] = None,
   217	    empty_text: str = "OK",
   218	) -> str:
   219	    converted = []
   220	
   221	    text = result_data.get("text", "")
   222	    if max_text_len is not None and len(text) > max_text_len:
   223	        text = text[:max_text_len] + f"\n\n... [Truncated due to context budget length > {max_text_len}]"
   224	
   225	    if text:
   226	        converted.append({"type": "text", "text": text})
   227	
   228	    if include_display:
   229	        display_files = result_data.get("display_files", [])
   230	        for f in display_files:
   231	            url = f.get("url", "")
   232	            if not url:
   233	                continue
   234	            if url.startswith("data:"):
   235	                # Data URL
   236	                parts = url.split(",", 1)
   237	                mime = "image/png"
   238	                if len(parts) >= 1 and ";" in parts[0]:
   239	                    mime = parts[0].split(";")[0].replace("data:", "")
   240	                data_b64 = parts[1] if len(parts) > 1 else ""
   241	
   242	                converted.append({
   243	                    "type": "image",
   244	                    "data": data_b64,
   245	                    "mimeType": mime,
   246	                })
   247	            else:
   248	                mime = str(f.get("mime_type") or f.get("mimeType") or "")
   249	                modality = str(f.get("modality") or "")
   250	                if url.startswith("blob://") and (modality == "image" or mime.startswith("image/")):
   251	                    converted.append({
   252	                        "type": "image_ref",
   253	                        "file_url": url,
   254	                        "mimeType": mime or "image/png",
   255	                        "size": f.get("size") or f.get("size_bytes") or 0,
   256	                    })
   257	                else:
   258	                    # Normal URL: represent unresolved display media as text.
   259	                    converted.append({"type": "text", "text": f"[Image: {url}]"})
   260	                
   261	    if not converted:
   262	        converted.append({"type": "text", "text": empty_text})
   263	
   264	    return json.dumps({"_mcp_content": converted}, ensure_ascii=False)

   308	    """Generate a concise text preview of a tool result."""
