    if "mpeg" in mime or "mp3" in mime:
        return "mp3"
    ext = file_url.split("?", 1)[0].rsplit(".", 1)[-1].lower() if "." in file_url else ""
    return {{"wav": "wav", "mp3": "mp3", "flac": "flac", "m4a": "mp4", "mp4": "mp4", "ogg": "ogg", "webm": "webm"}}.get(ext, "mp3")
