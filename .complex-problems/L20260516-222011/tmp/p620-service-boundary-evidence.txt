     1	"""Small filesystem helpers for sandboxd internals."""
     2	
     3	from __future__ import annotations
     4	
     5	import re
     6	from pathlib import Path
     7	from typing import Mapping
     8	
     9	FileStats = dict[str, tuple[int, int]]
    10	
    11	
    12	def file_stats(root: Path) -> FileStats:
    13	    """Return relative posix path -> (mtime_ns, size) for regular files."""
    14	    out: FileStats = {}
    15	    if not root.is_dir():
    16	        return out
    17	    for path in root.rglob("*"):
    18	        if path.is_file():
    19	            stat = path.stat()
    20	            rel = path.relative_to(root).as_posix()
    21	            out[rel] = (stat.st_mtime_ns, stat.st_size)
    22	    return out
    23	
    24	
    25	def changed_relative_paths(before: FileStats, after: FileStats) -> list[str]:
    26	    """Return created, modified, and removed relative paths."""
    27	    changed: list[str] = []
    28	    for rel in before:
    29	        if rel not in after:
    30	            changed.append(rel)
    31	    for rel, stats in after.items():
    32	        prev = before.get(rel)
    33	        if prev is None or prev != stats:
    34	            changed.append(rel)
    35	    return sorted(set(changed))
    36	
    37	
    38	def safe_path_component(value: str | None, default: str) -> str:
    39	    raw = (value or default).strip() or default
    40	    safe = re.sub(r"[^A-Za-z0-9_.-]+", "-", raw).strip(".-")
    41	    return safe or default
    42	
    43	
    44	def resolve_cwd_under_root(root: Path, cwd: str | None) -> Path:
    45	    root_resolved = root.resolve()
    46	    if cwd is None or cwd == "" or cwd == ".":
    47	        return root_resolved
    48	    candidate = (root_resolved / cwd).resolve()
    49	    try:
    50	        candidate.relative_to(root_resolved)
    51	    except ValueError as exc:
    52	        raise ValueError(f"cwd must be relative to sandbox root: {cwd!r}") from exc
    53	    return candidate
    54	
    55	
    56	def is_keep_placeholder(path: str, data: bytes) -> bool:
    57	    return path.endswith(".keep") and len(data) == 0
    58	
    59	
    60	def sanitize_paths(text: str, replacements: Mapping[str, str]) -> str:
    61	    """Replace actual filesystem paths with stable display paths."""
    62	    out = text
    63	    for actual, stable in replacements.items():
    64	        out = out.replace(actual, stable)
    65	    return out

     1	"""Linux mount namespace helpers for sandboxd internals."""
     2	
     3	from __future__ import annotations
     4	
     5	import os
     6	import shlex
     7	import shutil
     8	from pathlib import Path
     9	
    10	
    11	def mount_namespace_available() -> bool:
    12	    """Return whether the current process can create private bind mounts."""
    13	    return (
    14	        os.name == "posix"
    15	        and hasattr(os, "geteuid")
    16	        and os.geteuid() == 0
    17	        and shutil.which("unshare") is not None
    18	        and shutil.which("mount") is not None
    19	    )
    20	
    21	
    22	def build_bind_mount_command(
    23	    *,
    24	    command: str,
    25	    source_root: str | Path,
    26	    mount_point: str,
    27	    stable_cwd: str,
    28	) -> str:
    29	    """Build a shell command that bind-mounts source_root at mount_point.
    30	
    31	    The caller owns all policy: what root to mount, where to mount it, and which
    32	    stable cwd/command should run inside that private namespace.
    33	    """
    34	    source = shlex.quote(str(Path(source_root).resolve()))
    35	    mount = shlex.quote(mount_point)
    36	    cwd = shlex.quote(stable_cwd)
    37	    command_arg = shlex.quote(command)
    38	    inner = (
    39	        "set -e; "
    40	        f"mkdir -p {mount}; "
    41	        f"mount --bind {source} {mount}; "
    42	        f"cd {cwd}; "
    43	        f"exec /bin/sh -c {command_arg}"
    44	    )
    45	    return f"unshare --mount --propagation private /bin/bash -lc {shlex.quote(inner)}"

    76	            stderr_b = f"Command timed out after {spec.timeout}s".encode()
    77	            exit_code = -1
    78	
    79	        return ProcessRunResult(
    80	            stdout=stdout_b,
    81	            stderr=stderr_b,
    82	            exit_code=exit_code,
