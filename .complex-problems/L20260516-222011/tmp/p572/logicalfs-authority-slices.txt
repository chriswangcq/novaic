## LogicalFS authority path/key semantics
     1	"""Generic live LogicalFS file authority.
     2	
     3	This module owns realtime logical file operations over /ro and /rw. It is
     4	business-agnostic: callers provide explicit namespace/layout configuration and
     5	a generic object-store adapter. Cortex semantics, agents, scopes, Blob Service,
     6	and process execution live outside this boundary.
     7	"""
     8	
     9	from __future__ import annotations
    10	
    11	import asyncio
    12	import json
    13	from dataclasses import dataclass
    14	from typing import Iterable, Protocol, Sequence
    15	
    16	from logicalfs.contracts import LogicalFSDirectoryEntry, LogicalObjectInfo

    19	class LogicalObjectStore(Protocol):
    20	    async def put(self, key: str, data: bytes, content_type: str = "") -> None: ...
    21	
    22	    async def get(self, key: str) -> bytes: ...
    23	
    24	    async def exists(self, key: str) -> bool: ...
    25	
    26	    async def delete(self, key: str) -> None: ...
    27	
    28	    async def list_objects(self, prefix: str, delimiter: str = "/") -> Sequence[LogicalObjectInfo]: ...
    29	
    30	    async def list_recursive(self, prefix: str) -> Sequence[str]: ...
    31	
    32	    async def move_prefix(self, src_prefix: str, dst_prefix: str) -> int: ...
    33	
    34	
    35	@dataclass(frozen=True)
    36	class LogicalFileAuthorityLayout:
    37	    owner_prefix: str
    38	    ro_prefix: str = "ro"
    39	    rw_prefix: str = "rw"
    40	
    41	    def object_prefix(self) -> str:
    42	        return self.owner_prefix.strip("/")
    43	
    44	
    45	def _normalize_logical_path(path: str) -> str:
    46	    if not isinstance(path, str):
    47	        raise ValueError("path must be a string")
    48	    p = path.strip()
    49	    if "\x00" in p:
    50	        raise ValueError("path must not contain NUL")
    51	    if not p.startswith("/"):
    52	        p = "/" + p
    53	    while "//" in p:
    54	        p = p.replace("//", "/")
    55	    for segment in p.split("/"):
    56	        if segment == "..":
    57	            raise ValueError("path must not contain '..'")
    58	    if p in {"/ro", "/rw"}:
    59	        return p + "/"
    60	    if p.startswith("/ro/") or p.startswith("/rw/"):
    61	        return p
    62	    raise ValueError(f"path must start with /ro/ or /rw/: {path!r}")
    63	
    64	
    65	def logical_to_object_key(layout: LogicalFileAuthorityLayout, path: str) -> str:
    66	    logical = _normalize_logical_path(path)
    67	    prefix = layout.object_prefix()
    68	    base = f"{prefix}/" if prefix else ""
    69	    ro = layout.ro_prefix.strip("/")
    70	    rw = layout.rw_prefix.strip("/")
    71	    if logical.startswith("/ro/"):
    72	        rel = logical.removeprefix("/ro/").lstrip("/")
    73	        return f"{base}{ro}/{rel}"
    74	    if logical.startswith("/rw/"):
    75	        rel = logical.removeprefix("/rw/").lstrip("/")
    76	        return f"{base}{rw}/{rel}"
    77	    raise ValueError(f"path must start with /ro/ or /rw/: {path!r}")
    78

    80	class StoreBackedLogicalFileAuthority:
    81	    """Live /ro and /rw authority backed by a generic object store."""
    82	
    83	    def __init__(self, store: LogicalObjectStore, layout: LogicalFileAuthorityLayout) -> None:
    84	        self._store = store
    85	        self._layout = layout
    86	
    87	    @property
    88	    def layout(self) -> LogicalFileAuthorityLayout:
    89	        return self._layout
    90	
    91	    def key(self, path: str) -> str:
    92	        return logical_to_object_key(self._layout, path)
    93	
    94	    def _prefix_key(self, path: str) -> str:
    95	        key = self.key(path)
    96	        return key if key.endswith("/") else key + "/"
    97	
    98	    async def write_bytes(self, path: str, data: bytes, content_type: str = "") -> None:
    99	        await self._store.put(self.key(path), data, content_type)
   100	
   101	    async def write_text(self, path: str, content: str) -> None:
   102	        await self.write_bytes(path, content.encode("utf-8"))
   103	
   104	    async def write_json(self, path: str, obj: dict) -> None:
   105	        await self.write_bytes(
   106	            path,
   107	            json.dumps(obj, ensure_ascii=False, indent=2).encode("utf-8"),
   108	            "application/json",
   109	        )
   110	
   111	    async def read_bytes(self, path: str) -> bytes:
   112	        return await self._store.get(self.key(path))
   113	
   114	    async def read_text(self, path: str, encoding: str = "utf-8") -> str:
   115	        return (await self.read_bytes(path)).decode(encoding)
   116	
   117	    async def exists(self, path: str) -> bool:
   118	        return await self._store.exists(self.key(path))
   119	
   120	    async def delete(self, path: str) -> None:
   121	        await self._store.delete(self.key(path))
   122	
   123	    async def append_line(self, path: str, line: str) -> None:
   124	        await self.append_lines(path, line.rstrip("\n") + "\n")
   125	
   126	    async def append_lines(self, path: str, lines: str) -> None:
   127	        key = self.key(path)
   128	        try:
   129	            prev = await self._store.get(key)
   130	        except KeyError:
   131	            prev = b""
   132	        await self._store.put(key, prev + lines.encode("utf-8"))
   133	
   134	    async def list_tree_bytes(
   135	        self,
   136	        path: str,
   137	        *,
   138	        concurrency: int = 16,
   139	    ) -> list[tuple[str, bytes]]:
   140	        store_prefix = self._prefix_key(path)
   141	        keys = list(await self._store.list_recursive(store_prefix))
   142	
   143	        async def read_one(key: str) -> tuple[str, bytes] | None:
   144	            if not key.startswith(store_prefix):
   145	                return None
   146	            rel = key[len(store_prefix):].lstrip("/")
   147	            if not rel:
   148	                return None
   149	            return rel, await self._store.get(key)
   150	
   151	        if concurrency <= 1:
   152	            results = [await read_one(key) for key in keys]
   153	        else:
   154	            sem = asyncio.Semaphore(max(1, int(concurrency)))
   155	
   156	            async def limited_read(key: str) -> tuple[str, bytes] | None:
   157	                async with sem:
   158	                    return await read_one(key)
   159	
   160	            results = await asyncio.gather(*(limited_read(key) for key in keys))
   161	        return [item for item in results if item is not None]
   162	
   163	    async def list_dir(self, path: str) -> list[LogicalFSDirectoryEntry]:
   164	        logical_base = _normalize_logical_path(path)
   165	        if not logical_base.endswith("/"):
   166	            logical_base += "/"
   167	        store_prefix = self._prefix_key(logical_base)
   168	        infos = await self._store.list_objects(store_prefix)
   169	        entries: list[LogicalFSDirectoryEntry] = []
   170	        for item in infos:
   171	            name = item.key.rstrip("/").split("/")[-1]
   172	            logical = logical_base + name
   173	            if item.is_prefix:
   174	                logical = logical.rstrip("/") + "/"
   175	            entries.append(
   176	                LogicalFSDirectoryEntry(
   177	                    name=name,
   178	                    path=logical,
   179	                    is_dir=item.is_prefix,
   180	                    size=item.size,
   181	                )
   182	            )
   183	        return entries
   184	
   185	    async def move_tree(self, src_path: str, dst_path: str) -> int:
   186	        return await self._store.move_prefix(self._prefix_key(src_path), self._prefix_key(dst_path))
   187	
   188	    async def list_object_keys(self, path: str) -> list[str]:
   189	        return list(await self._store.list_recursive(self._prefix_key(path)))
   190	
   191	    async def initialize_layout(self, logical_dirs: Iterable[str]) -> None:
   192	        for logical_dir in logical_dirs:
   193	            directory = _normalize_logical_path(logical_dir).rstrip("/")
   194	            await self.write_bytes(f"{directory}/.keep", b"")
   195

   202	]

## Local materialization and patch observation
     1	"""Local LogicalFS materialization/diff provider."""
     2	
     3	from __future__ import annotations
     4	
     5	from pathlib import Path
     6	from typing import Mapping
     7	
     8	from logicalfs.contracts import (
     9	    LogicalFSEnv,
    10	    LogicalFSLayout,
    11	    LogicalFSPatch,

    18	FileStats = dict[str, tuple[int, int]]
    19	
    20	
    21	def _file_stats(root: Path) -> FileStats:
    22	    out: FileStats = {}
    23	    if not root.is_dir():
    24	        return out
    25	    for path in root.rglob("*"):
    26	        if path.is_file():
    27	            stat = path.stat()
    28	            rel = path.relative_to(root).as_posix()
    29	            out[rel] = (stat.st_mtime_ns, stat.st_size)
    30	    return out
    31	
    32	
    33	def _changed_relative_paths(before: Mapping[str, tuple[int, int]], after: Mapping[str, tuple[int, int]]) -> list[str]:
    34	    changed: list[str] = []
    35	    for rel in before:
    36	        if rel not in after:
    37	            changed.append(rel)
    38	    for rel, stats in after.items():
    39	        prev = before.get(rel)
    40	        if prev is None or prev != stats:
    41	            changed.append(rel)
    42	    return sorted(set(changed))
    43	
    44	
    45	def _resolve_cwd_under_root(root: Path, cwd: str | None) -> Path:
    46	    root_resolved = root.resolve()
    47	    if cwd is None or cwd == "" or cwd == ".":
    48	        return root_resolved
    49	    candidate = (root_resolved / cwd).resolve()
    50	    try:
    51	        candidate.relative_to(root_resolved)
    52	    except ValueError as exc:
    53	        raise ValueError(f"cwd must be relative to logical rw root: {cwd!r}") from exc
    54	    return candidate
    55	
    56	
    57	def _stable_path_for_actual(path: Path, *, view_root: Path, ro: Path, rw: Path, stable: StableLogicalPaths) -> str:
    58	    resolved = path.resolve()
    59	    for actual_root, stable_root in [
    60	        (rw.resolve(), stable.rw),
    61	        (ro.resolve(), stable.ro),
    62	        (view_root.resolve(), stable.root),
    63	    ]:
    64	        try:
    65	            rel = resolved.relative_to(actual_root)
    66	        except ValueError:
    67	            continue
    68	        if rel.as_posix() == ".":
    69	            return stable_root
    70	        suffix = rel.as_posix()
    71	        return stable_root if not suffix else f"{stable_root}/{suffix}"
    72	    return str(resolved)
    73	
    74	
    75	def _logical_rw_path(rel: str, stable: StableLogicalPaths) -> str:
    76	    return f"{stable.rw}/{rel}" if rel else stable.rw
    77	
    78	
    79	def sanitize_paths(text: str, replacements: Mapping[str, str]) -> str:
    80	    out = text
    81	    for actual, stable in replacements.items():
    82	        out = out.replace(actual, stable)
    83	    return out
    84	
    85	
    86	class LocalLogicalFSProvider:
    87	    """Materialize explicit snapshots into local RO/RW views and observe patches."""
    88	
    89	    def materialize(
    90	        self,
    91	        *,
    92	        snapshot: LogicalFSSnapshot,
    93	        root: Path,
    94	        layout: LogicalFSLayout,
    95	        env: LogicalFSEnv,
    96	        cwd: str | None = None,
    97	    ) -> LogicalFSView:
    98	        ro = root / "ro"
    99	        rw = root / "rw"
   100	        bin_dir = root / "bin"
   101	        ro.mkdir(parents=True)
   102	        rw.mkdir(parents=True)
   103	        bin_dir.mkdir(parents=True)
   104	
   105	        for item in snapshot.files:
   106	            clean = item.path.strip("/")
   107	            if not clean:
   108	                continue
   109	            base = rw if item.writable else ro
   110	            dest = base / clean
   111	            dest.parent.mkdir(parents=True, exist_ok=True)
   112	            dest.write_bytes(item.data)
   113	
   114	        for stable_path in [*layout.writable_env.values(), *layout.writable_dirs]:
   115	            if stable_path.startswith(layout.paths.rw + "/"):
   116	                rel = stable_path.removeprefix(layout.paths.rw + "/")
   117	                (rw / rel).mkdir(parents=True, exist_ok=True)
   118	
   119	        proc_cwd = _resolve_cwd_under_root(rw, cwd)
   120	        base_env = {
   121	            "PATH": f"{layout.paths.bin}:/usr/local/bin:/usr/bin:/bin",
   122	            "RO": layout.paths.ro,
   123	            "RW": layout.paths.rw,
   124	            "CORTEX_RO": layout.paths.ro,
   125	            "CORTEX_RW": layout.paths.rw,
   126	            "CORTEX_ROOT": layout.paths.root,
   127	            "CORTEX_FS_PROVIDER": "logicalfs",
   128	            "CORTEX_TRUE_STABLE_PATH_MOUNT": "1",
   129	            "HOME": layout.paths.rw,
   130	        }
   131	        base_env.update(dict(layout.writable_env))
   132	        base_env.update(dict(env.values))
   133	        return LogicalFSView(
   134	            root=root,
   135	            ro=ro,
   136	            rw=rw,
   137	            bin=bin_dir,
   138	            cwd=proc_cwd,
   139	            stable_cwd=_stable_path_for_actual(
   140	                proc_cwd,
   141	                view_root=root,
   142	                ro=ro,
   143	                rw=rw,
   144	                stable=layout.paths,
   145	            ),
   146	            env=base_env,
   147	            before_rw=_file_stats(rw),
   148	            paths=layout.paths,
   149	        )
   150	
   151	    def observe_patch(self, view: LogicalFSView) -> LogicalFSPatch:
   152	        after = _file_stats(view.rw)
   153	        upserts: list[LogicalFSPatchFile] = []
   154	        deletes: list[str] = []
   155	        for rel in _changed_relative_paths(view.before_rw, after):
   156	            logical = _logical_rw_path(rel, view.paths)
   157	            if rel not in after:
   158	                deletes.append(logical)
   159	            else:
   160	                upserts.append(LogicalFSPatchFile(path=logical, data=(view.rw / rel).read_bytes()))
   161	        return LogicalFSPatch(upserts=tuple(upserts), deletes=tuple(sorted(deletes)))
   162	
   163	    def sanitize_output(self, text: str, view: LogicalFSView) -> str:
   164	        return sanitize_paths(
   165	            text,
   166	            {
   167	                str(view.ro.resolve()): view.paths.ro,
   168	                str(view.rw.resolve()): view.paths.rw,
   169	                str(view.root.resolve()): view.paths.root,
   170	            },
   171	        )

    13	class MemoryObjectStore:
    14	    def __init__(self) -> None:
    15	        self.data: dict[str, bytes] = {}
    16	
    17	    async def put(self, key: str, data: bytes, content_type: str = "") -> None:
    18	        self.data[key] = data
    19	
    20	    async def get(self, key: str) -> bytes:
    21	        if key not in self.data:
    22	            raise KeyError(key)
    23	        return self.data[key]
    24	
    25	    async def exists(self, key: str) -> bool:
    26	        return key in self.data
    27	
    28	    async def delete(self, key: str) -> None:
    29	        self.data.pop(key, None)
    30	
    31	    async def list_objects(self, prefix: str, delimiter: str = "/") -> list[LogicalObjectInfo]:
    32	        if delimiter != "/":
    33	            raise NotImplementedError
    34	        seen: set[str] = set()
    35	        out: list[LogicalObjectInfo] = []
    36	        for key in sorted(self.data):
    37	            if not key.startswith(prefix):
    38	                continue
    39	            rest = key[len(prefix):]
    40	            if not rest:
    41	                continue
    42	            if "/" in rest:
    43	                child_prefix = prefix + rest.split("/", 1)[0] + "/"
    44	                if child_prefix not in seen:
    45	                    seen.add(child_prefix)
    46	                    out.append(LogicalObjectInfo(key=child_prefix, is_prefix=True))
    47	            else:
    48	                out.append(LogicalObjectInfo(key=key, size=len(self.data[key])))
    49	        return out
    50	
    51	    async def list_recursive(self, prefix: str) -> list[str]:
    52	        return sorted(key for key in self.data if key.startswith(prefix))
    53	
    54	    async def move_prefix(self, src_prefix: str, dst_prefix: str) -> int:
    55	        keys = [key for key in self.data if key.startswith(src_prefix)]
    56	        for key in keys:
    57	            self.data[dst_prefix + key[len(src_prefix):]] = self.data.pop(key)
    58	        return len(keys)
