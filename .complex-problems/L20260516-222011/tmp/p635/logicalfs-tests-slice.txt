    15	def test_materializes_snapshot_and_explicit_env_layout(tmp_path) -> None:
    16	    provider = LocalLogicalFSProvider()
    17	    view = provider.materialize(
    18	        snapshot=LogicalFSSnapshot(
    19	            files=(
    20	                LogicalFSFile(path="config/tool.json", data=b"ro"),
    21	                LogicalFSFile(path="scratch/existing.txt", data=b"rw", writable=True),
    22	            )
    23	        ),
    24	        root=tmp_path,
    25	        layout=LogicalFSLayout(
    26	            paths=StableLogicalPaths(),
    27	            writable_env={
    28	                "RW_SELF": "/cortex/rw/subagents/sub-1",
    29	                "RW_TMP": "/cortex/rw/tmp",
    30	            },
    31	            writable_dirs=("/cortex/rw/system",),
    32	        ),
    33	        env=LogicalFSEnv({"NOVAIC_TOKEN": "token-1", "NOVAIC_API": "http://cortex"}),
    34	        cwd="scratch",
    35	    )
    36	
    37	    assert (view.ro / "config/tool.json").read_bytes() == b"ro"
    38	    assert (view.rw / "scratch/existing.txt").read_bytes() == b"rw"
    39	    assert (view.rw / "subagents/sub-1").is_dir()
    40	    assert (view.rw / "tmp").is_dir()
    41	    assert (view.rw / "system").is_dir()
    42	    assert view.stable_cwd == "/cortex/rw/scratch"
    43	    assert view.env["RW_SELF"] == "/cortex/rw/subagents/sub-1"
    44	    assert view.env["NOVAIC_TOKEN"] == "token-1"

    61	def authority() -> StoreBackedLogicalFileAuthority:
    62	    return StoreBackedLogicalFileAuthority(
    63	        MemoryObjectStore(),
    64	        LogicalFileAuthorityLayout(owner_prefix="owners/o1"),
    65	    )

    68	def test_maps_logical_paths_from_explicit_layout() -> None:
    69	    layout = LogicalFileAuthorityLayout(owner_prefix="owners/o1", ro_prefix="readonly", rw_prefix="writeable")
    70	
    71	    assert logical_to_object_key(layout, "/ro/config/tool.json") == "owners/o1/readonly/config/tool.json"
    72	    assert logical_to_object_key(layout, "rw/scratch/a.txt") == "owners/o1/writeable/scratch/a.txt"

    75	@pytest.mark.asyncio
    76	async def test_read_write_append_exists_delete_roundtrip() -> None:
    77	    fs = authority()
    78	
    79	    await fs.write_text("/rw/scratch/a.txt", "one")
    80	    await fs.append_line("/rw/scratch/a.txt", "two")
    81	
    82	    assert await fs.exists("/rw/scratch/a.txt")
    83	    assert await fs.read_text("/rw/scratch/a.txt") == "onetwo\n"
    84	
    85	    await fs.delete("/rw/scratch/a.txt")
    86	
    87	    assert not await fs.exists("/rw/scratch/a.txt")
