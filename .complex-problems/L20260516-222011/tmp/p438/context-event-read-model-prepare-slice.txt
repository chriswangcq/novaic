    """Prepare LLM context from the authoritative ContextEvent stream."""

    def __init__(
        self,
        *,
        workspace: "Workspace",
        root_scope_path: str,
        config: CompactConfig | None = None,
        token_counter: TokenCounter | None = None,
        event_store: ContextEventStore | None = None,
    ) -> None:
        self._workspace = workspace
        self._root_scope_path = root_scope_path.rstrip("/")
        self._config = config or CompactConfig()
        self._token_counter = token_counter
        self._event_store = event_store or ContextEventStore(workspace)

    async def prepare(self) -> ContextEventPreparedContext:
        if not self._root_scope_path:
            return ContextEventPreparedContext(
                messages=[],
                stack=[],
                estimated_tokens=0,
                usage_ratio=0.0,
                applied_seq=0,
                projection=ContextProjectionSnapshot(),
            )

        events = await self._event_store.read_events(self._root_scope_path)
        if not events:
            raise ContextEventReadModelResetRequired(self._root_scope_path)
        projection = project_context_events(events)
        messages = budget_compact(
            projection.messages,
            self._config,
            counter=self._token_counter,
        )
        estimated_tokens = count_all_tokens(messages, self._token_counter)
        usage_ratio = (
            estimated_tokens / self._config.context_window
            if self._config.context_window > 0
            else 0.0
        )
        return ContextEventPreparedContext(
            messages=messages,
            stack=_top_first_stack(projection.stack),
            estimated_tokens=estimated_tokens,
            usage_ratio=usage_ratio,
            applied_seq=projection.applied_seq,
            projection=projection,
        )
