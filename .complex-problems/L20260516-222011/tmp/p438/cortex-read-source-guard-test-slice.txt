from __future__ import annotations

from pathlib import Path


API_SOURCE = Path(__file__).resolve().parents[1] / "novaic_cortex" / "api.py"
CORTEX_SOURCE_ROOT = Path(__file__).resolve().parents[1] / "novaic_cortex"


def _section(start_marker: str, end_marker: str) -> str:
    text = API_SOURCE.read_text()
    start = text.index(start_marker)
    end = text.index(end_marker, start)
    return text[start:end]


def _old_tree_walk_helper_name() -> str:
    return "_walk" + "_scope_tree"


def test_api_has_no_file_walk_active_stack_helper() -> None:
    text = API_SOURCE.read_text()

    assert "_collect_active_stack" not in text
    assert "resolve_active_scope_path" not in text


def test_prepare_for_llm_has_no_dfs_read_fallback() -> None:
    section = _section(
        '@app.post("/v1/context/prepare_for_llm")',
        "def _stack_top_id",
    )

    assert "ContextEventReadModel" in section
    assert "ContextEngine" not in section
    assert "prepare_messages_for_llm" not in section
    assert "read_context" not in section
    assert "StepTree" not in section
    assert "_collect_active_stack" not in section
