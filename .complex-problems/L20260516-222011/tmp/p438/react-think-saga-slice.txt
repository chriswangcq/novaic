"""
ReactThink Saga — ReAct Think phase

Flow:
1. Cortex assembles LLM context (ContextEvent read model + tools + transient warning)
2. Call LLM (with messages + tools from Step 1)
3. Save LLM response to context.jsonl (filter unknown tool_calls)
4. Decision:
   a. Has tool_calls → trigger ReactActions
   b. No tool_calls + first attempt → retry ReactThink
   c. No tool_calls + already retried → trigger wake_finalize
"""

import logging

from common.config import ServiceConfig
from task_queue.contracts.react_think import (
    ReactThinkDecisionInput,
    ReactThinkInput,
    build_check_skill_stack_payload,
    build_llm_call_payload,
    build_prepare_context_payload,
    build_save_response_payload,
    build_trigger_actions,
    build_trigger_finalize,
    build_trigger_retry_think,
    decide_react_think_route,
    known_tool_names_from_step_results,
    should_save_llm_response,
)

from ..saga import SagaDefinition
from . import register_saga_definition
from ..topics import TaskTopics, SagaTopics

logger = logging.getLogger(__name__)


def _get_known_tool_names(ctx):
    """Get valid tool names from the explicit prepare_context step result."""
    return known_tool_names_from_step_results((ctx or {}).get("step_results") or {})


def _build_prepare_context_payload(ctx):
    # PR-37 (2026-04-22) — forward the saga's own `no_tool_retry_count`
    # into the cortex prepare handler so the NO_TOOL_WARNING injection
    # can be driven by an explicit signal. Single source of truth is now
    # the saga, not inferred message-tail shape.
    return build_prepare_context_payload(ReactThinkInput.from_context(ctx))


def _build_llm_call_payload(ctx, prev_result):
    return build_llm_call_payload(ReactThinkInput.from_context(ctx), prev_result or {})


def _build_save_response_payload(ctx, prev_result):
    payload, unknown = build_save_response_payload(
        ReactThinkInput.from_context(ctx),
        prev_result or {},
        _get_known_tool_names(ctx),
    )
    if unknown:
        logger.info("[react_think] Filtered unknown tools: %s", ", ".join(unknown))
    return payload


def _should_save_llm_response(ctx):
    """Persist only LLM responses that are part of canonical dialogue state.

    A plain assistant message with no tool calls is invisible to the user in
    this product; it is a failed/no-tool attempt and must not become future
    LLM context. Raw payloads remain available through execution logs.
    """
    return should_save_llm_response(ctx or {}, _get_known_tool_names(ctx))


def _decide_and_build_actions(ctx, results):
    """Decide between actions / retry_think / finalize.

    P0-5 / INV-2 — **never finalize with a non-empty stack.** When the LLM returns
    no tool calls and the retry budget is exhausted, we consult the authoritative
    ``CORTEX_CHECK_STACK`` result before finalizing:

    - ``stack_depth == 0`` (and ``stack_known``) → safe to finalize.
    - ``stack_depth > 0`` or ``stack_known == False`` → retry_think once more
      (bounded by ``MAX_STACK_HOLD_RETRIES``) so the LLM gets a chance to emit
      ``skill_end``. After the bound, force finalize with a structured reason —
      leaving the session with a dangling skill is worse than a logged escape.
    - ``round_num >= MAX_ROUNDS_BEFORE_FORCE_FINALIZE`` → force finalize regardless
      (matches react_actions runaway guard).
    """
    decision = decide_react_think_route(ReactThinkDecisionInput(
        saga=ReactThinkInput.from_context(ctx),
        call_llm_result=(results or {}).get("call_llm") or {},
        prepare_context_result=(results or {}).get("prepare_context") or {},
        check_skill_stack_result=(results or {}).get("check_skill_stack") or {},
        max_rounds_before_force_finalize=ServiceConfig.MAX_ROUNDS_BEFORE_FORCE_FINALIZE,
        max_stack_hold_retries=ServiceConfig.MAX_STACK_HOLD_RETRIES,
    ))
    _log_react_think_decision(decision)
    return decision


def _log_react_think_decision(decision):
    if decision.get("has_tool_calls"):
        return
    scope_id = decision.get("scope_id")
    round_num = decision.get("round_num")
    if decision.get("should_retry"):
        logger.warning(
            "[react_think.decide] scope=%s round=%s stack_depth=%s stack_known=%s "
            "next=retry_think reason=%s",
            scope_id,
            round_num,
            decision.get("stack_depth"),
            decision.get("stack_known"),
            decision.get("retry_reason", "empty_response"),
        )
        return
    logger.warning(
        "[react_think.decide] scope=%s round=%s stack_depth=%s next=finalize reason=%s",
        scope_id,
        round_num,
        decision.get("stack_depth"),
        decision.get("finalize_reason"),
    )


def _build_trigger_actions(ctx, decision):
    return build_trigger_actions(ReactThinkInput.from_context(ctx), decision or {})


def _build_trigger_retry_think(ctx, decision):
    return build_trigger_retry_think(ReactThinkInput.from_context(ctx), decision or {})


def _build_trigger_finalize(ctx, decision):
    """Carry the finalize decision context into wake_finalize.

    P0-5: ``finalize_reason`` is always populated by ``_decide_and_build_actions``
    (``no_tool_exhausted`` / ``round_cap`` / ``stack_nonempty_after_retry``).
    """
    return build_trigger_finalize(ReactThinkInput.from_context(ctx), decision or {})


REACT_THINK_SAGA = SagaDefinition("react_think")

REACT_THINK_SAGA.add_task_step(
    name="prepare_context",
    topic=TaskTopics.CORTEX_PREPARE_LLM_CONTEXT,
    build_payload=_build_prepare_context_payload,
)

REACT_THINK_SAGA.add_task_step(
    name="call_llm",
    topic=TaskTopics.LLM_CALL,
    build_payload=_build_llm_call_payload,
)

REACT_THINK_SAGA.add_task_step(
    name="save_response",
    topic=TaskTopics.CONTEXT_APPEND,
    build_payload=_build_save_response_payload,
    condition=_should_save_llm_response,
)

# P0-5 / INV-2: Read authoritative stack depth BEFORE the terminate decision.
# react_actions already does this; react_think mirrors it so no terminate
# path bypasses the INV-2 check. ``check_skill_stack`` is a read-only call to
# Cortex ``/v1/context/status`` — safe to run every round.
REACT_THINK_SAGA.add_task_step(
    name="check_skill_stack",
    topic=TaskTopics.CORTEX_CHECK_STACK,
    build_payload=lambda ctx: build_check_skill_stack_payload(ReactThinkInput.from_context(ctx)),
)

REACT_THINK_SAGA.add_decision_step(
    name="decide_actions",
    decide=_decide_and_build_actions,
)

REACT_THINK_SAGA.add_task_step(
    name="trigger_actions",
    topic=SagaTopics.SAGA_TRIGGER,
    build_payload=_build_trigger_actions,
    condition=lambda d: d.get("has_tool_calls", False),
)
