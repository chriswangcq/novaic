# ── LLM Context assembly (Context Stack v2) ──


@register_handler(TaskTopics.CORTEX_PREPARE_LLM_CONTEXT)
def handle_cortex_prepare_llm_context(payload: Dict[str, Any], ctx: dict) -> Dict[str, Any]:
    """Collect explicit LLM assembly inputs and invoke the pure assembler.

    Boundary responsibilities:
    1. Read current Environment notification hints into Cortex.
    2. Ask Cortex for the prepared context snapshot.
    3. Ask Cortex/Common for active tool schemas.

    Pure assembly responsibilities live in ``common.contracts.llm_assembly``.
    """
    try:
        prepare_input = LLMPrepareContextInput.from_mapping(payload)
    except ValueError as exc:
        raise ValidationError(str(exc)) from exc

    # Step 1: merge current Environment notifications into Cortex context.
    from .context_handlers import handle_context_read
    read_result = handle_context_read({
        "scope_id": prepare_input.scope_id,
        "agent_root_scope_id": prepare_input.agent_root_scope_id,
        "wake_scope_path": prepare_input.wake_scope_path,
        "agent_id": prepare_input.agent_id,
        "subagent_id": prepare_input.subagent_id,
        "user_id": prepare_input.user_id,
    }, ctx)

    if not read_result.get("success"):
        return read_result

    # Step 2: ContextEvent-backed context preparation. `get_cortex_bridge` is
    # already imported at module top; the redundant local re-import that
    # used to sit here defeated `patch("…cortex_handlers.get_cortex_bridge")`
    # in tests (the local import would re-bind to the source module's
    # reference, bypassing the patch).
    bridge = get_cortex_bridge(
        prepare_input.agent_root_scope_id,
        prepare_input.user_id,
        prepare_input.agent_id,
    )
    prepare_result = bridge.prepare_for_llm(prepare_input.agent_root_scope_id)
    try:
        cortex_context = CortexPreparedContext.from_mapping(prepare_result)
    except ValueError as exc:
        raise ValidationError(str(exc)) from exc

    raw_tool_schemas = bridge.load_tool_schemas(
        subagent_id=prepare_input.subagent_id,
    )
    if not raw_tool_schemas:
        logger.warning("[cortex] No tool schemas returned from Cortex")

    # Transient NO_TOOL_WARNING is driven only by the explicit saga signal.
    # The pure assembler decides where to place it in the final request.
    no_tool_retry = prepare_input.no_tool_retry_count
    if no_tool_retry > 0:
        logger.info(
            "[cortex.prepare_llm_context] Injecting NO_TOOL_WARNING "
            "(no_tool_retry_count=%d, scope=%s)",
            no_tool_retry, prepare_input.scope_id,
        )

    assembly_output = assemble_llm_request_from_snapshot(
        LLMContextAssemblySnapshot(
            prepare_input=prepare_input,
            cortex_context=cortex_context,
            raw_tool_schemas=raw_tool_schemas,
            new_messages=read_result.get("new_messages", []),
        ),
        no_tool_warning=NO_TOOL_WARNING,
    )
    return assembly_output.to_dict()
