            ]
        }
    )

    assert parsed["text"] == "mcp image"
    assert parsed["display_files"][0]["url"] == "data:image/png;base64,abc"


def test_mcp_image_content_is_only_visual_in_display_perception_projection():
    parsed = parse_tool_result(
        {
            "_mcp_content": [
                {"type": "text", "text": "display image"},
                {"type": "image", "data": "abc", "mimeType": "image/png"},
            ]
        }
    )

    history = json.loads(format_for_history_llm(parsed))
    current = json.loads(format_for_current_tool_result_llm(parsed))
    display = json.loads(format_for_display_perception_llm(parsed))

    assert all(item["type"] == "text" for item in history["_mcp_content"])
    assert all(item["type"] == "text" for item in current["_mcp_content"])
    assert display["_mcp_content"][1] == {
        "type": "image",
        "data": "abc",
        "mimeType": "image/png",
    }
