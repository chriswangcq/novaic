  1902	    total = len(text)
  1903	    if mode in {"preview", "head"}:
  1904	        start = 0
  1905	        end = min(total, limit)
  1906	    elif mode == "tail":
  1907	        end = total
  1908	        start = max(0, total - limit)
  1909	    elif mode == "range":
  1910	        start = min(max(offset, 0), total)
  1911	        end = min(total, start + limit)

  1913	        raise HTTPException(status_code=400, detail=f"invalid payload read mode: {mode!r}")

  1915	    chunk = text[start:end]

  1918	        "end": end,

   700	    async def normalize_step(self, scope_path: str, step: dict) -> dict:
   701	        """Return the durable step shape used by step files and event records.
   702	
   703	        Normalization validates tool-step shape, externalizes any inline
   704	        ``payload`` into the scope payload store, and mirrors the resulting
   705	        ``payload_ref`` into the observation percept. The caller's input is not
   706	        mutated, so API/event paths can safely inspect the normalized step
   707	        before deciding which durable write path to invoke.
   708	        """
   709	        normalized = copy.deepcopy(step)
   710	        step_type = normalized.get("type", "tool")
   711	        step_id = normalized.get("call_id") or normalized.get("id") or "unknown"
   712	        if "/" in str(step_id) or ".." in str(step_id) or "\x00" in str(step_id):
   713	            raise ValueError(f"invalid step id: {step_id!r}")
   714	
   715	        if step_type == "tool" and "result" in normalized:
   716	            raise ValueError("tool steps must use payload/payload_ref, not inline result")
   717	        if step_type == "tool" and not isinstance(normalized.get("observation"), dict):
   718	            raise ValueError("tool steps require an observation percept")
   719	        if "payload" in normalized:
   720	            payload = normalized.pop("payload")
   721	            payload_ref = normalized.get("payload_ref")
   722	            if not payload_ref:
   723	                raise ValueError("payload_ref is required when writing payload")
   724	            stable_step_ref = str(normalized.get("step_ref") or payload_ref)
   725	            actual_payload_ref = await self.write_payload(
   726	                scope_path,
   727	                str(payload_ref),
   728	                payload,
   729	                step_ref=stable_step_ref,
   730	            )
   731	            normalized["step_ref"] = stable_step_ref
   732	            normalized["payload_ref"] = actual_payload_ref
   733	            if isinstance(normalized.get("observation"), dict):
   734	                normalized["observation"]["payload_ref"] = actual_payload_ref
   735	
   736	        return normalized

   308	    """Generate a concise text preview of a tool result."""
   309	    text = result_data.get("text", "")
   310	    if len(text) > max_text_len:
   311	        text = text[:max_text_len] + "... [truncated]"
   312	    
   313	    files = result_data.get("files_created", [])
   314	    displays = result_data.get("display_files", [])
   315	    artifacts = result_data.get("artifacts", [])
   316	    
   317	    parts = [text] if text else []
   318	    if files:
   319	        parts.append(f"[{len(files)} files created]")
   320	    if displays:
   321	        parts.append(f"[{len(displays)} media items]")
   322	    if artifacts:
   323	        parts.append(f"[{len(artifacts)} artifacts]")
   324	        
   325	    if not parts:
   326	        return "(empty)"
   327	    return "\n".join(parts)

    42	def _resolve_display_image_refs(content: str, *, user_id: Optional[str]) -> str:
    43	    if not user_id:
    44	        return content
    45	
    46	    try:
    47	        payload = json.loads(content)
    48	    except Exception:
    49	        return content
    50	    mcp_content = payload.get("_mcp_content") if isinstance(payload, dict) else None
    51	    if not isinstance(mcp_content, list):
    52	        return content
    53	
    54	    resolved: list[dict[str, Any]] = []
    55	    changed = False
    56	    for item in mcp_content:
    57	        if not isinstance(item, dict) or item.get("type") != "image_ref":
    58	            resolved.append(item)
    59	            continue
    60	
    61	        changed = True
    62	        file_url = str(item.get("file_url") or item.get("url") or "").strip()
    63	        if not file_url:
    64	            resolved.append(_image_ref_resolution_note("", "missing file_url"))
    65	            continue
    66	        try:
    67	            data, fetched_mime = _blob_get(file_url, user_id=user_id)
    68	        except Exception as exc:
    69	            resolved.append(_image_ref_resolution_note(file_url, str(exc)))
    70	            continue
    71	        if len(data) > _DISPLAY_IMAGE_REF_MAX_BYTES:
    72	            resolved.append(_image_ref_resolution_note(
    73	                file_url,
    74	                f"image too large for inline perception ({len(data)} bytes)",
    75	            ))
    76	            continue
    77	        mime_type = fetched_mime or str(item.get("mimeType") or item.get("mime_type") or "image/png")
    78	        if not mime_type.startswith("image/"):
    79	            resolved.append(_image_ref_resolution_note(file_url, f"not an image MIME type: {mime_type}"))
    80	            continue
    81	        resolved.append({
    82	            "type": "image",
    83	            "data": base64.b64encode(data).decode("ascii"),
    84	            "mimeType": mime_type,
    85	        })
    86	
    87	    if not changed:
    88	        return content
    89	
    90	    payload["_mcp_content"] = resolved
    91	    return json.dumps(payload, ensure_ascii=False)
