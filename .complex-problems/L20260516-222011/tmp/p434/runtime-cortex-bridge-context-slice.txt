            payload["scope_path"] = scope_path
        if session_generation is not None:
            payload["session_generation"] = require_positive_session_generation_value(
                session_generation,
                "CortexBridge.scope_end",
            )
        if finalize_reason is not None:
            payload["finalize_reason"] = finalize_reason
        if remaining_stack is not None:
            payload["remaining_stack"] = dict(remaining_stack)
        if round_num is not None:
            payload["round_num"] = int(round_num)
        result = self._post("/v1/scope/end", payload)
        archive_path = result.get("archive_path")
        logger.info("[CortexBridge] scope.archived scope_id=%s path=%s", scope_id, archive_path)
        return archive_path

    def read_context(self, scope_id: str, *, scope_path: str | None = None) -> list[dict]:
