    66	export function activityTimelinePublicTitle(record: ActivityTimelineRecord): string {
    67	  const fingerprint = `${record.kind} ${record.tool ?? ''} ${record.title}`.toLowerCase();
    68	  if (fingerprint.includes('skill_end')) return '已保存本轮上下文';
    69	  if (isAgentctlImReadRecord(record)) return '读取了你的消息';
    70	  if (isAgentctlImReplyRecord(record)) return '已回复你';
    71	  if (fingerprint.includes('message') || fingerprint.includes('消息')) return '读取了你的消息';
    72	  if (record.phase === 'reasoning') {
    73	    const reasoningText = record.text?.toLowerCase() ?? '';
    74	    if (
    75	      AGENTCTL_IM_READ_PATTERN.test(reasoningText) ||
    76	      reasoningText.includes('读取消息') ||
    77	      reasoningText.includes('新消息')
    78	    ) {
    79	      return '正在查看你的消息';
    80	    }
    81	    if (AGENTCTL_IM_REPLY_PATTERN.test(reasoningText) || reasoningText.includes('回复')) {
    82	      return '正在组织回复';
    83	    }
    84	    if (
    85	      reasoningText.includes('skill_end') ||
    86	      reasoningText.includes('summary') ||
    87	      reasoningText.includes('上下文') ||
    88	      reasoningText.includes('结束本轮')
    89	    ) {
    90	      return '正在整理上下文';
    91	    }
    92	    return isActivityTimelineRecordRunning(record) ? '正在思考' : '完成思考';
    93	  }
    94	  if (record.phase === 'summary') return '已保存本轮上下文';
    95	  return record.title || PHASE_FALLBACK_LABELS[record.phase];
    96	}

    98	function publicFullDetail(record: ActivityTimelineRecord): string | undefined {
    99	  const text = record.text?.trim();
   100	  if (!text) return record.has_payload ? '结果已保存，需要时可以继续查看细节' : undefined;
   101	  if (looksLikeRawPayloadText(text)) return RAW_PAYLOAD_DETAIL;
   102	  if (/[{}]/.test(text)) return undefined;
   103	  if (DEBUG_DETAIL_PATTERN.test(text)) return undefined;
   104	  if (record.phase === 'action' && record.tool) return undefined;
   105	  return text.replace(/\r\n/g, '\n');
   106	}
   107	
   108	function looksLikeRawPayloadText(text: string): boolean {
   109	  if (DATA_IMAGE_BASE64_PATTERN.test(text)) return true;
   110	  const compact = text.replace(/\s+/g, '');
   111	  if (compact.length < 120) return false;
   112	  if (BINARY_BASE64_PREFIX_PATTERN.test(compact)) return true;
   113	  return compact.length > 800 && /^[A-Za-z0-9+/=]+$/.test(compact) && /[+/=]/.test(compact);
   114	}

   116	function publicPreviewDetail(record: ActivityTimelineRecord): string | undefined {
   117	  const fullDetail = publicFullDetail(record);
   118	  if (!fullDetail) return undefined;
   119	  const oneLine = fullDetail.replace(/\s+/g, ' ');
   120	  return oneLine.length > DETAIL_PREVIEW_LIMIT ? `${oneLine.slice(0, DETAIL_PREVIEW_LIMIT)}…` : oneLine;
   121	}
   122	
   123	function isExpandableDetail(record: ActivityTimelineRecord): boolean {
   124	  const fullDetail = publicFullDetail(record);
   125	  if (!fullDetail) return false;
   126	  const oneLine = fullDetail.replace(/\s+/g, ' ');
   127	  return Boolean(record.truncated || fullDetail.includes('\n') || oneLine.length > DETAIL_PREVIEW_LIMIT);
   128	}

   134	function shouldHidePublicRecord(record: ActivityTimelineRecord): boolean {
   135	  const tool = normalizedTool(record);
   136	  if (record.phase === 'action' && BOOKKEEPING_TOOLS.has(tool)) return true;
   137	  if (record.phase === 'observation' && tool === 'skill_end') return true;
   138	  return false;

   153	export function projectActivityTimelineRecords(records: ActivityTimelineRecord[]): ActivityTimelineRecord[] {
   154	  const projected: ActivityTimelineRecord[] = [];
   155	  const sortedRecords = [...records].sort((a, b) => a.order - b.order);
   156	  for (const record of sortedRecords) {
   157	    if (shouldHidePublicRecord(record)) continue;
   158	    const previous = projected[projected.length - 1];
   159	    if (previous && isDuplicatePublicRecord(previous, record)) continue;
   160	    projected.push(record);
   161	  }
   162	  return projected;
   163	}

    24	export function normalizeActivityRecord(item: unknown): ActivityTimelineRecord | null {
    25	  if (!item || typeof item !== 'object' || Array.isArray(item)) return null;
    26	  const record = item as Record<string, unknown>;
    27	  if (typeof record.sort_order !== 'number') return null;
    28	  if (!isPhase(record.phase)) return null;
    29	  if (typeof record.kind !== 'string' || typeof record.title !== 'string') return null;
    30	  return {
    31	    order: record.sort_order,
    32	    phase: record.phase,
    33	    kind: record.kind,
    34	    title: record.title,
    35	    text: typeof record.text === 'string' ? record.text : undefined,
    36	    truncated: typeof record.truncated === 'boolean' ? record.truncated : undefined,
    37	    tool: typeof record.tool === 'string' ? record.tool : undefined,
    38	    status: typeof record.status === 'string' ? record.status : undefined,
    39	    has_payload: typeof record.has_payload === 'boolean' ? record.has_payload : undefined,
    40	  };
