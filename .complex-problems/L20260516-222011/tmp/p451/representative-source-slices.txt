#!/usr/bin/env python3
"""PR-76 Cortex boundary guard.

The guard scans active Cortex Python source for concepts that no longer
belong in Cortex. Documentation, ticket history, and tombstones are not
scanned by default; active-code exceptions must be narrow and cite a ticket.
"""

from __future__ import annotations

import argparse
import re
import sys
from dataclasses import dataclass
from pathlib import Path
from typing import Mapping, Sequence


DEFAULT_SCAN_DIRS = ("novaic-cortex/novaic_cortex",)
PYTHON_GLOB = "*.py"


@dataclass(frozen=True)
class BoundaryRule:
    rule_id: str
    description: str
    pattern: re.Pattern[str]


@dataclass(frozen=True)
class BoundaryViolation:
    path: str
    line: int
    rule_id: str
    description: str
    text: str


_LEGACY_REPLY_TOOL_NAME = "im" + "_reply"


RULES: tuple[BoundaryRule, ...] = (
    BoundaryRule(
        "auto_summary_pipeline",
        "Cortex must not reintroduce automatic summary generation.",
        re.compile(r"\b(auto_summary|generate_simple_summary|Summarizer|CompactResult|gem_fusion)\b"),
    ),
    BoundaryRule(
        "profile_ownership",
        "Cortex must not own user-profile inference or profile mutation.",
        re.compile(r"\b(user_profile|drive_update_profile|profile_inference|infer_user_profile)\b"),
    ),
    BoundaryRule(
        "business_product_context_ownership",
        "Cortex must not own Business product-context entities or binding schema.",
        re.compile(r"\b(agent-tools|agent-binding|mounted_tools|user_md|subagent_context)\b"),
    ),
    BoundaryRule(
        "business_memory_proxy",
        "Cortex must not proxy business memory/notebook/search commands.",
        re.compile(r"(_route_memory|_route_notebook|_route_search|cmd_memory|cmd_notebook|cmd_search|/memory/|/notebook/|novaic\s+(memory|notebook|search))"),
    ),
    BoundaryRule(
        "business_task_proxy",
        "Cortex must not proxy the business task system.",
        re.compile(r"(_route_task|cmd_task|quadrant-tasks|novaic\s+task)"),
    ),
    BoundaryRule(
        "wake_summary",
        "Cortex must not create or consume a wake-summary channel.",
        re.compile(r"\b(wake_summary|wake summary)\b", re.IGNORECASE),
    ),
    BoundaryRule(
        "reply_action_memory",
        "Cortex must not derive durable memory from user-visible reply text.",
        re.compile(
            rf"(({_LEGACY_REPLY_TOOL_NAME}|reply action).{{0,80}}(summary|memory)|"
            rf"(?:summary|memory).{{0,80}}({_LEGACY_REPLY_TOOL_NAME}|reply action))",
            re.IGNORECASE,
        ),
    ),
    BoundaryRule(
        "parallel_summary_channels",
        "Cortex must not grow parallel summary producers.",
        re.compile(r"\b(handoff_notes|historical_summary|summary_from_reply|reply_summary)\b"),
    ),
)


Allowlist = Mapping[str, set[str]]


def _rel(path: Path, root: Path) -> str:
    try:
        return path.relative_to(root).as_posix()
    except ValueError:
        return path.as_posix()


def _iter_python_files(root: Path, scan_dirs: Sequence[str]) -> list[Path]:
    files: list[Path] = []
    for scan_dir in scan_dirs:
        base = root / scan_dir
        if not base.exists():
            continue
        if base.is_file():
            if base.name.endswith(".py"):
                files.append(base)
            continue
        files.extend(
            p
            for p in sorted(base.rglob(PYTHON_GLOB))
            if "__pycache__" not in p.parts
        )
    return sorted(files)


def find_violations(
    root: Path,
    *,
    scan_dirs: Sequence[str] = DEFAULT_SCAN_DIRS,
    allowlist: Allowlist | None = None,
) -> list[BoundaryViolation]:
    """Return active-code boundary violations.

    ``allowlist`` maps repository-relative file paths to allowed rule ids.
    Use it sparingly and include the owning PR in the nearby source comment.
    """
    root = root.resolve()
    allowlist = allowlist or {}
    violations: list[BoundaryViolation] = []

    for path in _iter_python_files(root, scan_dirs):
        rel = _rel(path, root)
        allowed_rules = allowlist.get(rel, set())
        try:
            lines = path.read_text(encoding="utf-8").splitlines()
        except UnicodeDecodeError:
            continue
        for lineno, line in enumerate(lines, start=1):
            for rule in RULES:
                if rule.rule_id in allowed_rules:
                    continue
                if rule.pattern.search(line):
                    violations.append(
                        BoundaryViolation(
                            path=rel,
                            line=lineno,
                            rule_id=rule.rule_id,
                            description=rule.description,
                            text=line.strip(),
                        )
                    )
    return violations


def format_violations(violations: Sequence[BoundaryViolation]) -> str:
    out = ["Cortex boundary violations:"]
    for v in violations:
        out.append(f"- {v.path}:{v.line} [{v.rule_id}] {v.description}")
        out.append(f"  {v.text}")
    return "\n".join(out)


def main(argv: Sequence[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument("--root", default=".", help="Repository root to scan.")
    parser.add_argument(
        "--scan-dir",
        action="append",
        dest="scan_dirs",
        help="Repository-relative active source dir. Repeatable.",
    )
    args = parser.parse_args(argv)

    violations = find_violations(
        Path(args.root),
        scan_dirs=tuple(args.scan_dirs or DEFAULT_SCAN_DIRS),
    )
    if violations:
        print(format_violations(violations), file=sys.stderr)
        return 1
    print("Cortex boundary guard passed.")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())

"""
Cortex step result projection helpers.

Tool outputs are stored on Cortex steps. This module parses those stored
results and converts them into provider-facing LLM representations.
"""

_UNKNOWN_DICT_TEXT_LIMIT = 2_000


def _tool_output_size_label(size: Any) -> str:
    if not isinstance(size, int) or size < 0:
        return ""
    value = float(size)
    for unit in ("B", "KB", "MB", "GB"):
        if value < 1024:
            return f"{int(value)}B" if unit == "B" else f"{value:.1f}{unit}"
        value /= 1024
    return f"{value:.1f}TB"


def _tool_output_artifact_line(artifact: Dict[str, Any]) -> str:
    uri = str(artifact.get("uri") or "").strip()
    modality = str(artifact.get("modality") or "resource").strip() or "resource"
    name = str(artifact.get("name") or "").strip()
    mime_type = str(artifact.get("mime_type") or "").strip()
    summary = str(artifact.get("summary") or "").strip()
    size = _tool_output_size_label(artifact.get("size_bytes"))
    label = name or uri or "artifact"
    details = ", ".join(part for part in (modality, mime_type, size) if part)
    prefix = f"[Artifact: {label}"
    if details:
        prefix += f" ({details})"
    prefix += "]"

    access = artifact.get("access") if isinstance(artifact.get("access"), dict) else {}
    display = access.get("display") if isinstance(access.get("display"), dict) else {}
    display_args = display.get("args") if isinstance(display.get("args"), dict) else {}
    display_url = display_args.get("file_url") or uri

    parts = [prefix]
    if summary:
        parts.append(summary)
    if display_url:
        parts.append(f'Access: display(file_url="{display_url}")')
    return " — ".join(parts)


def _parse_tool_output_v1(raw: Dict[str, Any]) -> Dict[str, Any]:
    text_parts: list[str] = []
    text = raw.get("text")
    if text:
        text_parts.append(str(text))

    artifacts = raw.get("artifacts") if isinstance(raw.get("artifacts"), list) else []
    for artifact in artifacts:
        if isinstance(artifact, dict):
            text_parts.append(_tool_output_artifact_line(artifact))

    events = raw.get("events") if isinstance(raw.get("events"), list) else []
    for event in events:
        if not isinstance(event, dict):
            continue
        event_type = str(event.get("type") or "").strip()
        summary = str(event.get("summary") or "").strip()
        if event_type or summary:
            label = f"[Event: {event_type or 'event'}]"
            text_parts.append(f"{label} {summary}".strip())

    error = raw.get("error")
    if error and not raw.get("ok", True):
        text_parts.append(f"[Error] {error}")

    return {
        "text": "\n".join(text_parts),
        "files_created": [],
        "display_files": [],
        "artifacts": artifacts,
        "events": events,
    }


def parse_tool_result(raw: Any) -> Dict[str, Any]:
    """
    解析工具原始结果为三要素格式。

    支持的输入格式：
    1. 统一格式: {text, files_created, display_files}
    2. MCP 格式: {content: [{type, text/data}]}
    3. 纯文本: str

    Returns:
        {"text": "...", "files_created": [...], "display_files": [...]}
    """
    if raw is None:
        return {"text": "", "files_created": [], "display_files": []}

    if isinstance(raw, str):
        stripped = raw.strip()
        if stripped.startswith("{") or stripped.startswith("["):
            try:
                decoded = json.loads(stripped)
            except json.JSONDecodeError:
                decoded = None
            if isinstance(decoded, (dict, list)):
                return parse_tool_result(decoded)
        return {"text": raw, "files_created": [], "display_files": []}

    if not isinstance(raw, dict):
        return {"text": str(raw), "files_created": [], "display_files": []}

    if raw.get("version") == _TOOL_STEP_PAYLOAD_VERSION:
        return parse_tool_result(raw.get("llm_content", ""))

    if raw.get("version") == "tool-output.v1":
        return _parse_tool_output_v1(raw)

    # 1. 统一格式（三要素）
    if "display_files" in raw or "files_created" in raw:
        text_parts = []
        if raw.get("text"):
            text_parts.append(str(raw["text"]))
        # 把 files_created 的 URL 写入文本
        files_created = raw.get("files_created") or []
        if files_created and isinstance(files_created, list):
            urls = [f.get("url") for f in files_created if isinstance(f, dict) and f.get("url")]
            if urls:
                text_parts.append(f"file_url: {urls[0]}" if len(urls) == 1 else f"file_urls: {', '.join(urls)}")
        return {
            "text": "\n".join(text_parts),
            "files_created": files_created,
            "display_files": raw.get("display_files") or [],
        }

    # 2. MCP 格式 content 数组
    content = raw.get("content") or raw.get("_mcp_content")
    if isinstance(content, list):
        text_parts = []
        display_files = []
        for item in content:
            if not isinstance(item, dict):
                continue
            t = item.get("type", "")
            if t == "text":
                text_parts.append(item.get("text", ""))
            elif t == "image":
                # 支持 URL 形式或直接的 data(base64) 形式
                if item.get("url"):
                    display_files.append({
                        "url": item["url"],
                        "filename": "",
                        "modality": "image",
                    })
                elif item.get("data"):
                    mime = item.get("mimeType", "image/png")
                    display_files.append({
                        "url": f"data:{mime};base64,{item['data']}",
                        "filename": "",
                        "modality": "image",
                    })
            elif t == "resource":
                if item.get("url"):
                    display_files.append({
                        "url": item["url"],
                        "filename": "",
                        "modality": "resource",
                    })
        return {
            "text": "\n".join(text_parts),
            "files_created": [],
            "display_files": display_files,
        }

    # 4. 其他 dict：提取 text 字段或序列化
    if raw.get("text"):
        return {"text": str(raw["text"]), "files_created": [], "display_files": []}

    # 5. Unknown dict fallback: diagnostic text only, bounded so unexpected
    # payload shapes cannot flood LLM context.
    def safe_default(obj):
        return f"<Unserializable {type(obj).__name__}>"
    try:
        serialized = json.dumps(raw, ensure_ascii=False, default=safe_default)
    except Exception as e:
        serialized = f"[StepResultProjection Parse Error] Failed to serialize object: {str(e)}"
    if len(serialized) > _UNKNOWN_DICT_TEXT_LIMIT:
        serialized = (
            serialized[:_UNKNOWN_DICT_TEXT_LIMIT]
            + f"\n\n... [Truncated unknown tool result JSON length > {_UNKNOWN_DICT_TEXT_LIMIT}]"
        )
    serialized = "[Unknown tool result JSON]\n" + serialized
    return {"text": serialized, "files_created": [], "display_files": []}

"""Workspace — /ro + /rw ACL + tree-structured scopes.

Two zones:
  - /ro/ = Cortex-managed (scopes, config, skills, knowledge).
           Agent can READ but NOT write. All scope operations (create, step,
           archive) are Cortex system operations and use _sys_* methods.
  - /rw/ = Agent's free space (scratch files, agent-managed data).
           Agent can read AND write freely via public write/append_line/etc.

Scope tree:
  - Active scopes live at /ro/active/{scope_id}/
  - Archived scopes live at /ro/scopes/{scope_id}/
  - Child scopes nest inside parent's steps/ as directories
  - LIFO: children close before parents → root archive moves tree atomically
"""

import copy
import inspect

import json
import logging
import time
import warnings
from typing import Any, Callable, Dict, List, Optional

class PayloadReadError(RuntimeError):
    """Domain error for Cortex payload read failures."""

    def __init__(self, *, code: str, payload_ref: str, message: str) -> None:
        super().__init__(message)
        self.code = code
        self.payload_ref = payload_ref
        self.message = message


def _payload_filename(payload_ref: str) -> str:
    digest = hashlib.sha256(payload_ref.encode("utf-8")).hexdigest()
    return f"{digest}.json"


def _scope_id_from_scope_dir(name: str) -> str:
    if "_scope_" in name:
        return name.split("_scope_", 1)[1]
    return name


def _payload_scope_ids(scope_path: str) -> tuple[str, str]:
    parts = [part for part in scope_path.strip("/").split("/") if part]
    if len(parts) < 3 or parts[0] != "ro" or parts[1] not in {"active", "scopes"}:
        raise ValueError(f"cannot derive payload scope ids from {scope_path!r}")

    root_scope_id = parts[2]
    scope_id = root_scope_id
    for index, part in enumerate(parts):
        if part == "steps" and index + 1 < len(parts):
            scope_id = _scope_id_from_scope_dir(parts[index + 1])
    return root_scope_id, scope_id

class Workspace:
    def __init__(
        self,
        file_authority: Any,
        agent_id: str,
        hooks: CortexHooks | None = None,
        payload_blob_client: PayloadBlobClient | None = None,
        payload_blob_policy: BlobPayloadPolicy | None = None,
        clock: Callable[[], float] | None = None,
        operational_store: Any | None = None,
    ) -> None:
        validate_agent_id(agent_id)
        self._files = file_authority
        self._agent_id = agent_id
        self._hooks = hooks
        self._payload_blob_client = payload_blob_client
        self._payload_blob_policy = payload_blob_policy or BlobPayloadPolicy()
        self._clock = clock or time.time
        self._operational_store = operational_store

    def _now(self) -> float:
        return float(self._clock())

    async def _emit(self, hook_list_name: str, *args: object) -> None:
        if self._hooks is None:
            return
        cbs = getattr(self._hooks, hook_list_name, None)
        if not cbs:
            return
        for cb in cbs:
            try:
                if inspect.iscoroutinefunction(cb):
                    await cb(*args)
                else:
                    cb(*args)
            except Exception as e:
                warnings.warn(
                    f"Cortex hook {hook_list_name!r} failed: {e}",
                    UserWarning,
                    stacklevel=2,
                )

    @property
    def agent_id(self) -> str:
        return self._agent_id

    @property
    def operational_store(self) -> Any | None:
        return self._operational_store

    def _require_operational_store(self) -> Any:
        try:
            operational_store = self._operational_store
        except AttributeError:
            operational_store = None
        if operational_store is None:
            raise RuntimeError("Workspace operational_store is required for scope lifecycle transitions")
        return operational_store

    def _update_payload_manifest_error(
        self,
        *,
        payload_ref: str,
        status: str,
        code: str,
        message: str,
    ) -> None:
        operational_store = self._require_operational_store()
        manifest = operational_store.get_payload_manifest(payload_ref=payload_ref)
        if manifest is None:
            return
        operational_store.put_payload_manifest(
            payload_ref=manifest["payload_ref"],
            source_payload_ref=manifest.get("source_payload_ref"),
            root_scope_id=manifest["root_scope_id"],
            scope_id=manifest.get("scope_id"),
            step_ref=manifest.get("step_ref"),
            blob_ref=manifest.get("blob_ref"),
            mime_type=manifest["mime_type"],
            size_bytes=int(manifest["size_bytes"]),
            sha256=manifest["sha256"],
            status=status,
            retention_class=manifest["retention_class"],
            error={"code": code, "message": message},
        )

    def _payload_read_error(
        self,
        *,
        payload_ref: str,
        code: str,
        status: str,
        message: str,
    ) -> "PayloadReadError":
        self._update_payload_manifest_error(
            payload_ref=payload_ref,
            status=status,
            code=code,
            message=message,
        )
        return PayloadReadError(code=code, payload_ref=payload_ref, message=message)

    def _validate_write(self, logical: str) -> None:
        """Agent ACL: only /rw/ is writable by the agent."""
        if not logical.strip().startswith("/rw/"):
            raise PermissionError(f"agent cannot write {logical!r} (only /rw/)")

    # ── System I/O (Cortex-managed, bypasses agent ACL) ──

    async def _sys_write(self, path: str, content: str) -> None:
        """Cortex system write — no agent ACL check."""
        await self._files.write_text(path, content)

    async def _sys_write_json(self, path: str, obj: dict) -> None:
        await self._files.write_json(path, obj)

    async def _sys_append_line(self, path: str, line: str) -> None:
        await self._files.append_line(path, line)

    # ── Agent I/O (subject to /rw/ ACL) ──

    async def read(self, path: str) -> str:
        return await self._files.read_text(path)

    async def read_bytes(self, path: str) -> bytes:
        return await self._files.read_bytes(path)

Flow:

def _decide_and_build_actions(ctx, results):
    """Decide between actions / retry_think / finalize.

    P0-5 / INV-2 — **never finalize with a non-empty stack.** When the LLM returns
    no tool calls and the retry budget is exhausted, we consult the authoritative
    ``CORTEX_CHECK_STACK`` result before finalizing:

    - ``stack_depth == 0`` (and ``stack_known``) → safe to finalize.
    - ``stack_depth > 0`` or ``stack_known == False`` → retry_think once more
      (bounded by ``MAX_STACK_HOLD_RETRIES``) so the LLM gets a chance to emit
      ``skill_end``. After the bound, force finalize with a structured reason —
      leaving the session with a dangling skill is worse than a logged escape.
    - ``round_num >= MAX_ROUNDS_BEFORE_FORCE_FINALIZE`` → force finalize regardless
      (matches react_actions runaway guard).
    """
    decision = decide_react_think_route(ReactThinkDecisionInput(
        saga=ReactThinkInput.from_context(ctx),
        call_llm_result=(results or {}).get("call_llm") or {},
        prepare_context_result=(results or {}).get("prepare_context") or {},
        check_skill_stack_result=(results or {}).get("check_skill_stack") or {},
        max_rounds_before_force_finalize=ServiceConfig.MAX_ROUNDS_BEFORE_FORCE_FINALIZE,
        max_stack_hold_retries=ServiceConfig.MAX_STACK_HOLD_RETRIES,
    ))
    _log_react_think_decision(decision)
    return decision


def _log_react_think_decision(decision):
    if decision.get("has_tool_calls"):
        return
    scope_id = decision.get("scope_id")
    round_num = decision.get("round_num")
    if decision.get("should_retry"):
        logger.warning(
            "[react_think.decide] scope=%s round=%s stack_depth=%s stack_known=%s "
            "next=retry_think reason=%s",
            scope_id,
            round_num,
            decision.get("stack_depth"),
            decision.get("stack_known"),
            decision.get("retry_reason", "empty_response"),
        )
        return
    logger.warning(
        "[react_think.decide] scope=%s round=%s stack_depth=%s next=finalize reason=%s",
        scope_id,
        round_num,
        decision.get("stack_depth"),
        decision.get("finalize_reason"),
    )


def _build_trigger_actions(ctx, decision):
    return build_trigger_actions(ReactThinkInput.from_context(ctx), decision or {})


def _build_trigger_retry_think(ctx, decision):
    return build_trigger_retry_think(ReactThinkInput.from_context(ctx), decision or {})


def _build_trigger_finalize(ctx, decision):
    """Carry the finalize decision context into wake_finalize.

    P0-5: ``finalize_reason`` is always populated by ``_decide_and_build_actions``
    (``no_tool_exhausted`` / ``round_cap`` / ``stack_nonempty_after_retry``).
    """
    return build_trigger_finalize(ReactThinkInput.from_context(ctx), decision or {})
