        else:
            matched = entry.get("id") == tool_call_id

        if matched:
            path = f"{scope_path}/steps/{filename}"
            try:
                raw = await ws.read(path)
                step = json.loads(raw)
                step["_scope_path"] = scope_path
                return step
            except (KeyError, FileNotFoundError, json.JSONDecodeError):
                pass
    return None


async def _read_step_payload(ws, step: dict) -> object | None:
    payload_ref = step.get("payload_ref")
    scope_path = step.get("_scope_path")
    if not payload_ref or not scope_path:
        return None
    try:
        return await ws.read_payload(scope_path, payload_ref)
    except PayloadReadError as exc:
        status_code = {
            "payload_record_missing": 404,
            "payload_ref_mismatch": 409,
            "payload_record_corrupt": 422,
            "payload_blob_client_missing": 502,
            "payload_blob_fetch_failed": 502,
        }.get(exc.code, 500)
        raise HTTPException(
            status_code=status_code,
            detail={
                "code": exc.code,
                "message": exc.message,
                "payload_ref": exc.payload_ref,
            },
        ) from exc
    except (KeyError, FileNotFoundError, ValueError, json.JSONDecodeError):
        return None


@app.post("/v1/steps/read_formatted")
async def steps_read_formatted(req: StepFormattedRequest):
    """Read a step and return its result formatted as _mcp_content for LLM."""
    from novaic_cortex.step_result_projection import (
        parse_tool_result,
        format_for_current_tool_result_llm,
        format_for_display_perception_llm,
        format_for_history_llm,
    )

    ws = _get_workspace(req.user_id, req.agent_id)
    step = await _find_step_by_call_id(ws, req.scope_id, req.tool_call_id, req.step_ref)
    if step is None:
        return {"content": None}

    raw_result = await _read_step_payload(ws, step)
    if raw_result is None:
        return {"content": None}

    parsed = parse_tool_result(raw_result)
    projection = (req.projection or "").strip()
    if projection == "display_perception":
        content = format_for_display_perception_llm(parsed, req.provider)
    elif projection == "current_tool_result":
        content = format_for_current_tool_result_llm(parsed, req.provider)
    elif projection == "history":
        content = format_for_history_llm(parsed, req.provider)
    else:
        raise HTTPException(
            status_code=400,
            detail={
                "error_code": "unsupported_step_projection",
                "projection": projection,
                "allowed": ["history", "current_tool_result", "display_perception"],
            },
        )
    return {"content": content}
