    return int(float(now()) * 1000)


def _write_active_stack_projection(
    ws: Any,
    *,
    root_scope_id: str,
    frames: list[dict],
    reason: str,
    idempotency_key: str | None = None,
) -> dict[str, Any]:
    return write_active_stack_projection(
        ws.operational_store,
        root_scope_id=root_scope_id,
        frames=frames,
        generation=_active_stack_generation_ms(ws),
        reason=reason,
        idempotency_key=idempotency_key,
    )


@app.post("/v1/context/skill_begin")
async def context_skill_begin(req: ContextSkillBeginRequest):
    """skill_begin → create child scope under the current active (stack-top) scope.

    LLM-chosen child_scope_id must be globally unique across this agent's entire
    scope tree (both active and archived).
    """
    async with _scope_lock_cm(req.user_id, req.agent_id, req.scope_id, op="skill_begin"):
        return await _context_skill_begin_locked(req)


async def _context_skill_begin_locked(req):
    ws = _get_workspace(req.user_id, req.agent_id)
    root_path = _root_scope_path(req.scope_id)
    child_scope_id = (req.child_scope_id or "").strip()
    active_stack = read_active_stack_projection(
        ws.operational_store,
        root_scope_id=req.scope_id,
        root_scope_path=root_path,
    )
    current_stack = active_stack["frames"]

    if not child_scope_id:
        return {
            "ok": False,
            "error": "child_scope_id is required",
            "stack": current_stack,
            "stack_depth": len(current_stack),
        }
    if "/" in child_scope_id or ".." in child_scope_id or "\x00" in child_scope_id:
        return {
            "ok": False,
            "error": f"invalid child_scope_id: {child_scope_id!r}",
            "stack": current_stack,
            "stack_depth": len(current_stack),
        }

    try:
        existing_scope = ws.operational_store.get_scope_projection(
            root_scope_id=req.scope_id,
            scope_id=child_scope_id,
        )
        if existing_scope is not None:
            return {
                "ok": False,
                "error": (
                    f"scope_id '{child_scope_id}' already used in this session. "
                    "Pick a different, globally unique id."
                ),
                "stack": current_stack,
                "stack_depth": len(current_stack),
            }

        # Depth ceiling for LLM-opened child skills:
        # Prevent runaway nested skill_begin without a matching skill_end.
        engine_cfg = await load_engine_config(ws)
        max_depth = getattr(engine_cfg, "max_skill_depth", 4)
        if len(current_stack) >= max_depth:
            log_cortex(
                "skill.begin_rejected",
                reason="depth_limit",
                stack_depth=len(current_stack),
                max_depth=max_depth,
            )
            return {
                "ok": False,
                "error": (
                    f"Skill stack depth limit reached ({max_depth}). "
                    f"Cannot push '{req.skill_name}'. Close an active skill first."
                ),
                "stack": current_stack,
                "stack_depth": len(current_stack),
            }

        parent_path = active_stack["active_scope_path"]
        child_path = await ws.create_scope_projection(
            scope_id=child_scope_id,
            name=req.task or req.skill_name,
            skill=req.skill_name,
            parent_path=parent_path,
        )
        root_scope_id, root_scope_path = _root_scope_identity_from_path(parent_path)
        await _context_event_writer(ws).skill_scope_opened(
            _event_write_context(
                req,
                root_scope_id=root_scope_id,
                root_scope_path=root_scope_path,
            ),
            scope_id=child_scope_id,
            parent_scope_id=_scope_id_from_path(parent_path),
            skill_name=req.skill_name,
            name=req.task or req.skill_name,
        )
        stack = [
            _active_stack_frame(
                scope_path=child_path,
                parent_path=parent_path,
                scope_id=child_scope_id,
                skill_name=req.skill_name,
                name=req.task or req.skill_name,
                depth=len(current_stack) + 1,
            ),
            *current_stack,
        ]
        _write_active_stack_projection(
            ws,
            root_scope_id=req.scope_id,
            frames=stack,
            reason="skill_begin",
        )
        log_cortex(
            "skill.begun",
            user_id=req.user_id,
            agent_id=req.agent_id,
            root_scope_id=req.scope_id,
            child_scope_id=child_scope_id,
            skill_name=req.skill_name,
            stack_depth=len(stack),
        )
        return {
            "ok": True,
            "scope_id": child_scope_id,
            "child_path": child_path,
            "parent_path": parent_path,
            "skill_name": req.skill_name,
            "stack": stack,
            "stack_depth": len(stack),
        }
    except Exception as e:
        logger.error("skill_begin failed: %s", e)
        return {
            "ok": False,
            "error": str(e),
            "stack": current_stack,
            "stack_depth": len(current_stack),
        }


@app.post("/v1/context/skill_end")
async def context_skill_end(req: ContextSkillEndRequest):
    """skill_end → close the innermost open scope, if its id matches child_scope_id.

    Strict LIFO: the caller must pass the id of the current stack-top scope.
    """
    async with _scope_lock_cm(req.user_id, req.agent_id, req.scope_id, op="skill_end"):
        return await _context_skill_end_locked(req)


async def _context_skill_end_locked(req):
    ws = _get_workspace(req.user_id, req.agent_id)
    root_path = _root_scope_path(req.scope_id)
    requested_id = (req.child_scope_id or "").strip()
    active_stack = read_active_stack_projection(
        ws.operational_store,
        root_scope_id=req.scope_id,
        root_scope_path=root_path,
    )
    stack = active_stack["frames"]

    if not requested_id:
        return {
            "ok": False,
            "error_code": "missing_scope_id",
            "error": "child_scope_id is required",
            "requested_scope_id": requested_id,
            "actual_stack_top": _stack_top_id(stack),
            "stack": stack,
            "stack_depth": len(stack),
        }

    try:
        # Root-level ``skill_end`` is not a summary staging API. ``summary.md``
        # belongs to the scope that is actually closed, and it must equal that
        # close report exactly.
        if not stack:
            return {
                "ok": False,
                "error_code": "stack_empty",
                "error": "No open scope to close (active scope stack is empty at root level)",
                "requested_scope_id": requested_id,
                "actual_stack_top": None,
                "stack": [],
                "stack_depth": 0,
            }

        active_path = active_stack["active_scope_path"]
        top_id = active_stack["top_scope_id"]
        if top_id != requested_id:
            return {
                "ok": False,
                "error_code": "scope_mismatch",
                "error": (
                    f"scope_id mismatch: you tried to close '{requested_id}' "
                    f"but current stack top is '{top_id}'. "
                    "Close the innermost scope first (LIFO)."
                ),
                "requested_scope_id": requested_id,
                "actual_stack_top": top_id,
                "stack_top": top_id,
                "stack": stack,
                "stack_depth": len(stack),
            }

        await _context_event_writer(ws).skill_scope_closed(
            _event_write_context(
                req,
                root_scope_id=req.scope_id,
                root_scope_path=root_path,
            ),
            scope_id=top_id,
            report=req.report,
        )
        await ws.complete_child_scope_projection(active_path, req.report)
        stack = stack[1:]
        _write_active_stack_projection(
            ws,
            root_scope_id=req.scope_id,
            frames=stack,
            reason="skill_end",
        )
        log_cortex(
            "skill.ended",
            user_id=req.user_id,
            agent_id=req.agent_id,
            root_scope_id=req.scope_id,
            child_scope_id=top_id,
            stack_depth=len(stack),
        )
        return {
            "ok": True,
            "child_path": active_path,
            "scope_id": top_id,
            "summary": req.report[:500],
            "stack": stack,
            "stack_depth": len(stack),
        }
    except Exception as e:
        logger.error("skill_end failed: %s", e)
        return {
            "ok": False,
            "error_code": "skill_end_exception",
            "error": str(e),
            "requested_scope_id": requested_id,
            "actual_stack_top": _stack_top_id(stack),
            "stack": stack,
            "stack_depth": len(stack),
        }
