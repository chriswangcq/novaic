    20	interface ParsedContent {
    21	  text: string;
    22	  attachments?: Attachment[];
    23	}
    24	
    25	export function parseMessageContent(
    26	  content: string | { text?: string; attachments?: MessageAttachmentContent[] } | null | undefined,
    27	  messageId: string,
    28	): ParsedContent {
    29	  if (!content) return { text: '' };
    30	
    31	  if (typeof content === 'object') {
    32	    const text = content.text ?? '';
    33	    const attachments = content.attachments?.map((a, i) => buildAttachment(a, messageId, i));
    34	    return { text, attachments: attachments?.length ? attachments : undefined };
    35	  }
    36	
    37	  if (typeof content === 'string') {
    38	    try {
    39	      const parsed = JSON.parse(content);
    40	      if (typeof parsed === 'object' && parsed !== null && ('text' in parsed || 'attachments' in parsed)) {
    41	        const text = parsed.text ?? '';
    42	        const attachments = parsed.attachments?.map((a: MessageAttachmentContent, i: number) =>
    43	          buildAttachment(a, messageId, i)
    44	        );
    45	        return { text, attachments: attachments?.length ? attachments : undefined };
    46	      }
    47	    } catch { /* invalid message content shape */ }
    48	    return { text: '' };
    49	  }
    50	
    51	  return { text: '' };
    52	}

    54	function buildAttachment(
    55	  a: MessageAttachmentContent,
    56	  messageId: string,
    57	  i: number,
    58	): Attachment {

    61	    id:        `att-${messageId}-${i}`,
    62	    name:      a.filename,
