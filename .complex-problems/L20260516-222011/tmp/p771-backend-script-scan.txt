## start-backends.sh
#!/usr/bin/env bash
# start-backends.sh - Start the local desktop backend subset from source.
#
# Run this script BEFORE launching NovAIC.app when you only need the local
# desktop subset: Gateway, Queue Service, Blob Service, and runtime workers.
# It does not start Entangled, Business, Device, Cortex, or Sandboxd. Workers
# that call those services need them running separately.
#
# Usage:
#   ./start-backends.sh              # start the local subset
#   ./start-backends.sh --stop       # stop the local subset
#   ./start-backends.sh --status     # show local subset status

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
APP_DIR="$(cd "$SCRIPT_DIR/.." && pwd)"
ROOT="$(cd "$APP_DIR/.." && pwd)"

# ─── Ports (must match Tauri app constants) ───────────────────────────────────
PORT_GATEWAY=19999
PORT_BUSINESS=19998
PORT_QUEUE_SERVICE=19997
PORT_CORTEX=19996
PORT_BLOB_SERVICE=19995

DATA_DIR="$HOME/Library/Application Support/com.novaic.app"
LOG_DIR="$DATA_DIR/logs"

# Colors
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
RED='\033[0;31m'
NC='\033[0m'

mkdir -p "$LOG_DIR"

# ─── Helpers ──────────────────────────────────────────────────────────────────

python_for_repo() {
    local repo_dir="$1"
    if [ -f "$repo_dir/.venv/bin/python" ]; then
        echo "$repo_dir/.venv/bin/python"
    elif [ -f "$repo_dir/venv/bin/python" ]; then
        echo "$repo_dir/venv/bin/python"
    else
        echo "python3"
    fi
}

pip_for_repo() {
    local repo_dir="$1"
    if [ -f "$repo_dir/.venv/bin/pip" ]; then
        echo "$repo_dir/.venv/bin/pip"
    elif [ -f "$repo_dir/venv/bin/pip" ]; then
        echo "$repo_dir/venv/bin/pip"
    else
        echo "pip3"
    fi
}

# Install/sync requirements for a repo (quiet, only shows output on failure).
ensure_deps_installed() {
    local repo_dir="$1"
    local req_file="$repo_dir/requirements.txt"
    if [ ! -f "$req_file" ]; then
        return 0
    fi
    local pip
    pip=$(pip_for_repo "$repo_dir")
    # Run from the repo dir so that relative -e paths (e.g. ../novaic-shared-*) resolve correctly.
    if ! (cd "$repo_dir" && "$pip" install -q -r requirements.txt 2>&1); then
        echo -e "  ${YELLOW}⚠ pip install failed for $(basename "$repo_dir"), check deps${NC}"
    fi
}

port_in_use() {
    lsof -ti :"$1" >/dev/null 2>&1
}

wait_port() {
    local port="$1" name="$2" secs="${3:-15}"
    local i=0
    while [ $i -lt $((secs * 4)) ]; do
        if port_in_use "$port"; then
            echo -e "  ${GREEN}✓ $name ready on :$port${NC}"
            return 0
        fi
        sleep 0.25
        i=$((i + 1))
    done
    echo -e "  ${YELLOW}⚠ $name did not bind :$port within ${secs}s${NC}"
}

# ─── Stop ─────────────────────────────────────────────────────────────────────

stop_backends() {
    echo -e "${YELLOW}Stopping local backend subset...${NC}"
    local patterns=(
        "novaic-gateway"
        "novaic-agent-runtime"
        "novaic-blob-service"
        "main_gateway.py"
        "main_novaic.py"
        "main_blob_service.py"
    )
    for pattern in "${patterns[@]}"; do
        pkill -9 -f "$pattern" 2>/dev/null && echo "  Killed: $pattern" || true
    done
    # Kill by port
    for port in $PORT_GATEWAY $PORT_QUEUE_SERVICE \
                $PORT_BLOB_SERVICE; do
        local pid
        pid=$(lsof -ti :"$port" 2>/dev/null || true)
        if [ -n "$pid" ]; then
            kill -9 "$pid" 2>/dev/null && echo "  Killed PID $pid on :$port" || true
        fi
    done
    echo -e "${GREEN}Done.${NC}"
}

# ─── Status ───────────────────────────────────────────────────────────────────

show_status() {
    echo "Local backend subset status:"
    local services=(
        "$PORT_GATEWAY:Gateway"
        "$PORT_QUEUE_SERVICE:Queue Service"
        "$PORT_BLOB_SERVICE:Blob Service"
    )
    for entry in "${services[@]}"; do
        local port="${entry%%:*}" name="${entry##*:}"
        if port_in_use "$port"; then
            echo -e "  ${GREEN}● $name (:$port)${NC}"
        else
            echo -e "  ${RED}○ $name (:$port)${NC}"
        fi
    done
}

if [ -d "$GW_DIR" ]; then

    PY=$(python_for_repo "$GW_DIR")
    "$PY" "$GW_DIR/main_gateway.py" \
        --host 127.0.0.1 --port "$PORT_GATEWAY" \
        --data-dir "$DATA_DIR" \
        --queue-service-url "$QS_URL" \
        --blob-service-url "$BLOB_URL" \
        --blob-upload-url "$BLOB_URL" \
        >> "$LOG_DIR/gateway.log" 2>&1 &
    wait_port "$PORT_GATEWAY" "Gateway" 30
else
    echo -e "  ${YELLOW}⚠ novaic-gateway not found at $GW_DIR${NC}"

AR_DIR="$ROOT/novaic-agent-runtime"

        --data-dir "$DATA_DIR" \
        >> "$LOG_DIR/queue-service.log" 2>&1 &
    wait_port "$PORT_QUEUE_SERVICE" "Queue Service"
else
    echo -e "  ${YELLOW}⚠ novaic-agent-runtime not found at $AR_DIR${NC}"
fi

# 3. Blob Service (novaic-blob-service)
BLOB_DIR="$ROOT/novaic-blob-service"
if [ -d "$BLOB_DIR" ]; then
    echo "Starting Blob Service..."
    ensure_deps_installed "$BLOB_DIR"
    PY=$(python_for_repo "$BLOB_DIR")
    "$PY" "$BLOB_DIR/main_blob_service.py" \
        --host 127.0.0.1 --port "$PORT_BLOB_SERVICE" \
        --data-dir "$DATA_DIR" \
        >> "$LOG_DIR/blob-service.log" 2>&1 &
    wait_port "$PORT_BLOB_SERVICE" "Blob Service"
else
    echo -e "  ${YELLOW}⚠ novaic-blob-service not found at $BLOB_DIR${NC}"
fi

# 4. Workers (task, saga, durable outbox, health, scheduler)
if [ -d "$AR_DIR" ]; then
    echo "Starting Workers..."
    PY=$(python_for_repo "$AR_DIR")

    # Task workers (2 control + 2 execution)
    for pool in control execution; do
        for i in 1 2; do
            "$PY" "$AR_DIR/main_novaic.py" task-worker \
                --business-url "$BIZ_URL" \
                --queue-service-url "$QS_URL" \
                --cortex-url "$CORTEX_URL" \
                --pool "$pool" \
                --num-workers 1 \
                --data-dir "$DATA_DIR" \
                >> "$LOG_DIR/task-worker-${pool}-${i}.log" 2>&1 &
        done
    done

    # Saga workers (2)
    for i in 1 2; do
        "$PY" "$AR_DIR/main_novaic.py" saga-worker \
            --queue-service-url "$QS_URL" \
            --cortex-url "$CORTEX_URL" \
            --max-concurrent 4 \
            --data-dir "$DATA_DIR" \
            >> "$LOG_DIR/saga-worker-${i}.log" 2>&1 &
    done

    "$PY" "$AR_DIR/main_novaic.py" session-outbox-worker \
        --data-dir "$DATA_DIR" \
        >> "$LOG_DIR/session-outbox-worker.log" 2>&1 &

    "$PY" "$AR_DIR/main_novaic.py" saga-outbox-worker \
        --data-dir "$DATA_DIR" \
        >> "$LOG_DIR/saga-outbox-worker.log" 2>&1 &

    # Health monitor
    "$PY" "$AR_DIR/main_novaic.py" health \
        --queue-service-url "$QS_URL" \
        --data-dir "$DATA_DIR" \
        >> "$LOG_DIR/health.log" 2>&1 &
