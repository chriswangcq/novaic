    40	def _bounded_body(value: object, *, limit: int) -> tuple[object, int, bool]:
    41	    if not isinstance(value, str):
    42	        return value, 0, False
    43	    total = len(value)
    44	    if total <= limit:
    45	        return value, total, False
    46	    return (
    47	        value[:limit] + f"\n\n... [truncated log body to {limit} chars; original chars={total}]",
    48	        total,
    49	        True,
    50	    )

    53	def _with_bounded_bodies(log: dict, *, body_limit: int) -> dict:
    54	    bounded = dict(log)
    55	    limit_input = body_limit if isinstance(body_limit, int) else _DEFAULT_LOG_BODY_LIMIT_CHARS
    56	    limit = max(0, min(limit_input, _MAX_LOG_BODY_LIMIT_CHARS))
    57	    for key in ("request_body", "response_body"):
    58	        value, total, truncated = _bounded_body(bounded.get(key), limit=limit)
    59	        bounded[key] = value
    60	        bounded[f"{key}_chars"] = total
    61	        bounded[f"{key}_truncated"] = truncated
    62	    return bounded
