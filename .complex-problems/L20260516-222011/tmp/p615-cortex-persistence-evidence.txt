    29	    """Read and eventually append ContextEvents for one root-scope stream."""
    30	
    31	    def __init__(
    32	        self,
    33	        workspace: "Workspace",
    34	        *,
    35	        clock: Callable[[], str] | None = None,
    36	        id_provider: Callable[[], str] | None = None,

    42	    @staticmethod
    43	    def event_log_path(root_scope_path: str) -> str:
    44	        root_scope_path = root_scope_path.rstrip("/")
    45	        if not root_scope_path:
    46	            raise ContextEventStoreError("root_scope_path must be non-empty")
    47	        return f"{root_scope_path}/context_events/events.jsonl"
    48	
    49	    async def read_events(self, root_scope_path: str) -> list[ContextEvent]:
    50	        path = self.event_log_path(root_scope_path)
    51	        try:
    52	            raw = await self._workspace.read(path)
    53	        except KeyError:
    54	            return []
    55	
    56	        events: list[ContextEvent] = []
    57	        for lineno, line in enumerate(raw.splitlines(), start=1):
    58	            if not line.strip():
    59	                continue
    60	            try:
    61	                row = json.loads(line)
    62	            except json.JSONDecodeError as exc:
    63	                raise ContextEventStoreError(
    64	                    f"invalid ContextEvent JSON at {path}:{lineno}: {exc.msg}"
    65	                ) from exc
    66	            if not isinstance(row, dict):
    67	                raise ContextEventStoreError(
    68	                    f"invalid ContextEvent row at {path}:{lineno}: expected object"
    69	                )
    70	            try:
    71	                events.append(ContextEvent.from_dict(row))
    72	            except ContextEventError as exc:
    73	                raise ContextEventStoreError(
    74	                    f"invalid ContextEvent envelope at {path}:{lineno}: {exc}"
    75	                ) from exc
    76	        return events

    78	    async def append_event(
    79	        self,
    80	        root_scope_path: str,
    81	        *,
    82	        stream_id: str,
    83	        root_scope_id: str,
    84	        actor: str,
    85	        type: str,
    86	        payload: dict,
    87	        idempotency_key: str | None = None,
    88	    ) -> ContextEvent:
    89	        existing = await self.read_events(root_scope_path)
    90	        if idempotency_key:
    91	            candidate = _canonical_semantic_json(
    92	                stream_id=stream_id,
    93	                root_scope_id=root_scope_id,
    94	                actor=actor,
    95	                type=type,
    96	                payload=payload,
    97	            )
    98	            for event in existing:
    99	                if event.idempotency_key != idempotency_key:
   100	                    continue
   101	                if event.canonical_semantic_json() == candidate:
   102	                    return event
   103	                raise ContextEventIdempotencyConflict(
   104	                    f"idempotency key {idempotency_key!r} already maps to a different ContextEvent"
   105	                )
   106	
   107	        if self._clock is None:
   108	            raise ContextEventStoreError("ContextEventStore append requires clock provider")
   109	        if self._id_provider is None:
   110	            raise ContextEventStoreError("ContextEventStore append requires id provider")
   111	
   112	        event = ContextEvent.create(
   113	            event_id=self._id_provider(),
   114	            stream_id=stream_id,
   115	            root_scope_id=root_scope_id,
   116	            seq=len(existing) + 1,
   117	            idempotency_key=idempotency_key,
   118	            occurred_at=self._clock(),
   119	            actor=actor,
   120	            type=type,
   121	            payload=payload,
   122	        )
   123	        await self._workspace._sys_append_line(
   124	            self.event_log_path(root_scope_path),
   125	            event.to_json_line(),
   126	        )
   127	        return event

    67	    """Prepare LLM context from the authoritative ContextEvent stream."""
    68	
    69	    def __init__(
    70	        self,
    71	        *,
    72	        workspace: "Workspace",
    73	        root_scope_path: str,
    74	        config: CompactConfig | None = None,
    75	        token_counter: TokenCounter | None = None,
    76	        event_store: ContextEventStore | None = None,
    77	    ) -> None:
    78	        self._workspace = workspace
    79	        self._root_scope_path = root_scope_path.rstrip("/")
    80	        self._config = config or CompactConfig()
    81	        self._token_counter = token_counter
    82	        self._event_store = event_store or ContextEventStore(workspace)

    84	    async def prepare(self) -> ContextEventPreparedContext:
    85	        if not self._root_scope_path:
    86	            return ContextEventPreparedContext(
    87	                messages=[],
    88	                stack=[],
    89	                estimated_tokens=0,
    90	                usage_ratio=0.0,
    91	                applied_seq=0,
    92	                projection=ContextProjectionSnapshot(),
    93	            )
    94	
    95	        events = await self._event_store.read_events(self._root_scope_path)
    96	        if not events:
    97	            raise ContextEventReadModelResetRequired(self._root_scope_path)
    98	        projection = project_context_events(events)
    99	        messages = budget_compact(
   100	            projection.messages,
   101	            self._config,
   102	            counter=self._token_counter,
   103	        )
   104	        estimated_tokens = count_all_tokens(messages, self._token_counter)
   105	        usage_ratio = (
   106	            estimated_tokens / self._config.context_window
   107	            if self._config.context_window > 0
   108	            else 0.0
   109	        )
   110	        return ContextEventPreparedContext(
   111	            messages=messages,
   112	            stack=_top_first_stack(projection.stack),
   113	            estimated_tokens=estimated_tokens,
   114	            usage_ratio=usage_ratio,
   115	            applied_seq=projection.applied_seq,
   116	            projection=projection,
   117	        )

   120	def _top_first_stack(stack: list[dict]) -> list[dict]:
   121	    """Return stack frames in API/control order: current top first."""
   122	
   123	    frames: list[dict] = []
   124	    for frame in reversed(stack):

   128	        frames.append(normalized)
