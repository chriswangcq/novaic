    @classmethod
    def from_dict(cls, data: Mapping[str, Any]) -> "SandboxdExecuteRequest":
        env_raw = data.get("env", {})
        if not isinstance(env_raw, Mapping):
            raise ValueError("env must be an object")
        env: dict[str, str] = {}
        for key, value in env_raw.items():
            if not isinstance(key, str) or not isinstance(value, str):
                raise ValueError("env keys and values must be strings")
            env[key] = value

        mount_raw = data.get("mount")
        mount: SandboxBindMount | None = None
        if mount_raw is not None:
            if not isinstance(mount_raw, Mapping):
                raise ValueError("mount must be an object or null")
            mount = SandboxBindMount(
                source_root=_require_str(mount_raw, "source_root"),
                mount_point=_require_str(mount_raw, "mount_point"),
                stable_cwd=_require_str(mount_raw, "stable_cwd"),
            )

        return cls(
            command=_require_str(data, "command"),
            cwd=_require_str(data, "cwd"),
            env=env,
            timeout=_require_int(data, "timeout"),
            display_command=_optional_str(data, "display_command"),
            mount=mount,
        )

    def to_spec(self) -> SandboxExecSpec:
        return SandboxExecSpec(
            command=self.command,
            cwd=self.cwd,
            env=dict(self.env),
            timeout=self.timeout,
            display_command=self.display_command,
            mount=self.mount,
        )

    def to_dict(self) -> JsonObject:
        mount: JsonObject | None = None
        if self.mount is not None:
            mount = {
                "source_root": self.mount.source_root,
                "mount_point": self.mount.mount_point,
                "stable_cwd": self.mount.stable_cwd,
            }
        return {
            "command": self.command,
            "cwd": self.cwd,
            "env": dict(self.env),
            "timeout": self.timeout,
            "display_command": self.display_command,
            "mount": mount,
        }

def _detect_logicalfs_capabilities() -> LogicalFSCapabilities:
    return LogicalFSCapabilities(
        provider="sandboxd-logicalfs",
        true_stable_path_mount=True,
        outer_command_path_adapter=False,
        notes=(
            "Cortex projects Workspace data into LogicalFS and asks sandboxd to "
            "bind-mount the materialized view at /cortex. Cortex does not "
            "execute local shell processes."
        ),
    )



        The writable tree can contain large generated artifacts. Shell startup
        must not become proportional to all historical RW files, so only direct
        root files and explicit shared/current-subagent work directories are
        mounted by default. Any new file the command writes under /rw is still
        persisted on release.

            await self._add_tree(files, logical_prefix)

        await self._add_tree(files, f"/rw/subagents/{subagent_id}/")

    async def _workspace_snapshot(
        self,
        *,
        generated_rw_files: Mapping[str, bytes],
        extra_env: Mapping[str, str] | None,
    ) -> LogicalFSSnapshot:
        files: dict[tuple[bool, str], LogicalFSFile] = {}

        # Keep shell startup bounded. The previous implementation recursively
        # materialized all of /ro, which made every shell call proportional to
        # total agent history. Shell does not need the runtime tool-schema
        # store or the installed skill library as local files; capability CLIs
        # are generated into /cortex/bin and call Cortex/Business APIs when
        # they need agent-harness data. Historical or broad reads must be
        # explicit API reads, not implicit full mounts.
        await self._add_file_if_exists(files, "/ro/config/engine.json")
        await self._add_file_if_exists(files, "/ro/config/tools/_index.json")

        scope_id = str((extra_env or {}).get("NOVAIC_SCOPE_ID") or "").strip()
        wake_scope_path = str((extra_env or {}).get("NOVAIC_WAKE_SCOPE_PATH") or "").strip().rstrip("/")
        if scope_id:
            root_path = f"/ro/active/{scope_id}"
            await self._add_file_if_exists(files, f"{root_path}/meta.json")
            await self._add_file_if_exists(files, f"{root_path}/steps/_index.jsonl")
            await self._add_file_if_exists(files, f"{root_path}/context_events/events.jsonl")
            writable_dirs=(f"{STABLE_CORTEX_RW}/system",),
        )

    def _execution_env(self, extra_env: Mapping[str, str] | None, token: str | None) -> LogicalFSEnv:
        values = {
            "NOVAIC_API": self._cortex_api_url,
            "CORTEX_FS_PROVIDER": self.capabilities.provider,
            "CORTEX_TRUE_STABLE_PATH_MOUNT": "1" if self.capabilities.true_stable_path_mount else "0",
        }
        values.update(_capability_env(dict(extra_env or {})))
        if token is not None:
            values["NOVAIC_TOKEN"] = token
        return LogicalFSEnv(values)

    def _generated_env_file(self, token: str | None) -> bytes:
        env_json: dict[str, str] = {
            "api": self._cortex_api_url,
            "fs_provider": self.capabilities.provider,
        }
        if token is not None:
            env_json["token"] = token
        return _json.dumps(env_json).encode("utf-8")

    async def acquire_view(
        self,
        root: Path,
        *,
        cwd: str | None,
        extra_env: dict[str, str] | None,
    ) -> LogicalFSView:
        token = self._token_factory() if self._token_factory is not None else None
        snapshot = await self._workspace_snapshot(
            generated_rw_files={".novaic_env.json": self._generated_env_file(token)},
            extra_env=extra_env,
        )

        if self._max_sync_bytes is not None:
            total = self._snapshot_bytes(snapshot)
            if total > self._max_sync_bytes:
                raise ValueError(
                    f"logicalfs sync would read {total} bytes from workspace files, "
                    f"exceeds max_sync_bytes={self._max_sync_bytes}"
                )

        substrate = self._provider.materialize(
            snapshot=snapshot,
            root=root,
            layout=self._rw_layout(extra_env),
            env=self._execution_env(extra_env, token),
            cwd=cwd,
        )
        _write_capability_scripts(substrate.bin)
        return LogicalFSView(substrate=substrate, capabilities=self.capabilities)

    def process_mount_plan(self, view: LogicalFSView) -> SandboxBindMount:
        if not view.capabilities.true_stable_path_mount:
            raise RuntimeError(view.capabilities.notes)
        return SandboxBindMount(
            source_root=str(view.root.resolve()),
            mount_point=STABLE_CORTEX_ROOT,
            stable_cwd=view.stable_cwd,
        )

def _validate_spec(spec: ProcessSpec) -> None:
    if spec.timeout <= 0:
        raise ValueError("timeout must be positive")
    if spec.mount is not None:
        if not spec.mount.source_root.is_dir():
            raise ValueError("mount.source_root must be an existing directory")
        if not spec.mount.mount_point.startswith("/"):
            raise ValueError("mount.mount_point must be absolute")
        if not spec.mount.stable_cwd.startswith("/"):
            raise ValueError("mount.stable_cwd must be absolute")
        return
    if not spec.cwd.is_dir():
        raise ValueError("cwd must be an existing directory")


def _sdk_request_to_core_spec(request: SandboxdExecuteRequest) -> ProcessSpec:
    spec = request.to_spec()
    mount: BindMountPlan | None = None
    if spec.mount is not None:
        mount = BindMountPlan(
            source_root=Path(spec.mount.source_root),

        )
    return ProcessSpec(
        command=spec.command,
        cwd=Path(spec.cwd),
        env=dict(spec.env),
        timeout=spec.timeout,
        display_command=spec.display_command,
        mount=mount,
    )

from __future__ import annotations

import os
import shlex
import shutil
from pathlib import Path


def mount_namespace_available() -> bool:
    """Return whether the current process can create private bind mounts."""
    return (
        os.name == "posix"
        and hasattr(os, "geteuid")
        and os.geteuid() == 0
        and shutil.which("unshare") is not None
        and shutil.which("mount") is not None
    )


def build_bind_mount_command(
    *,
    command: str,
    source_root: str | Path,
    mount_point: str,
    stable_cwd: str,
) -> str:
    """Build a shell command that bind-mounts source_root at mount_point.

    The caller owns all policy: what root to mount, where to mount it, and which
    stable cwd/command should run inside that private namespace.
    """
    source = shlex.quote(str(Path(source_root).resolve()))
    mount = shlex.quote(mount_point)
    cwd = shlex.quote(stable_cwd)
    command_arg = shlex.quote(command)
    inner = (
        "set -e; "
        f"mkdir -p {mount}; "
        f"mount --bind {source} {mount}; "
        f"cd {cwd}; "
        f"exec /bin/sh -c {command_arg}"
    )
    return f"unshare --mount --propagation private /bin/bash -lc {shlex.quote(inner)}"
