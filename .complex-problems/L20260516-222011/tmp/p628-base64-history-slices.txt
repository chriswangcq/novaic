def _display_durable_payload(result: dict[str, Any]) -> dict[str, Any] | None:
    file_url = str(result.get("file_url") or "").strip()
    if not file_url:
        return None

    size = result.get("size", 0)
    mime_type = str(result.get("mime_type") or "").strip()
    if not mime_type.startswith("image/"):
        return None

    mcp_content = result.get("_mcp_content")
    if not isinstance(mcp_content, list):
        mcp_content = []

    for item in mcp_content:
        if not isinstance(item, dict) or item.get("type") != "image":
            continue
        mime_type = str(item.get("mimeType") or item.get("mime_type") or mime_type)

    message = str(result.get("message") or f"Loaded {file_url}")
    llm_mcp_content: list[dict[str, Any]] = [
        {"type": "text", "text": str(item.get("text"))}
        for item in mcp_content
        if isinstance(item, dict) and item.get("type") == "text" and item.get("text")
    ]
    if not llm_mcp_content:
        llm_mcp_content.append({"type": "text", "text": message})
    llm_mcp_content.append({
        "type": "image_ref",
        "file_url": file_url,
        "mimeType": mime_type,
        "size": size,
    })

    display_files = [{
        "url": file_url,
        "filename": file_url.rstrip("/").rsplit("/", 1)[-1] or "file",
        "modality": "image",
        "mime_type": mime_type,
        "size": size,
    }]

    return {
        "version": _DURABLE_PAYLOAD_VERSION,
        "llm_content": {
            "success": result.get("success") is not False,
            "message": message,
            "file_url": file_url,
            "mime_type": mime_type,
            "size": size,
            "_mcp_content": llm_mcp_content,
            "display_files": display_files,
        },
        "raw": {
            "type": "display_result",
            "file_url": file_url,
            "mime_type": mime_type,
            "size": size,
        },
    }


_SHELL_STREAM_PREVIEW_CHARS = 1600
_SHELL_TEXT_LIMIT_CHARS = 4000


def _preview_shell_stream(value: Any, *, limit: int = _SHELL_STREAM_PREVIEW_CHARS) -> tuple[str, bool, int]:
    text = "" if value is None else str(value)
    if len(text) <= limit:
        return text, False, len(text)
    return text[:limit] + f"\n... [truncated {len(text) - limit} chars]", True, len(text)


def _shell_result_text(result: dict[str, Any]) -> tuple[str, dict[str, Any]]:
    exit_code = result.get("exit_code")
    stdout_preview, stdout_truncated, stdout_chars = _preview_shell_stream(result.get("stdout", ""))
    stderr_preview, stderr_truncated, stderr_chars = _preview_shell_stream(result.get("stderr", ""))
    files_changed = result.get("files_changed") if isinstance(result.get("files_changed"), list) else []

    lines = [f"exit_code={exit_code}"]
    if stdout_preview:
        lines.extend(["stdout:", stdout_preview])
    if stderr_preview:
        lines.extend(["stderr:", stderr_preview])
    if files_changed:
        lines.append("files_changed:")
        lines.extend(f"- {path}" for path in files_changed[:40])
        if len(files_changed) > 40:
            lines.append(f"... [truncated {len(files_changed) - 40} changed files]")
    if len(lines) == 1:
        lines.append("shell completed with no stdout/stderr")

    diagnostics = {
        "shell_result": {
            "exit_code": exit_code,
            "stdout_chars": stdout_chars,
            "stderr_chars": stderr_chars,
            "stdout_truncated": stdout_truncated,
            "stderr_truncated": stderr_truncated,
            "files_changed_count": len(files_changed),
        },
    }
    return "\n".join(lines), diagnostics


def _shell_result_output(result: dict[str, Any]) -> dict[str, Any]:
    exit_code = result.get("exit_code")
    ok = exit_code == 0
    text, diagnostics = _shell_result_text(result)
    output = tool_text(
        text,
        ok=ok,
        text_limit=_SHELL_TEXT_LIMIT_CHARS,
        diagnostics=diagnostics,
    ).to_dict()
    if not ok:
        stderr = str(result.get("stderr") or "").strip()
        output["error"] = stderr[:500] if stderr else f"shell exited with code {exit_code}"
    output[_DURABLE_PAYLOAD_KEY] = {
        "version": _DURABLE_PAYLOAD_VERSION,
        "llm_content": {
            key: value
            for key, value in output.items()
            if key != _DURABLE_PAYLOAD_KEY
        },
        "raw": {
            "type": "shell_result",
            "exit_code": exit_code,
            "stdout": result.get("stdout", ""),
            "stderr": result.get("stderr", ""),
            "files_changed": (
                result.get("files_changed")
                if isinstance(result.get("files_changed"), list)
                else []
            ),
        },
    }
    return output


def _ok(scope_id, round_id, tool_call_id, tool_name, result):
    tool_success = _executor_success(result)
    public_output = _tool_output_dict(result, tool_success=tool_success, tool_name=tool_name)
    content = json.dumps(public_output, ensure_ascii=False)
    response = {
        "success": True, "status": "completed",
        "tool_success": tool_success,
        "tool_status": "completed" if tool_success else "error",
        "scope_id": scope_id, "round_id": round_id,
        "tool_call_id": tool_call_id, "tool_name": tool_name,
        "content": content, "error": None,
    }
    if isinstance(result, dict) and isinstance(result.get(_DURABLE_PAYLOAD_KEY), dict):
        response["durable_payload"] = result[_DURABLE_PAYLOAD_KEY]
    elif tool_name == "display" and isinstance(result, dict):
        durable_payload = _display_durable_payload(result)
        if durable_payload is not None:
            response["durable_payload"] = durable_payload
    return response

const BINARY_BASE64_PREFIX_PATTERN = /^(\/9j\/|iVBORw0KGgo|R0lGODlh|UklGR)/;
const RAW_PAYLOAD_DETAIL = '输出包含较大的原始文件数据，已在监控中隐藏。';

function isNearBottom(element: HTMLElement): boolean {
  return element.scrollHeight - element.scrollTop - element.clientHeight <= BOTTOM_FOLLOW_THRESHOLD_PX;
}

export function isActivityTimelineRecordRunning(record: ActivityTimelineRecord): boolean {
  return record.status === 'running';
}

function isFailed(record: ActivityTimelineRecord): boolean {
  return record.status === 'failed' || record.status === 'error';
}

function recordSearchText(record: ActivityTimelineRecord): string {
  return `${record.kind} ${record.tool ?? ''} ${record.title} ${record.text ?? ''}`.toLowerCase();
}

function isAgentctlImReadRecord(record: ActivityTimelineRecord): boolean {
  return normalizedTool(record) === 'shell' && AGENTCTL_IM_READ_PATTERN.test(recordSearchText(record));
}

function isAgentctlImReplyRecord(record: ActivityTimelineRecord): boolean {
  return normalizedTool(record) === 'shell' && AGENTCTL_IM_REPLY_PATTERN.test(recordSearchText(record));
}

export function activityTimelinePublicTitle(record: ActivityTimelineRecord): string {
  const fingerprint = `${record.kind} ${record.tool ?? ''} ${record.title}`.toLowerCase();
  if (fingerprint.includes('skill_end')) return '已保存本轮上下文';
  if (isAgentctlImReadRecord(record)) return '读取了你的消息';
  if (isAgentctlImReplyRecord(record)) return '已回复你';
  if (fingerprint.includes('message') || fingerprint.includes('消息')) return '读取了你的消息';
  if (record.phase === 'reasoning') {
    const reasoningText = record.text?.toLowerCase() ?? '';
    if (
      AGENTCTL_IM_READ_PATTERN.test(reasoningText) ||
      reasoningText.includes('读取消息') ||
      reasoningText.includes('新消息')
    ) {
      return '正在查看你的消息';
    }
    if (AGENTCTL_IM_REPLY_PATTERN.test(reasoningText) || reasoningText.includes('回复')) {
      return '正在组织回复';
    }
    if (
      reasoningText.includes('skill_end') ||
      reasoningText.includes('summary') ||
      reasoningText.includes('上下文') ||
      reasoningText.includes('结束本轮')
    ) {
      return '正在整理上下文';
    }
    return isActivityTimelineRecordRunning(record) ? '正在思考' : '完成思考';
  }
  if (record.phase === 'summary') return '已保存本轮上下文';
  return record.title || PHASE_FALLBACK_LABELS[record.phase];
}

function publicFullDetail(record: ActivityTimelineRecord): string | undefined {
  const text = record.text?.trim();
  if (!text) return record.has_payload ? '结果已保存，需要时可以继续查看细节' : undefined;
  if (looksLikeRawPayloadText(text)) return RAW_PAYLOAD_DETAIL;
  if (/[{}]/.test(text)) return undefined;
  if (DEBUG_DETAIL_PATTERN.test(text)) return undefined;
  if (record.phase === 'action' && record.tool) return undefined;
  return text.replace(/\r\n/g, '\n');
}

function looksLikeRawPayloadText(text: string): boolean {
  if (DATA_IMAGE_BASE64_PATTERN.test(text)) return true;
  const compact = text.replace(/\s+/g, '');
  if (compact.length < 120) return false;
  if (BINARY_BASE64_PREFIX_PATTERN.test(compact)) return true;
  return compact.length > 800 && /^[A-Za-z0-9+/=]+$/.test(compact) && /[+/=]/.test(compact);
}
