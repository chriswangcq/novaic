## novaic-sandbox-service/sandbox_service/main.py
"""Independent sandbox execution service."""

from __future__ import annotations

import argparse
import logging
from pathlib import Path
from typing import Any

from fastapi import FastAPI, HTTPException, Request

from sandbox_sdk import (
    SandboxExecResult,
    SandboxdExecuteRequest,
    SandboxdExecuteResponse,
)
from sandbox_service.core.process import AsyncProcessRunner, BindMountPlan, ProcessRunResult, ProcessSpec

logger = logging.getLogger("sandboxd")


def _validate_spec(spec: ProcessSpec) -> None:
    if spec.timeout <= 0:
        raise ValueError("timeout must be positive")
    if spec.mount is not None:
        if not spec.mount.source_root.is_dir():
            raise ValueError("mount.source_root must be an existing directory")
        if not spec.mount.mount_point.startswith("/"):
            raise ValueError("mount.mount_point must be absolute")
        if not spec.mount.stable_cwd.startswith("/"):
            raise ValueError("mount.stable_cwd must be absolute")
        return
    if not spec.cwd.is_dir():
        raise ValueError("cwd must be an existing directory")


def _sdk_request_to_core_spec(request: SandboxdExecuteRequest) -> ProcessSpec:
    spec = request.to_spec()
    mount: BindMountPlan | None = None
    if spec.mount is not None:
        mount = BindMountPlan(
            source_root=Path(spec.mount.source_root),
            mount_point=spec.mount.mount_point,
            stable_cwd=spec.mount.stable_cwd,
        )
    return ProcessSpec(
        command=spec.command,
        cwd=Path(spec.cwd),
        env=dict(spec.env),
        timeout=spec.timeout,
        display_command=spec.display_command,
        mount=mount,
    )


def _to_response(result: ProcessRunResult) -> SandboxdExecuteResponse:
    return SandboxdExecuteResponse.from_result(
        SandboxExecResult(
            stdout=result.stdout,
            stderr=result.stderr,
            exit_code=result.exit_code,
            duration_s=result.duration_s,
        )
    )


def create_app(*, runner: Any | None = None):
    app = FastAPI(title="NovaIC Sandboxd", version="1.0.0")
    app.state.runner = runner or AsyncProcessRunner()

    @app.get("/health")
    @app.get("/api/health")
    def health() -> dict[str, str]:
        return {
            "status": "ok",
            "service": "sandboxd",
            "contract_version": "sandboxd/v1",
        }

    @app.post("/v1/execute")
    async def execute(request: Request) -> dict[str, Any]:
        try:
            payload = await request.json()
            if not isinstance(payload, dict):
                raise ValueError("request body must be a JSON object")
            execute_request = SandboxdExecuteRequest.from_dict(payload)
            spec = _sdk_request_to_core_spec(execute_request)
            _validate_spec(spec)
        except ValueError as exc:
            raise HTTPException(status_code=400, detail=str(exc)) from exc

        result = await app.state.runner.run(spec)
        return _to_response(result).to_dict()

    return app


def main() -> None:
    parser = argparse.ArgumentParser(description="NovaIC sandboxd service")
    parser.add_argument("--host", default="127.0.0.1", help="Host to bind")
    parser.add_argument("--port", type=int, default=19994, help="Port to listen")
    args = parser.parse_args()

    logging.basicConfig(
        level=logging.INFO,
        format="[%(asctime)s] [%(levelname)s] %(message)s",
        datefmt="%H:%M:%S",
    )
    logger.info("Starting sandboxd on %s:%s", args.host, args.port)

    import uvicorn

    uvicorn.run(create_app(), host=args.host, port=args.port)


if __name__ == "__main__":
    main()

## novaic-sandbox-service/sandbox_service/core/filesystem.py
"""Small filesystem helpers for sandboxd internals."""

from __future__ import annotations

import re
from pathlib import Path
from typing import Mapping

FileStats = dict[str, tuple[int, int]]


def file_stats(root: Path) -> FileStats:
    """Return relative posix path -> (mtime_ns, size) for regular files."""
    out: FileStats = {}
    if not root.is_dir():
        return out
    for path in root.rglob("*"):
        if path.is_file():
            stat = path.stat()
            rel = path.relative_to(root).as_posix()
            out[rel] = (stat.st_mtime_ns, stat.st_size)
    return out


def changed_relative_paths(before: FileStats, after: FileStats) -> list[str]:
    """Return created, modified, and removed relative paths."""
    changed: list[str] = []
    for rel in before:
        if rel not in after:
            changed.append(rel)
    for rel, stats in after.items():
        prev = before.get(rel)
        if prev is None or prev != stats:
            changed.append(rel)
    return sorted(set(changed))


def safe_path_component(value: str | None, default: str) -> str:
    raw = (value or default).strip() or default
    safe = re.sub(r"[^A-Za-z0-9_.-]+", "-", raw).strip(".-")
    return safe or default


def resolve_cwd_under_root(root: Path, cwd: str | None) -> Path:
    root_resolved = root.resolve()
    if cwd is None or cwd == "" or cwd == ".":
        return root_resolved
    candidate = (root_resolved / cwd).resolve()
    try:
        candidate.relative_to(root_resolved)
    except ValueError as exc:
        raise ValueError(f"cwd must be relative to sandbox root: {cwd!r}") from exc
    return candidate


def is_keep_placeholder(path: str, data: bytes) -> bool:
    return path.endswith(".keep") and len(data) == 0


def sanitize_paths(text: str, replacements: Mapping[str, str]) -> str:
    """Replace actual filesystem paths with stable display paths."""
    out = text
    for actual, stable in replacements.items():
        out = out.replace(actual, stable)
    return out

## novaic-sandbox-service/sandbox_service/core/mount_namespace.py
"""Linux mount namespace helpers for sandboxd internals."""

from __future__ import annotations

import os
import shlex
import shutil
from pathlib import Path


def mount_namespace_available() -> bool:
    """Return whether the current process can create private bind mounts."""
    return (
        os.name == "posix"
        and hasattr(os, "geteuid")
        and os.geteuid() == 0
        and shutil.which("unshare") is not None
        and shutil.which("mount") is not None
    )


def build_bind_mount_command(
    *,
    command: str,
    source_root: str | Path,
    mount_point: str,
    stable_cwd: str,
) -> str:
    """Build a shell command that bind-mounts source_root at mount_point.

    The caller owns all policy: what root to mount, where to mount it, and which
    stable cwd/command should run inside that private namespace.
    """
    source = shlex.quote(str(Path(source_root).resolve()))
    mount = shlex.quote(mount_point)
    cwd = shlex.quote(stable_cwd)
    command_arg = shlex.quote(command)
    inner = (
        "set -e; "
        f"mkdir -p {mount}; "
        f"mount --bind {source} {mount}; "
        f"cd {cwd}; "
        f"exec /bin/sh -c {command_arg}"
    )
    return f"unshare --mount --propagation private /bin/bash -lc {shlex.quote(inner)}"

## novaic-sandbox-service/sandbox_service/core/process.py
"""Async process execution substrate used inside sandboxd."""

from __future__ import annotations

import asyncio
import time
from dataclasses import dataclass
from pathlib import Path
from typing import Mapping

from sandbox_service.core.mount_namespace import build_bind_mount_command


@dataclass(frozen=True)
class BindMountPlan:
    source_root: Path
    mount_point: str
    stable_cwd: str


@dataclass(frozen=True)
class ProcessSpec:
    command: str
    cwd: Path
    env: Mapping[str, str]
    timeout: int
    display_command: str = ""
    mount: BindMountPlan | None = None


@dataclass(frozen=True)
class ProcessRunResult:
    stdout: bytes
    stderr: bytes
    exit_code: int
    duration_s: float


class AsyncProcessRunner:
    """Run a shell command with explicit cwd/env/timeout inputs."""

    async def run(self, spec: ProcessSpec) -> ProcessRunResult:
        command = spec.command
        cwd = spec.cwd
        if spec.mount is not None:
            command = build_bind_mount_command(
                command=spec.command,
                source_root=spec.mount.source_root,
                mount_point=spec.mount.mount_point,
                stable_cwd=spec.mount.stable_cwd,
            )
            cwd = spec.mount.source_root

        t0 = time.monotonic()
        proc = await asyncio.create_subprocess_shell(
            command,
            cwd=str(cwd),
            env=dict(spec.env),
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
        )

        try:
            stdout_b, stderr_b = await asyncio.wait_for(
                proc.communicate(),
                timeout=spec.timeout,
            )
            exit_code = proc.returncode if proc.returncode is not None else -1
        except asyncio.TimeoutError:
            try:
                proc.kill()
            except ProcessLookupError:
                pass
            await proc.wait()
            stdout_b = b""
            stderr_b = f"Command timed out after {spec.timeout}s".encode()
            exit_code = -1

        return ProcessRunResult(
            stdout=stdout_b,
            stderr=stderr_b,
            exit_code=exit_code,
            duration_s=time.monotonic() - t0,
        )

## novaic-sandbox-sdk/sandbox_sdk/client.py
"""HTTP client for sandboxd."""

from __future__ import annotations

import asyncio
import json
import urllib.error
import urllib.request
from dataclasses import dataclass

from sandbox_sdk.contracts import SandboxdExecuteRequest, SandboxdExecuteResponse
from sandbox_sdk.types import SandboxExecResult, SandboxExecSpec


class SandboxdClientError(RuntimeError):
    pass


@dataclass(frozen=True)
class SandboxdClient:
    base_url: str
    request_timeout: int = 30

    async def execute(self, spec: SandboxExecSpec) -> SandboxExecResult:
        return await asyncio.to_thread(self._execute_sync, spec)

    def _execute_sync(self, spec: SandboxExecSpec) -> SandboxExecResult:
        url = self.base_url.rstrip("/") + "/v1/execute"
        payload = json.dumps(
            SandboxdExecuteRequest.from_spec(spec).to_dict(),
            ensure_ascii=False,
        ).encode("utf-8")
        request = urllib.request.Request(
            url,
            data=payload,
            headers={"Content-Type": "application/json"},
            method="POST",
        )
        try:
            with urllib.request.urlopen(request, timeout=self.request_timeout) as response:
                raw = response.read()
        except urllib.error.HTTPError as exc:
            body = exc.read().decode("utf-8", errors="replace")
            raise SandboxdClientError(f"sandboxd returned HTTP {exc.code}: {body}") from exc
        except urllib.error.URLError as exc:
            raise SandboxdClientError(f"sandboxd request failed: {exc.reason}") from exc
        except TimeoutError as exc:
            raise SandboxdClientError("sandboxd request timed out") from exc

        try:
            data = json.loads(raw.decode("utf-8"))
            return SandboxdExecuteResponse.from_dict(data).to_result()
        except (json.JSONDecodeError, UnicodeDecodeError, ValueError) as exc:
            raise SandboxdClientError("sandboxd returned an invalid response") from exc

from __future__ import annotations

import base64
from dataclasses import dataclass
from typing import Any, Mapping

from sandbox_sdk.types import SandboxBindMount, SandboxExecResult, SandboxExecSpec

JsonObject = dict[str, Any]


def _require_str(data: Mapping[str, Any], key: str) -> str:
    value = data.get(key)
    if not isinstance(value, str):
        raise ValueError(f"{key} must be a string")
    return value


def _require_int(data: Mapping[str, Any], key: str) -> int:
    value = data.get(key)
    if isinstance(value, bool) or not isinstance(value, int):
        raise ValueError(f"{key} must be an integer")
    return value


def _optional_str(data: Mapping[str, Any], key: str, default: str = "") -> str:
    value = data.get(key, default)
    if not isinstance(value, str):
        raise ValueError(f"{key} must be a string")
    return value


@dataclass(frozen=True)
class SandboxdExecuteRequest:
    command: str
    cwd: str
    env: Mapping[str, str]
    timeout: int
    display_command: str = ""
    mount: SandboxBindMount | None = None

    @classmethod
    def from_spec(cls, spec: SandboxExecSpec) -> "SandboxdExecuteRequest":
        return cls(
            command=spec.command,
            cwd=spec.cwd,
            env=dict(spec.env),
            timeout=spec.timeout,
            display_command=spec.display_command,
            mount=spec.mount,
        )

    @classmethod
    def from_dict(cls, data: Mapping[str, Any]) -> "SandboxdExecuteRequest":
        env_raw = data.get("env", {})
        if not isinstance(env_raw, Mapping):
            raise ValueError("env must be an object")
        env: dict[str, str] = {}
        for key, value in env_raw.items():
            if not isinstance(key, str) or not isinstance(value, str):
                raise ValueError("env keys and values must be strings")
            env[key] = value

        mount_raw = data.get("mount")
        mount: SandboxBindMount | None = None
        if mount_raw is not None:
            if not isinstance(mount_raw, Mapping):
                raise ValueError("mount must be an object or null")
            mount = SandboxBindMount(
                source_root=_require_str(mount_raw, "source_root"),
                mount_point=_require_str(mount_raw, "mount_point"),
                stable_cwd=_require_str(mount_raw, "stable_cwd"),
            )

        return cls(
            command=_require_str(data, "command"),
            cwd=_require_str(data, "cwd"),
            env=env,
            timeout=_require_int(data, "timeout"),
            display_command=_optional_str(data, "display_command"),
            mount=mount,
        )

    def to_spec(self) -> SandboxExecSpec:
        return SandboxExecSpec(
            command=self.command,
            cwd=self.cwd,
            env=dict(self.env),
            timeout=self.timeout,
            display_command=self.display_command,
            mount=self.mount,
        )

    def to_dict(self) -> JsonObject:
        mount: JsonObject | None = None
        if self.mount is not None:
            mount = {
                "source_root": self.mount.source_root,
                "mount_point": self.mount.mount_point,
                "stable_cwd": self.mount.stable_cwd,
            }
        return {
            "command": self.command,
            "cwd": self.cwd,
            "env": dict(self.env),
            "timeout": self.timeout,
            "display_command": self.display_command,
            "mount": mount,
        }


@dataclass(frozen=True)
class SandboxdExecuteResponse:
    stdout: bytes
    stderr: bytes
    exit_code: int
    duration_s: float

    @classmethod
    def from_result(cls, result: SandboxExecResult) -> "SandboxdExecuteResponse":
        return cls(
            stdout=result.stdout,
            stderr=result.stderr,
            exit_code=result.exit_code,
            duration_s=result.duration_s,
        )

    @classmethod
    def from_dict(cls, data: Mapping[str, Any]) -> "SandboxdExecuteResponse":
        duration = data.get("duration_s")
        if not isinstance(duration, (int, float)) or isinstance(duration, bool):
            raise ValueError("duration_s must be a number")
        return cls(
            stdout=base64.b64decode(_require_str(data, "stdout_b64")),
            stderr=base64.b64decode(_require_str(data, "stderr_b64")),
            exit_code=_require_int(data, "exit_code"),
            duration_s=float(duration),
        )

    def to_result(self) -> SandboxExecResult:
        return SandboxExecResult(
            stdout=self.stdout,
            stderr=self.stderr,
            exit_code=self.exit_code,
            duration_s=self.duration_s,
        )

    def to_dict(self) -> JsonObject:
        return {
            "stdout_b64": base64.b64encode(self.stdout).decode("ascii"),
            "stderr_b64": base64.b64encode(self.stderr).decode("ascii"),
            "exit_code": self.exit_code,
            "duration_s": self.duration_s,
        }

## novaic-logicalfs/logicalfs/local.py
"""Local LogicalFS materialization/diff provider."""

from __future__ import annotations

from pathlib import Path
from typing import Mapping

from logicalfs.contracts import (
    LogicalFSEnv,
    LogicalFSLayout,
    LogicalFSPatch,
    LogicalFSPatchFile,
    LogicalFSSnapshot,
    LogicalFSView,
    StableLogicalPaths,
)

FileStats = dict[str, tuple[int, int]]


def _file_stats(root: Path) -> FileStats:
    out: FileStats = {}
    if not root.is_dir():
        return out
    for path in root.rglob("*"):
        if path.is_file():
            stat = path.stat()
            rel = path.relative_to(root).as_posix()
            out[rel] = (stat.st_mtime_ns, stat.st_size)
    return out


def _changed_relative_paths(before: Mapping[str, tuple[int, int]], after: Mapping[str, tuple[int, int]]) -> list[str]:
    changed: list[str] = []
    for rel in before:
        if rel not in after:
            changed.append(rel)
    for rel, stats in after.items():
        prev = before.get(rel)
        if prev is None or prev != stats:
            changed.append(rel)
    return sorted(set(changed))


def _resolve_cwd_under_root(root: Path, cwd: str | None) -> Path:
    root_resolved = root.resolve()
    if cwd is None or cwd == "" or cwd == ".":
        return root_resolved
    candidate = (root_resolved / cwd).resolve()
    try:
        candidate.relative_to(root_resolved)
    except ValueError as exc:
        raise ValueError(f"cwd must be relative to logical rw root: {cwd!r}") from exc
    return candidate


def _stable_path_for_actual(path: Path, *, view_root: Path, ro: Path, rw: Path, stable: StableLogicalPaths) -> str:
    resolved = path.resolve()
    for actual_root, stable_root in [
        (rw.resolve(), stable.rw),
        (ro.resolve(), stable.ro),
        (view_root.resolve(), stable.root),
    ]:
        try:
            rel = resolved.relative_to(actual_root)
        except ValueError:
            continue
        if rel.as_posix() == ".":
            return stable_root
        suffix = rel.as_posix()
        return stable_root if not suffix else f"{stable_root}/{suffix}"
    return str(resolved)


def _logical_rw_path(rel: str, stable: StableLogicalPaths) -> str:
    return f"{stable.rw}/{rel}" if rel else stable.rw


def sanitize_paths(text: str, replacements: Mapping[str, str]) -> str:
    out = text
    for actual, stable in replacements.items():
        out = out.replace(actual, stable)
    return out


class LocalLogicalFSProvider:
    """Materialize explicit snapshots into local RO/RW views and observe patches."""

    def materialize(
        self,
        *,
        snapshot: LogicalFSSnapshot,
        root: Path,
        layout: LogicalFSLayout,
        env: LogicalFSEnv,
        cwd: str | None = None,
    ) -> LogicalFSView:
        ro = root / "ro"
        rw = root / "rw"
        bin_dir = root / "bin"
        ro.mkdir(parents=True)
        rw.mkdir(parents=True)
        bin_dir.mkdir(parents=True)

        for item in snapshot.files:
            clean = item.path.strip("/")
            if not clean:
                continue
            base = rw if item.writable else ro
            dest = base / clean
            dest.parent.mkdir(parents=True, exist_ok=True)
            dest.write_bytes(item.data)

        for stable_path in [*layout.writable_env.values(), *layout.writable_dirs]:
            if stable_path.startswith(layout.paths.rw + "/"):
                rel = stable_path.removeprefix(layout.paths.rw + "/")
                (rw / rel).mkdir(parents=True, exist_ok=True)

        proc_cwd = _resolve_cwd_under_root(rw, cwd)
        base_env = {
            "PATH": f"{layout.paths.bin}:/usr/local/bin:/usr/bin:/bin",
            "RO": layout.paths.ro,
            "RW": layout.paths.rw,
            "CORTEX_RO": layout.paths.ro,
            "CORTEX_RW": layout.paths.rw,
            "CORTEX_ROOT": layout.paths.root,
            "CORTEX_FS_PROVIDER": "logicalfs",
            "CORTEX_TRUE_STABLE_PATH_MOUNT": "1",
            "HOME": layout.paths.rw,
        }
        base_env.update(dict(layout.writable_env))
        base_env.update(dict(env.values))
        return LogicalFSView(
            root=root,
            ro=ro,
            rw=rw,
            bin=bin_dir,
            cwd=proc_cwd,
            stable_cwd=_stable_path_for_actual(
                proc_cwd,
                view_root=root,
                ro=ro,
                rw=rw,
                stable=layout.paths,
            ),
            env=base_env,
            before_rw=_file_stats(rw),
            paths=layout.paths,
        )

    def observe_patch(self, view: LogicalFSView) -> LogicalFSPatch:
        after = _file_stats(view.rw)
        upserts: list[LogicalFSPatchFile] = []
        deletes: list[str] = []
        for rel in _changed_relative_paths(view.before_rw, after):
            logical = _logical_rw_path(rel, view.paths)
            if rel not in after:
                deletes.append(logical)
            else:
                upserts.append(LogicalFSPatchFile(path=logical, data=(view.rw / rel).read_bytes()))
        return LogicalFSPatch(upserts=tuple(upserts), deletes=tuple(sorted(deletes)))

    def sanitize_output(self, text: str, view: LogicalFSView) -> str:
        return sanitize_paths(
            text,
            {
                str(view.ro.resolve()): view.paths.ro,
                str(view.rw.resolve()): view.paths.rw,
                str(view.root.resolve()): view.paths.root,
            },
        )

## novaic-logicalfs/logicalfs/authority.py
"""Generic live LogicalFS file authority.

This module owns realtime logical file operations over /ro and /rw. It is
business-agnostic: callers provide explicit namespace/layout configuration and
a generic object-store adapter. Cortex semantics, agents, scopes, Blob Service,
and process execution live outside this boundary.
"""

from __future__ import annotations

import asyncio
import json
from dataclasses import dataclass
from typing import Iterable, Protocol, Sequence

from logicalfs.contracts import LogicalFSDirectoryEntry, LogicalObjectInfo


class LogicalObjectStore(Protocol):
    async def put(self, key: str, data: bytes, content_type: str = "") -> None: ...

    async def get(self, key: str) -> bytes: ...

    async def exists(self, key: str) -> bool: ...

    async def delete(self, key: str) -> None: ...

    async def list_objects(self, prefix: str, delimiter: str = "/") -> Sequence[LogicalObjectInfo]: ...

    async def list_recursive(self, prefix: str) -> Sequence[str]: ...

    async def move_prefix(self, src_prefix: str, dst_prefix: str) -> int: ...


@dataclass(frozen=True)
class LogicalFileAuthorityLayout:
    owner_prefix: str
    ro_prefix: str = "ro"
    rw_prefix: str = "rw"

    def object_prefix(self) -> str:
        return self.owner_prefix.strip("/")


def _normalize_logical_path(path: str) -> str:
    if not isinstance(path, str):
        raise ValueError("path must be a string")
    p = path.strip()
    if "\x00" in p:
        raise ValueError("path must not contain NUL")
    if not p.startswith("/"):
        p = "/" + p
    while "//" in p:
        p = p.replace("//", "/")
    for segment in p.split("/"):
        if segment == "..":
            raise ValueError("path must not contain '..'")
    if p in {"/ro", "/rw"}:
        return p + "/"
    if p.startswith("/ro/") or p.startswith("/rw/"):
        return p
    raise ValueError(f"path must start with /ro/ or /rw/: {path!r}")


def logical_to_object_key(layout: LogicalFileAuthorityLayout, path: str) -> str:
    logical = _normalize_logical_path(path)
    prefix = layout.object_prefix()
    base = f"{prefix}/" if prefix else ""
    ro = layout.ro_prefix.strip("/")
    rw = layout.rw_prefix.strip("/")
    if logical.startswith("/ro/"):
        rel = logical.removeprefix("/ro/").lstrip("/")
        return f"{base}{ro}/{rel}"
    if logical.startswith("/rw/"):
        rel = logical.removeprefix("/rw/").lstrip("/")
        return f"{base}{rw}/{rel}"
    raise ValueError(f"path must start with /ro/ or /rw/: {path!r}")


class StoreBackedLogicalFileAuthority:
    """Live /ro and /rw authority backed by a generic object store."""

    def __init__(self, store: LogicalObjectStore, layout: LogicalFileAuthorityLayout) -> None:
        self._store = store
        self._layout = layout

    @property
    def layout(self) -> LogicalFileAuthorityLayout:
        return self._layout

    def key(self, path: str) -> str:
        return logical_to_object_key(self._layout, path)

    def _prefix_key(self, path: str) -> str:
        key = self.key(path)
        return key if key.endswith("/") else key + "/"

    async def write_bytes(self, path: str, data: bytes, content_type: str = "") -> None:
        await self._store.put(self.key(path), data, content_type)

    async def write_text(self, path: str, content: str) -> None:
        await self.write_bytes(path, content.encode("utf-8"))

    async def write_json(self, path: str, obj: dict) -> None:
        await self.write_bytes(
            path,
            json.dumps(obj, ensure_ascii=False, indent=2).encode("utf-8"),
            "application/json",
        )

    async def read_bytes(self, path: str) -> bytes:
        return await self._store.get(self.key(path))

    async def read_text(self, path: str, encoding: str = "utf-8") -> str:
        return (await self.read_bytes(path)).decode(encoding)

    async def exists(self, path: str) -> bool:
        return await self._store.exists(self.key(path))

    async def delete(self, path: str) -> None:
        await self._store.delete(self.key(path))

    async def append_line(self, path: str, line: str) -> None:
        await self.append_lines(path, line.rstrip("\n") + "\n")

    async def append_lines(self, path: str, lines: str) -> None:
        key = self.key(path)
        try:
            prev = await self._store.get(key)
        except KeyError:
            prev = b""
        await self._store.put(key, prev + lines.encode("utf-8"))

    async def list_tree_bytes(
        self,
        path: str,
        *,
        concurrency: int = 16,
    ) -> list[tuple[str, bytes]]:
        store_prefix = self._prefix_key(path)
        keys = list(await self._store.list_recursive(store_prefix))

        async def read_one(key: str) -> tuple[str, bytes] | None:
            if not key.startswith(store_prefix):
                return None
            rel = key[len(store_prefix):].lstrip("/")
            if not rel:
                return None
            return rel, await self._store.get(key)

        if concurrency <= 1:
            results = [await read_one(key) for key in keys]
        else:
            sem = asyncio.Semaphore(max(1, int(concurrency)))

            async def limited_read(key: str) -> tuple[str, bytes] | None:
                async with sem:
                    return await read_one(key)

            results = await asyncio.gather(*(limited_read(key) for key in keys))
        return [item for item in results if item is not None]

    async def list_dir(self, path: str) -> list[LogicalFSDirectoryEntry]:
        logical_base = _normalize_logical_path(path)
        if not logical_base.endswith("/"):
            logical_base += "/"
        store_prefix = self._prefix_key(logical_base)
        infos = await self._store.list_objects(store_prefix)
        entries: list[LogicalFSDirectoryEntry] = []
        for item in infos:
            name = item.key.rstrip("/").split("/")[-1]
            logical = logical_base + name
            if item.is_prefix:
                logical = logical.rstrip("/") + "/"
            entries.append(
                LogicalFSDirectoryEntry(
                    name=name,
                    path=logical,
                    is_dir=item.is_prefix,
                    size=item.size,
                )
            )
        return entries

    async def move_tree(self, src_path: str, dst_path: str) -> int:
        return await self._store.move_prefix(self._prefix_key(src_path), self._prefix_key(dst_path))

    async def list_object_keys(self, path: str) -> list[str]:
        return list(await self._store.list_recursive(self._prefix_key(path)))

    async def initialize_layout(self, logical_dirs: Iterable[str]) -> None:
        for logical_dir in logical_dirs:
            directory = _normalize_logical_path(logical_dir).rstrip("/")
            await self.write_bytes(f"{directory}/.keep", b"")

## novaic-logicalfs/logicalfs/blob_store.py
"""Blob Service object-store adapter below LogicalFS.

Blob Service is a cheap byte/object server. LogicalFS owns realtime /ro and /rw
file authority above this adapter; callers should not use this class as a
semantic workspace API.
"""

from __future__ import annotations

from typing import Any
from urllib.parse import quote

from common.http.clients import internal_async_client

from logicalfs.contracts import LogicalObjectInfo


class BlobObjectStore:
    """Async object-store adapter over Blob Service object APIs."""

    def __init__(
        self,
        *,
        base_url: str,
        tenant_id: str,
        namespace: str = "cortex-store",
        timeout: float = 10.0,
        client_service_name: str = "logicalfs",
    ) -> None:
        if not base_url or not base_url.strip():
            raise ValueError("base_url is required")
        if not tenant_id or not tenant_id.strip():
            raise ValueError("tenant_id is required")
        self._base_url = base_url.rstrip("/")
        self._tenant_id = tenant_id
        self._namespace = namespace
        self._timeout = timeout
        self._client_service_name = client_service_name

    def _headers(self) -> dict[str, str]:
        return {"X-Tenant-ID": self._tenant_id}

    def _object_url(self, key: str) -> str:
        return f"{self._base_url}/v1/objects/{quote(self._namespace)}/{quote(key, safe='/')}"

    async def put(self, key: str, data: bytes, content_type: str = "") -> None:
        headers = self._headers()
        if content_type:
            headers["Content-Type"] = content_type
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.put(self._object_url(key), headers=headers, content=data)
            resp.raise_for_status()

    async def get(self, key: str) -> bytes:
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.get(self._object_url(key), headers=self._headers())
        if resp.status_code == 404:
            raise KeyError(key)
        resp.raise_for_status()
        return resp.content

    async def exists(self, key: str) -> bool:
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.get(
                f"{self._object_url(key)}/info",
                headers=self._headers(),
            )
        if resp.status_code == 404:
            return False
        resp.raise_for_status()
        return True

    async def delete(self, key: str) -> None:
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.delete(self._object_url(key), headers=self._headers())
        if resp.status_code == 404:
            return
        resp.raise_for_status()

    async def list_objects(self, prefix: str, delimiter: str = "/") -> list[LogicalObjectInfo]:
        if delimiter != "/":
            raise NotImplementedError("BlobObjectStore only supports delimiter='/'")
        data = await self._list(prefix=prefix, delimiter=delimiter, recursive=False)
        return [
            LogicalObjectInfo(
                key=str(item["key"]),
                size=int(item.get("size") or 0),
                is_prefix=bool(item.get("is_prefix")),
            )
            for item in data.get("objects", [])
        ]

    async def list_recursive(self, prefix: str) -> list[str]:
        data = await self._list(prefix=prefix, delimiter="/", recursive=True)
        return sorted(str(item["key"]) for item in data.get("objects", []))

    async def _list(self, *, prefix: str, delimiter: str, recursive: bool) -> dict[str, Any]:
        params = {
            "namespace": self._namespace,
            "prefix": prefix,
            "delimiter": delimiter,
            "recursive": "true" if recursive else "false",
        }
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.get(
                f"{self._base_url}/v1/objects/_list",
                params=params,
                headers=self._headers(),
            )
            resp.raise_for_status()
            return resp.json()

    async def move(self, src: str, dst: str) -> None:
        data = await self.get(src)
        await self.put(dst, data)
        await self.delete(src)

    async def move_prefix(self, src_prefix: str, dst_prefix: str) -> int:
        body = {
            "tenant_id": self._tenant_id,
            "namespace": self._namespace,
            "src_prefix": src_prefix,
            "dst_prefix": dst_prefix,
        }
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.post(f"{self._base_url}/v1/objects/_move-prefix", json=body)
            resp.raise_for_status()
            data = resp.json()
        return int(data.get("moved") or 0)


__all__ = ["BlobObjectStore"]
