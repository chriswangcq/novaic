def _human_size(n: int) -> str:
    size = float(max(n, 0))
    for unit in ("B", "KB", "MB", "GB"):
        if size < 1024:
            return f"{int(size)}B" if unit == "B" else f"{size:.1f}{unit}"
        size /= 1024
    return f"{size:.1f}TB"

    # ── Shell execution (3-tool model) ───────────────────────────────────

    def shell_exec(
        self,
        command: str,
        timeout: int = 30,
        *,
        capability_env: dict[str, str] | None = None,
    ) -> dict | None:
        """Run a shell command. Transport errors surface as exit_code=-1 so
        the tool result path remains consistent (shell errors == bad exit).
        """
        try:
            http_timeout = max(float(self._timeout), float(timeout or _DEFAULT_TIMEOUT) + float(self._timeout))
            result = self._post("/v1/internal/shell", {
                **self._tenant(),
                "command": command,
                "timeout": timeout,
                "env": {str(k): str(v) for k, v in dict(capability_env or {}).items() if v is not None},
            }, timeout=http_timeout)
            return result
        except Exception as e:
            logger.error("[CortexBridge] shell_exec transport error: %s", e)
            return {"stdout": "", "stderr": str(e), "exit_code": -1, "files_changed": []}

    # ── Steps ────────────────────────────────────────────────────────────

def test_cortex_bridge_shell_exec_sends_explicit_capability_env(monkeypatch):

    posted = {}

    def _post(path, payload, **kwargs):
        posted["path"] = path
        posted["payload"] = payload
        posted["kwargs"] = kwargs
        return {"stdout": "ok", "stderr": "", "exit_code": 0, "files_changed": []}

from __future__ import annotations

import json

from task_queue.handlers.tool_handlers import handle_tool_execute


class _Bridge:
    def __init__(self, result):
        self.result = result
        self.calls = []

    def shell_exec(self, command, timeout, *, capability_env=None):
        self.calls.append((command, timeout, dict(capability_env or {})))
        return self.result


def _payload(args=None):
    return {
        "scope_id": "wake-1",
        "round_id": "round-1",
        "tool_call_id": "shell-call-1",
        "tool_name": "shell",
        "arguments": args or {"command": "printf hi", "timeout": 9},
        "agent_id": "agent-1",
        "user_id": "user-1",
    }


def _run_shell(monkeypatch, shell_result, args=None):
    bridge = _Bridge(shell_result)
    monkeypatch.setattr(
        "task_queue.utils.cortex_bridge.get_cortex_bridge",
        lambda *_args, **_kwargs: bridge,
    )
    result = handle_tool_execute(_payload(args), ctx={})
    return result, json.loads(result["content"]), bridge

    assert "hello" in content["text"]
    assert "files_changed:" in content["text"]
