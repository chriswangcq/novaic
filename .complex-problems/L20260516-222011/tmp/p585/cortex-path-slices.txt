    29	def _tool_output_artifact_line(artifact: Dict[str, Any]) -> str:
    30	    uri = str(artifact.get("uri") or "").strip()
    31	    modality = str(artifact.get("modality") or "resource").strip() or "resource"
    32	    name = str(artifact.get("name") or "").strip()
    33	    mime_type = str(artifact.get("mime_type") or "").strip()
    34	    summary = str(artifact.get("summary") or "").strip()
    35	    size = _tool_output_size_label(artifact.get("size_bytes"))
    36	    label = name or uri or "artifact"
    37	    details = ", ".join(part for part in (modality, mime_type, size) if part)
    38	    prefix = f"[Artifact: {label}"
    39	    if details:
    40	        prefix += f" ({details})"
    41	    prefix += "]"

    43	    access = artifact.get("access") if isinstance(artifact.get("access"), dict) else {}
    44	    display = access.get("display") if isinstance(access.get("display"), dict) else {}
    45	    display_args = display.get("args") if isinstance(display.get("args"), dict) else {}
    46	    display_url = display_args.get("file_url") or uri
    47	
    48	    parts = [prefix]
    49	    if summary:
    50	        parts.append(summary)
    51	    if display_url:
    52	        parts.append(f'Access: display(file_url="{display_url}")')
    53	    return " — ".join(parts)

    57	    text_parts: list[str] = []
    58	    text = raw.get("text")

    62	    artifacts = raw.get("artifacts") if isinstance(raw.get("artifacts"), list) else []

    79	        text_parts.append(f"[Error] {error}")

    90	def parse_tool_result(raw: Any) -> Dict[str, Any]:
    91	    """
    92	    解析工具原始结果为三要素格式。
    93	
    94	    支持的输入格式：
    95	    1. 统一格式: {text, files_created, display_files}
    96	    2. MCP 格式: {content: [{type, text/data}]}
    97	    3. 纯文本: str
    98	
    99	    Returns:
   100	        {"text": "...", "files_created": [...], "display_files": [...]}
   101	    """
   102	    if raw is None:
   103	        return {"text": "", "files_created": [], "display_files": []}
   104	
   105	    if isinstance(raw, str):
   106	        stripped = raw.strip()
   107	        if stripped.startswith("{") or stripped.startswith("["):
   108	            try:
   109	                decoded = json.loads(stripped)
   110	            except json.JSONDecodeError:
   111	                decoded = None
   112	            if isinstance(decoded, (dict, list)):
   113	                return parse_tool_result(decoded)
   114	        return {"text": raw, "files_created": [], "display_files": []}
   115	
   116	    if not isinstance(raw, dict):
   117	        return {"text": str(raw), "files_created": [], "display_files": []}
   118	
   119	    if raw.get("version") == _TOOL_STEP_PAYLOAD_VERSION:
   120	        return parse_tool_result(raw.get("llm_content", ""))
   121	
   122	    if raw.get("version") == "tool-output.v1":
   123	        return _parse_tool_output_v1(raw)
   124	
   125	    # 1. 统一格式（三要素）
   126	    if "display_files" in raw or "files_created" in raw:
   127	        text_parts = []
   128	        if raw.get("text"):
   129	            text_parts.append(str(raw["text"]))
   130	        # 把 files_created 的 URL 写入文本
   131	        files_created = raw.get("files_created") or []
   132	        if files_created and isinstance(files_created, list):
   133	            urls = [f.get("url") for f in files_created if isinstance(f, dict) and f.get("url")]
   134	            if urls:
   135	                text_parts.append(f"file_url: {urls[0]}" if len(urls) == 1 else f"file_urls: {', '.join(urls)}")
   136	        return {
   137	            "text": "\n".join(text_parts),
   138	            "files_created": files_created,
   139	            "display_files": raw.get("display_files") or [],
   140	        }
   141	
   142	    # 2. MCP 格式 content 数组
   143	    content = raw.get("content") or raw.get("_mcp_content")
   144	    if isinstance(content, list):
   145	        text_parts = []
   146	        display_files = []
   147	        for item in content:
   148	            if not isinstance(item, dict):
   149	                continue
   150	            t = item.get("type", "")
   151	            if t == "text":
   152	                text_parts.append(item.get("text", ""))
   153	            elif t == "image":
   154	                # 支持 URL 形式或直接的 data(base64) 形式
   155	                if item.get("url"):
   156	                    display_files.append({
   157	                        "url": item["url"],
   158	                        "filename": "",
   159	                        "modality": "image",
   160	                    })
   161	                elif item.get("data"):
   162	                    mime = item.get("mimeType", "image/png")
   163	                    display_files.append({
   164	                        "url": f"data:{mime};base64,{item['data']}",
   165	                        "filename": "",
   166	                        "modality": "image",
   167	                    })
   168	            elif t == "resource":
   169	                if item.get("url"):
   170	                    display_files.append({
   171	                        "url": item["url"],
   172	                        "filename": "",
   173	                        "modality": "resource",
   174	                    })
   175	        return {
   176	            "text": "\n".join(text_parts),
   177	            "files_created": [],
   178	            "display_files": display_files,
   179	        }
   180	
   181	    # 4. 其他 dict：提取 text 字段或序列化
   182	    if raw.get("text"):
   183	        return {"text": str(raw["text"]), "files_created": [], "display_files": []}
   184	
   185	    # 5. Unknown dict fallback: diagnostic text only, bounded so unexpected
   186	    # payload shapes cannot flood LLM context.
   187	    def safe_default(obj):
   188	        return f"<Unserializable {type(obj).__name__}>"
   189	    try:
   190	        serialized = json.dumps(raw, ensure_ascii=False, default=safe_default)
   191	    except Exception as e:
   192	        serialized = f"[StepResultProjection Parse Error] Failed to serialize object: {str(e)}"
   193	    if len(serialized) > _UNKNOWN_DICT_TEXT_LIMIT:
   194	        serialized = (
   195	            serialized[:_UNKNOWN_DICT_TEXT_LIMIT]
   196	            + f"\n\n... [Truncated unknown tool result JSON length > {_UNKNOWN_DICT_TEXT_LIMIT}]"
   197	        )
   198	    serialized = "[Unknown tool result JSON]\n" + serialized
   199	    return {"text": serialized, "files_created": [], "display_files": []}

   202	def _format_mcp_content(
   203	    result_data: Dict[str, Any],
   204	    *,
   205	    include_display: bool,
   206	    max_text_len: Optional[int] = None,
   207	    empty_text: str = "OK",
   208	) -> str:
   209	    converted = []
   210	
   211	    text = result_data.get("text", "")
   212	    if max_text_len is not None and len(text) > max_text_len:
   213	        text = text[:max_text_len] + f"\n\n... [Truncated due to context budget length > {max_text_len}]"
   214	
   215	    if text:
   216	        converted.append({"type": "text", "text": text})
   217	
   218	    if include_display:
   219	        display_files = result_data.get("display_files", [])
   220	        for f in display_files:
   221	            url = f.get("url", "")
   222	            if not url:
   223	                continue
   224	            if url.startswith("data:"):
   225	                # Data URL
   226	                parts = url.split(",", 1)
   227	                mime = "image/png"
   228	                if len(parts) >= 1 and ";" in parts[0]:
   229	                    mime = parts[0].split(";")[0].replace("data:", "")
   230	                data_b64 = parts[1] if len(parts) > 1 else ""
   231	
   232	                converted.append({
   233	                    "type": "image",
   234	                    "data": data_b64,
   235	                    "mimeType": mime,
   236	                })
   237	            else:
   238	                # Normal URL: represent unresolved display media as text.
   239	                converted.append({"type": "text", "text": f"[Image: {url}]"})
   240	                
   241	    if not converted:
   242	        converted.append({"type": "text", "text": empty_text})
   243	
   244	    return json.dumps({"_mcp_content": converted}, ensure_ascii=False)

   623	    async def read_payload(self, scope_path: str, payload_ref: str) -> Any:
   624	        """Read a raw tool payload by ref from the scope-local payload store."""
   625	        path = f"{scope_path.rstrip('/')}/payloads/{_payload_filename(payload_ref)}"
   626	        try:
   627	            raw = await self.read(path)
   628	        except (KeyError, FileNotFoundError) as exc:
   629	            raise self._payload_read_error(
   630	                payload_ref=payload_ref,
   631	                code="payload_record_missing",
   632	                status="missing",
   633	                message=f"payload record not found for {payload_ref!r}",
   634	            ) from exc
   635	        try:
   636	            record = json.loads(raw)
   637	        except json.JSONDecodeError as exc:
   638	            raise self._payload_read_error(
   639	                payload_ref=payload_ref,
   640	                code="payload_record_corrupt",
   641	                status="corrupt",
   642	                message=f"payload record is not valid JSON for {payload_ref!r}",
   643	            ) from exc
   644	        if record.get("payload_ref") != payload_ref:
   645	            raise self._payload_read_error(
   646	                payload_ref=payload_ref,
   647	                code="payload_ref_mismatch",
   648	                status="corrupt",
   649	                message=f"payload record ref mismatch for {payload_ref!r}",
   650	            )
   651	        if record.get("blob_ref"):
   652	            if self._payload_blob_client is None:
   653	                raise self._payload_read_error(
   654	                    payload_ref=payload_ref,
   655	                    code="payload_blob_client_missing",

   659	            try:
   660	                return await self._payload_blob_client.get_payload(
   661	                    tenant_id=self._agent_id,

   666	                    payload_ref=payload_ref,

   668	                    status="unavailable",
   669	                    message=f"failed to fetch external Cortex payload {payload_ref!r}",
   670	                ) from exc
   671	        return record.get("content")

   700	    async def normalize_step(self, scope_path: str, step: dict) -> dict:
   701	        """Return the durable step shape used by step files and event records.
   702	
   703	        Normalization validates tool-step shape, externalizes any inline
   704	        ``payload`` into the scope payload store, and mirrors the resulting
   705	        ``payload_ref`` into the observation percept. The caller's input is not
   706	        mutated, so API/event paths can safely inspect the normalized step
   707	        before deciding which durable write path to invoke.
   708	        """
   709	        normalized = copy.deepcopy(step)
   710	        step_type = normalized.get("type", "tool")
   711	        step_id = normalized.get("call_id") or normalized.get("id") or "unknown"
   712	        if "/" in str(step_id) or ".." in str(step_id) or "\x00" in str(step_id):
   713	            raise ValueError(f"invalid step id: {step_id!r}")
   714	
   715	        if step_type == "tool" and "result" in normalized:
   716	            raise ValueError("tool steps must use payload/payload_ref, not inline result")
   717	        if step_type == "tool" and not isinstance(normalized.get("observation"), dict):
   718	            raise ValueError("tool steps require an observation percept")
   719	        if "payload" in normalized:
   720	            payload = normalized.pop("payload")
   721	            payload_ref = normalized.get("payload_ref")
   722	            if not payload_ref:
   723	                raise ValueError("payload_ref is required when writing payload")
   724	            stable_step_ref = str(normalized.get("step_ref") or payload_ref)
   725	            actual_payload_ref = await self.write_payload(
   726	                scope_path,
   727	                str(payload_ref),
   728	                payload,
   729	                step_ref=stable_step_ref,
   730	            )
   731	            normalized["step_ref"] = stable_step_ref
   732	            normalized["payload_ref"] = actual_payload_ref
   733	            if isinstance(normalized.get("observation"), dict):
   734	                normalized["observation"]["payload_ref"] = actual_payload_ref
   735
