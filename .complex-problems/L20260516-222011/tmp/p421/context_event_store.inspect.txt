"""Workspace-backed ContextEvent append-only store."""

from __future__ import annotations

import json
from typing import TYPE_CHECKING, Callable

from novaic_cortex.context_events import (
    CONTEXT_EVENT_SCHEMA_VERSION,
    ContextEvent,
    ContextEventError,
    build_stream_id,
    stable_json,
)

if TYPE_CHECKING:
    from novaic_cortex.workspace import Workspace


class ContextEventStoreError(ValueError):
    """Raised when the persisted ContextEvent log cannot be replayed."""


class ContextEventIdempotencyConflict(ContextEventStoreError):
    """Raised when a retry idempotency key maps to a different event body."""


class ContextEventStore:
    """Read and eventually append ContextEvents for one root-scope stream."""

    def __init__(
        self,
        workspace: "Workspace",
        *,
        clock: Callable[[], str] | None = None,
        id_provider: Callable[[], str] | None = None,
    ) -> None:
        self._workspace = workspace
        self._clock = clock
        self._id_provider = id_provider

    @staticmethod
    def event_log_path(root_scope_path: str) -> str:
        root_scope_path = root_scope_path.rstrip("/")
        if not root_scope_path:
            raise ContextEventStoreError("root_scope_path must be non-empty")
        return f"{root_scope_path}/context_events/events.jsonl"

    async def read_events(self, root_scope_path: str) -> list[ContextEvent]:
        path = self.event_log_path(root_scope_path)
        try:
            raw = await self._workspace.read(path)
        except KeyError:
            return []

        events: list[ContextEvent] = []
        for lineno, line in enumerate(raw.splitlines(), start=1):
            if not line.strip():
                continue
            try:
                row = json.loads(line)
            except json.JSONDecodeError as exc:
                raise ContextEventStoreError(
                    f"invalid ContextEvent JSON at {path}:{lineno}: {exc.msg}"
                ) from exc
            if not isinstance(row, dict):
                raise ContextEventStoreError(
                    f"invalid ContextEvent row at {path}:{lineno}: expected object"
                )
            try:
                events.append(ContextEvent.from_dict(row))
            except ContextEventError as exc:
                raise ContextEventStoreError(
                    f"invalid ContextEvent envelope at {path}:{lineno}: {exc}"
                ) from exc
        return events

    async def append_event(
        self,
        root_scope_path: str,
        *,
        stream_id: str,
        root_scope_id: str,
        actor: str,
        type: str,
        payload: dict,
        idempotency_key: str | None = None,
    ) -> ContextEvent:
        existing = await self.read_events(root_scope_path)
        if idempotency_key:
            candidate = _canonical_semantic_json(
                stream_id=stream_id,
                root_scope_id=root_scope_id,
                actor=actor,
                type=type,
                payload=payload,
            )
            for event in existing:
                if event.idempotency_key != idempotency_key:
                    continue
                if event.canonical_semantic_json() == candidate:
                    return event
                raise ContextEventIdempotencyConflict(
                    f"idempotency key {idempotency_key!r} already maps to a different ContextEvent"
                )

        if self._clock is None:
            raise ContextEventStoreError("ContextEventStore append requires clock provider")
        if self._id_provider is None:
            raise ContextEventStoreError("ContextEventStore append requires id provider")

        event = ContextEvent.create(
            event_id=self._id_provider(),
            stream_id=stream_id,
            root_scope_id=root_scope_id,
            seq=len(existing) + 1,
            idempotency_key=idempotency_key,
            occurred_at=self._clock(),
            actor=actor,
            type=type,
            payload=payload,
        )
        await self._workspace._sys_append_line(
            self.event_log_path(root_scope_path),
            event.to_json_line(),
        )
        return event

    async def initialize_root(
        self,
        root_scope_path: str,
        *,
        user_id: str,
        agent_id: str,
        root_scope_id: str,
        subagent_id: str,
        scope_name: str,
        actor: str = "cortex",
    ) -> ContextEvent:
        stream_id = build_stream_id(
            user_id=user_id,
            agent_id=agent_id,
            root_scope_id=root_scope_id,
        )
        return await self.append_event(
            root_scope_path,
            stream_id=stream_id,
            root_scope_id=root_scope_id,
            actor=actor,
            type="RootInitialized",
            payload={
                "root_scope_id": root_scope_id,
                "subagent_id": subagent_id,
                "scope_name": scope_name,
            },
            idempotency_key=f"context-root:init:{stream_id}",
        )


def _canonical_semantic_json(
    *,
    stream_id: str,
    root_scope_id: str,
    actor: str,
    type: str,
    payload: dict,
) -> str:
    return stable_json(
        {
            "schema_version": CONTEXT_EVENT_SCHEMA_VERSION,
            "stream_id": stream_id,
            "root_scope_id": root_scope_id,
            "actor": actor,
            "type": type,
            "payload": payload,
        }
    )
