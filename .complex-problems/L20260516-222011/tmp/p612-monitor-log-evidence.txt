    93	        return value
    94	
    95	    item_type = value.get("type")
    96	    if item_type == "image_url" and isinstance(value.get("image_url"), dict):
    97	        image_url = {
    98	            key: _redact_message_media_for_log(item)
    99	            for key, item in value["image_url"].items()
   100	        }
   101	        url = image_url.get("url")
   102	        if isinstance(url, str):
   103	            image_url["url"] = _redact_data_url_for_log(url)
   104	        return {
   105	            **{
   106	                key: _redact_message_media_for_log(item)
   107	                for key, item in value.items()
   108	                if key != "image_url"
   109	            },
   110	            "image_url": image_url,
   111	        }
   112	
   113	    if item_type == "image" and isinstance(value.get("source"), dict):
   114	        source = {
   115	            key: _redact_message_media_for_log(item)
   116	            for key, item in value["source"].items()
   117	        }
   118	        if source.get("type") == "base64" and isinstance(source.get("data"), str):
   119	            media_type = source.get("media_type") or "application/octet-stream"
   120	            source["data"] = f"[redacted {media_type} base64 chars={len(source['data'])}]"
   121	        return {
   122	            **{
   123	                key: _redact_message_media_for_log(item)
   124	                for key, item in value.items()
   125	                if key != "source"
   126	            },
   127	            "source": source,
   128	        }
   129	
   130	    return {
   131	        key: _redact_message_media_for_log(item)
   132	        for key, item in value.items()
   133	    }
