## novaic-cortex/novaic_cortex/blob_payload.py
"""Blob raw-byte payload storage for Cortex.

Cortex owns work trace semantics. Blob Service owns large raw bytes. This
module is the narrow adapter between the two: it stores JSON-serialized tool
payloads in Blob Service and returns ``blob://cortex-payload/...`` refs.
"""

from __future__ import annotations

import hashlib
import json
import os
from dataclasses import dataclass
from typing import Any, Protocol
from urllib.parse import quote

from common.http.clients import internal_async_client


DEFAULT_PAYLOAD_BLOB_THRESHOLD_BYTES = 64 * 1024


class PayloadBlobClient(Protocol):
    async def put_payload(
        self,
        *,
        tenant_id: str,
        scope_path: str,
        source_payload_ref: str,
        payload: Any,
        payload_bytes: bytes,
    ) -> str:
        ...

    async def get_payload(self, *, tenant_id: str, blob_ref: str) -> Any:
        ...


@dataclass(frozen=True)
class BlobPayloadPolicy:
    threshold_bytes: int = DEFAULT_PAYLOAD_BLOB_THRESHOLD_BYTES

    @classmethod
    def from_env(cls) -> "BlobPayloadPolicy":
        raw = os.getenv("CORTEX_PAYLOAD_BLOB_THRESHOLD_BYTES", "").strip()
        if not raw:
            return cls()
        try:
            value = int(raw)
        except ValueError:
            value = DEFAULT_PAYLOAD_BLOB_THRESHOLD_BYTES
        return cls(threshold_bytes=max(0, value))

    def should_externalize(self, payload_size: int) -> bool:
        return payload_size > self.threshold_bytes


class HttpBlobPayloadClient:
    def __init__(self, base_url: str, *, timeout: float = 10.0) -> None:
        self._base_url = base_url.rstrip("/")
        self._timeout = timeout

    @classmethod
    def from_env(cls) -> "HttpBlobPayloadClient | None":
        base_url = os.getenv("CORTEX_BLOB_SERVICE_URL", "").strip()
        if not base_url:
            return None
        return cls(base_url)

    async def put_payload(
        self,
        *,
        tenant_id: str,
        scope_path: str,
        source_payload_ref: str,
        payload: Any,
        payload_bytes: bytes,
    ) -> str:
        blob_id = _payload_blob_id(tenant_id, scope_path, source_payload_ref)
        digest = hashlib.sha256(payload_bytes).hexdigest()
        create_body = {
            "tenant_id": tenant_id,
            "namespace": "cortex-payload",
            "mime_type": "application/json",
            "blob_id": blob_id,
            "expected_size": len(payload_bytes),
            "expected_sha256": digest,
            "created_by": "cortex",
            "source_ref": source_payload_ref,
        }
        async with internal_async_client("cortex", timeout=self._timeout) as client:
            create_resp = await client.post(f"{self._base_url}/v1/blobs/uploads", json=create_body)
            create_resp.raise_for_status()
            session_id = create_resp.json().get("session_id")
            if not isinstance(session_id, str) or not session_id:
                raise ValueError("Blob Service did not return an upload session_id")

            parts: list[int] = []
            if payload_bytes:
                part_resp = await client.put(
                    f"{self._base_url}/v1/blobs/uploads/{quote(session_id)}/parts/1",
                    content=payload_bytes,
                    headers={"X-Tenant-ID": tenant_id},
                )
                part_resp.raise_for_status()
                parts = [1]

            complete_resp = await client.post(
                f"{self._base_url}/v1/blobs/uploads/{quote(session_id)}/complete",
                json={"parts": parts, "final_sha256": digest},
                headers={"X-Tenant-ID": tenant_id},
            )
            complete_resp.raise_for_status()
            data = complete_resp.json().get("blob") or {}
        blob_ref = data.get("blob_ref")
        if not isinstance(blob_ref, str) or not blob_ref.startswith("blob://cortex-payload/"):
            raise ValueError("Blob Service did not return a cortex-payload BlobRef")
        return blob_ref

    async def get_payload(self, *, tenant_id: str, blob_ref: str) -> Any:
        namespace, blob_id = _parse_blob_ref(blob_ref)
        async with internal_async_client("cortex", timeout=self._timeout) as client:
            resp = await client.get(
                f"{self._base_url}/v1/blobs/{quote(namespace)}/{quote(blob_id)}",
                headers={"X-Tenant-ID": tenant_id},
            )
            resp.raise_for_status()
            raw = resp.content
        return json.loads(raw.decode("utf-8"))


def encode_payload_bytes(payload: Any) -> bytes:
    return json.dumps(payload, ensure_ascii=False, default=str).encode("utf-8")


def _payload_blob_id(tenant_id: str, scope_path: str, source_payload_ref: str) -> str:
    digest = hashlib.sha256(f"{tenant_id}:{scope_path}:{source_payload_ref}".encode("utf-8")).hexdigest()
    return f"cpx-{digest[:40]}"


def _parse_blob_ref(blob_ref: str) -> tuple[str, str]:
    prefix = "blob://"
    if not isinstance(blob_ref, str) or not blob_ref.startswith(prefix):
        raise ValueError("payload blob_ref must start with blob://")
    rest = blob_ref[len(prefix):]
    parts = rest.split("/")
    if len(parts) != 2:
        raise ValueError("payload blob_ref must be blob://{namespace}/{blob_id}")
    return parts[0], parts[1]

def _request(method: str, path: str, body: dict | None = None) -> str:
    api, token = _load_config()
    if not token:
        print(f"{{COMMAND}}: NOVAIC_TOKEN is not set", file=sys.stderr)
        raise SystemExit(1)
    headers = {{"Authorization": f"Bearer {{token}}"}}
    data = None
    if body is not None:
        headers["Content-Type"] = "application/json"
        data = json.dumps(body).encode("utf-8")
    request = urllib.request.Request(f"{{api}}{{path}}", data=data, headers=headers, method=method)
    try:
        with urllib.request.urlopen(request, timeout=30) as response:
            return response.read().decode("utf-8")
    except urllib.error.HTTPError as exc:
        detail = exc.read().decode("utf-8", errors="replace") if exc.fp else str(exc)
        print(f"{{COMMAND}}: HTTP {{exc.code}}: {{detail}}", file=sys.stderr)
        raise SystemExit(1)
    except OSError as exc:
        print(f"{{COMMAND}}: connection failed: {{exc}}", file=sys.stderr)
        raise SystemExit(1)

def _server_for_device_and_blob(*, screenshot_bytes: bytes = b"", file_bytes: bytes = b""):
    state: dict[str, object] = {
        "uploads": [],
        "parts": [],
        "device_bodies": [],
    }

    class Handler(BaseHTTPRequestHandler):
        def log_message(self, *_args) -> None:
            return

        def _read_json(self) -> dict:
            length = int(self.headers.get("content-length", "0") or "0")
            raw = self.rfile.read(length) if length else b""
            if not raw:
                return {}
            return json.loads(raw.decode("utf-8"))

        def _write_json(self, payload: dict) -> None:
            raw = json.dumps(payload).encode("utf-8")
            self.send_response(200)
            self.send_header("Content-Type", "application/json")
            self.send_header("Content-Length", str(len(raw)))
            self.end_headers()
            self.wfile.write(raw)

        def do_POST(self) -> None:
            body = self._read_json()
            if self.path.endswith("/hd/desktop/screenshot"):
                state["device_bodies"].append(body)
                self._write_json(
                    {
                        "success": True,
                        "format": "jpeg",
                        "width": 640,
                        "height": 360,
                        "screenshot": base64.b64encode(screenshot_bytes).decode("ascii"),
                    }
                )
                return
            if self.path.endswith("/hd/file/pull"):
                state["device_bodies"].append(body)
                self._write_json(
                    {
                        "success": True,
                        "data": base64.b64encode(file_bytes).decode("ascii"),
                        "is_base64": True,
                        "size": len(file_bytes),
                    }
                )
                return
            if self.path == "/v1/blobs/uploads":
                upload = body
                state["uploads"].append(upload)
                self._write_json({"session_id": "upload-" + "a" * 32, "status": "created", "parts": []})
                return
            if self.path.endswith("/complete"):
                parts = state["parts"]
                data = b"".join(part["data"] for part in parts) if parts else b""
                upload = state["uploads"][-1]
                blob_id = upload["blob_id"]
                self._write_json(
                    {
                        "blob": {
                            "blob_ref": f"blob://runtime-artifact/{blob_id}",
                            "namespace": "runtime-artifact",
                            "blob_id": blob_id,
                            "mime_type": upload["mime_type"],
                            "size": len(data),
                            "sha256": hashlib.sha256(data).hexdigest(),
                        }
                    }
                )
                return
            self.send_error(404)

        def do_PUT(self) -> None:
            length = int(self.headers.get("content-length", "0") or "0")
            data = self.rfile.read(length) if length else b""
            state["parts"].append({"path": self.path, "data": data})
            self._write_json({"part_number": 1, "status": "receiving"})

    server = ThreadingHTTPServer(("127.0.0.1", 0), Handler)
    thread = threading.Thread(target=server.serve_forever, daemon=True)
    thread.start()
    return server, thread, state


def test_devicectl_hd_screenshot_uploads_image_and_prints_tool_output_artifact(tmp_path: Path) -> None:
    image = b"\xff\xd8jpeg-bytes"
    server, thread, state = _server_for_device_and_blob(screenshot_bytes=image)
    try:
        base_url = f"http://127.0.0.1:{server.server_port}"
        completed = _run_devicectl(tmp_path, ["hd", "screenshot"], server_url=base_url)
    finally:
        server.shutdown()
        thread.join(timeout=5)

    assert completed.returncode == 0, completed.stderr
    body = json.loads(completed.stdout)
    assert body["version"] == "tool-output.v1"
    assert body["ok"] is True
    assert body["artifacts"][0]["uri"].startswith("blob://runtime-artifact/")
    assert body["artifacts"][0]["modality"] == "image"
    assert body["artifacts"][0]["mime_type"] == "image/jpeg"
    assert body["artifacts"][0]["access"]["display"]["args"]["file_url"] == body["artifacts"][0]["uri"]
    assert state["uploads"][0]["namespace"] == "runtime-artifact"
    assert state["uploads"][0]["mime_type"] == "image/jpeg"
    assert state["parts"][0]["data"] == image
    assert base64.b64encode(image).decode("ascii") not in completed.stdout
    assert '"screenshot":' not in completed.stdout

"""Blob Service object-store adapter below LogicalFS.

Blob Service is a cheap byte/object server. LogicalFS owns realtime /ro and /rw
file authority above this adapter; callers should not use this class as a
semantic workspace API.
"""

from __future__ import annotations

from typing import Any
from urllib.parse import quote

from common.http.clients import internal_async_client

from logicalfs.contracts import LogicalObjectInfo


class BlobObjectStore:
    """Async object-store adapter over Blob Service object APIs."""

    def __init__(
        self,
        *,
        base_url: str,
        tenant_id: str,
        namespace: str = "cortex-store",
        timeout: float = 10.0,
        client_service_name: str = "logicalfs",
    ) -> None:
        if not base_url or not base_url.strip():
            raise ValueError("base_url is required")
        if not tenant_id or not tenant_id.strip():
            raise ValueError("tenant_id is required")
        self._base_url = base_url.rstrip("/")
        self._tenant_id = tenant_id
        self._namespace = namespace
        self._timeout = timeout
        self._client_service_name = client_service_name

    def _headers(self) -> dict[str, str]:
        return {"X-Tenant-ID": self._tenant_id}

    def _object_url(self, key: str) -> str:
        return f"{self._base_url}/v1/objects/{quote(self._namespace)}/{quote(key, safe='/')}"

    async def put(self, key: str, data: bytes, content_type: str = "") -> None:
        headers = self._headers()
        if content_type:
            headers["Content-Type"] = content_type
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.put(self._object_url(key), headers=headers, content=data)
            resp.raise_for_status()

    async def get(self, key: str) -> bytes:
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.get(self._object_url(key), headers=self._headers())
        if resp.status_code == 404:
            raise KeyError(key)
        resp.raise_for_status()
        return resp.content

    async def exists(self, key: str) -> bool:
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.get(
                f"{self._object_url(key)}/info",
                headers=self._headers(),
            )
        if resp.status_code == 404:
            return False
        resp.raise_for_status()
        return True

    async def delete(self, key: str) -> None:
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.delete(self._object_url(key), headers=self._headers())
        if resp.status_code == 404:
            return
        resp.raise_for_status()

    async def list_objects(self, prefix: str, delimiter: str = "/") -> list[LogicalObjectInfo]:
        if delimiter != "/":
            raise NotImplementedError("BlobObjectStore only supports delimiter='/'")
        data = await self._list(prefix=prefix, delimiter=delimiter, recursive=False)
        return [
            LogicalObjectInfo(
                key=str(item["key"]),
                size=int(item.get("size") or 0),
                is_prefix=bool(item.get("is_prefix")),
            )
            for item in data.get("objects", [])
        ]

    async def list_recursive(self, prefix: str) -> list[str]:
        data = await self._list(prefix=prefix, delimiter="/", recursive=True)
        return sorted(str(item["key"]) for item in data.get("objects", []))

    async def _list(self, *, prefix: str, delimiter: str, recursive: bool) -> dict[str, Any]:
        params = {
            "namespace": self._namespace,
            "prefix": prefix,
            "delimiter": delimiter,
            "recursive": "true" if recursive else "false",
        }
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.get(
                f"{self._base_url}/v1/objects/_list",
                params=params,
                headers=self._headers(),
            )
            resp.raise_for_status()
            return resp.json()

    async def move(self, src: str, dst: str) -> None:
        data = await self.get(src)
        await self.put(dst, data)
        await self.delete(src)

    async def move_prefix(self, src_prefix: str, dst_prefix: str) -> int:
        body = {
            "tenant_id": self._tenant_id,
            "namespace": self._namespace,
            "src_prefix": src_prefix,
            "dst_prefix": dst_prefix,
        }
        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
            resp = await client.post(f"{self._base_url}/v1/objects/_move-prefix", json=body)
            resp.raise_for_status()
            data = resp.json()
        return int(data.get("moved") or 0)

## novaic-blob-service/blob_service/object_storage.py
"""Generic object storage facade owned by Blob Service.

This is the infrastructure boundary for services that need object-tree
semantics. Product services provide tenant, namespace, and object keys; Blob
Service owns the backing byte store and optional S3/OSS configuration.
"""

from __future__ import annotations

import re
from dataclasses import dataclass
from typing import Optional

from .contracts import (
    ObjectIdentity,
    ObjectListInput,
    ObjectMovePrefixInput,
    ObjectPutInput,
    ObjectRecursiveListInput,
)
from .storage import BlobByteStore

_SAFE_TENANT_ID = re.compile(r"^[a-zA-Z0-9][a-zA-Z0-9_.:-]{0,255}$")
_SAFE_NAMESPACE = re.compile(r"^[a-zA-Z0-9][a-zA-Z0-9_.:-]{0,127}$")


@dataclass(frozen=True)
class ObjectEntry:
    key: str
    size: int = 0
    is_prefix: bool = False


def _validate_tenant_id(tenant_id: str) -> str:
    if not isinstance(tenant_id, str) or not _SAFE_TENANT_ID.match(tenant_id):
        raise ValueError(f"invalid tenant_id: {tenant_id!r}")
    if "/" in tenant_id or ".." in tenant_id:
        raise ValueError(f"invalid tenant_id: {tenant_id!r}")
    return tenant_id


def _validate_namespace(namespace: str) -> str:
    if not isinstance(namespace, str) or not _SAFE_NAMESPACE.match(namespace):
        raise ValueError(f"invalid namespace: {namespace!r}")
    if "/" in namespace or ".." in namespace:
        raise ValueError(f"invalid namespace: {namespace!r}")
    return namespace


def _normalize_key(key: str, *, allow_empty: bool = False) -> str:
    if not isinstance(key, str):
        raise ValueError("object key must be a string")
    key = key.replace("\\", "/").strip("/")
    if not key and allow_empty:
        return ""
    if not key:
        raise ValueError("object key is required")
    parts = [part for part in key.split("/") if part]
    if any(part == ".." for part in parts):
        raise ValueError(f"invalid object key: {key!r}")
    return "/".join(parts)


def _object_rel_path(namespace: str, key: str) -> str:
    namespace = _validate_namespace(namespace)
    key = _normalize_key(key)
    return f"objects/{namespace}/{key}"


def _object_rel_prefix(namespace: str, prefix: str = "") -> str:
    namespace = _validate_namespace(namespace)
    prefix = _normalize_key(prefix, allow_empty=True)
    base = f"objects/{namespace}"
    return f"{base}/{prefix}/" if prefix else f"{base}/"


class ObjectStorage:
    """Object-tree storage backed by BlobByteStore."""

    def __init__(self, byte_store: BlobByteStore) -> None:
        self._byte_store = byte_store

    def put_object(self, request: ObjectPutInput) -> dict:
        tenant_id = _validate_tenant_id(request.tenant_id)
        rel_path = _object_rel_path(request.namespace, request.key)
        ok = self._byte_store.backend.save_bytes(request.data, tenant_id, rel_path)
        if not ok:
            raise RuntimeError("failed to save object")
        return {
            "tenant_id": tenant_id,
            "namespace": request.namespace,
            "key": _normalize_key(request.key),
            "size": len(request.data),
            "content_type": request.content_type or "application/octet-stream",
        }

    def get_object(self, request: ObjectIdentity) -> Optional[bytes]:
        tenant_id = _validate_tenant_id(request.tenant_id)
        return self._byte_store.backend.load_bytes(
            tenant_id,
            _object_rel_path(request.namespace, request.key),
        )

    def stat_object(self, request: ObjectIdentity) -> Optional[dict]:
        tenant_id = _validate_tenant_id(request.tenant_id)
        info = self._byte_store.backend.get_info(
            tenant_id,
            _object_rel_path(request.namespace, request.key),
        )
        if info is None:
            return None
        return {
            "tenant_id": tenant_id,
            "namespace": request.namespace,
            "key": _normalize_key(request.key),
            "size": int(info.get("size") or 0),
            "created": info.get("created"),
        }

    def delete_object(self, request: ObjectIdentity) -> bool:
        tenant_id = _validate_tenant_id(request.tenant_id)
        return bool(
            self._byte_store.backend.delete(
                tenant_id,
                _object_rel_path(request.namespace, request.key),
            )
        )

    def list_objects(self, request: ObjectListInput) -> list[ObjectEntry]:
        if request.delimiter != "/":
            raise ValueError("only delimiter='/' is supported")

        prefix_norm = _normalize_key(request.prefix, allow_empty=True)
        tenant_id = _validate_tenant_id(request.tenant_id)
        single = (
            self.stat_object(ObjectIdentity(tenant_id=tenant_id, namespace=request.namespace, key=prefix_norm))
            if prefix_norm
            else None
        )
        if single is not None:
            return [ObjectEntry(key=prefix_norm, size=single["size"])]

        rel_prefix = _object_rel_prefix(request.namespace, prefix_norm)
        base = _object_rel_prefix(request.namespace, "")
        seen_prefixes: set[str] = set()
        out: list[ObjectEntry] = []
        for item in self._byte_store.backend.list(tenant_id, rel_prefix):
            rel_path = str(item.get("path") or "")
            if not rel_path.startswith(base):
                continue
            key = rel_path[len(base) :]
            if prefix_norm:
                key = key[len(prefix_norm.strip("/") + "/") :] if key.startswith(prefix_norm.strip("/") + "/") else key
            if "/" in key:
                child = f"{prefix_norm.rstrip('/') + '/' if prefix_norm else ''}{key.split('/', 1)[0]}/"
                if child not in seen_prefixes:
                    seen_prefixes.add(child)
                    out.append(ObjectEntry(key=child, is_prefix=True))
            else:
                out.append(
                    ObjectEntry(
                        key=f"{prefix_norm.rstrip('/') + '/' if prefix_norm else ''}{key}",
                        size=int(item.get("size") or 0),
                    )
                )
        return sorted(out, key=lambda entry: entry.key)

    def list_recursive(self, request: ObjectRecursiveListInput) -> list[str]:
        prefix_norm = _normalize_key(request.prefix, allow_empty=True)
        tenant_id = _validate_tenant_id(request.tenant_id)
        rel_prefix = _object_rel_prefix(request.namespace, prefix_norm)
        base = _object_rel_prefix(request.namespace, "")
        keys: list[str] = []
        for item in self._byte_store.backend.list(tenant_id, rel_prefix):
            rel_path = str(item.get("path") or "")
            if rel_path.startswith(base):
                keys.append(rel_path[len(base) :])
        return sorted(keys)

    def move_object(self, *, tenant_id: str, namespace: str, src: str, dst: str) -> None:
        data = self.get_object(ObjectIdentity(tenant_id=tenant_id, namespace=namespace, key=src))
        if data is None:
            raise KeyError(src)
        self.put_object(ObjectPutInput(
            tenant_id=tenant_id,
            namespace=namespace,
            key=dst,
            data=data,
            content_type="application/octet-stream",
        ))
        self.delete_object(ObjectIdentity(tenant_id=tenant_id, namespace=namespace, key=src))

    def move_prefix(self, request: ObjectMovePrefixInput) -> int:
        tenant_id = _validate_tenant_id(request.tenant_id)
        src_prefix_norm = _normalize_key(request.src_prefix, allow_empty=True)
        dst_prefix_norm = _normalize_key(request.dst_prefix, allow_empty=True)
        moved = 0
        for key in self.list_recursive(ObjectRecursiveListInput(
            tenant_id=tenant_id,
            namespace=request.namespace,
            prefix=src_prefix_norm,
        )):
            suffix = key[len(src_prefix_norm) :].lstrip("/") if src_prefix_norm else key
            dst_key = f"{dst_prefix_norm.rstrip('/')}/{suffix}" if dst_prefix_norm else suffix
            self.move_object(tenant_id=tenant_id, namespace=request.namespace, src=key, dst=dst_key)
            moved += 1
        return moved

## docs/architecture/logicalfs-realtime-file-authority.md
# LogicalFS RO/RW Realtime File Authority

## Decision

LogicalFS is the semantic authority for Cortex/shell realtime `RO` / `RW`
views. Blob Service is the cheap file server for durable bytes and does not
need realtime semantics.

Blob Service owns durable bytes, object storage primitives, multipart upload,
and blob references. Display and artifacts still use Blob for byte storage and
delivery. The rule is narrower and sharper: every Cortex-related shell/workspace
file operation that depends on live `RO` / `RW` semantics must go through
LogicalFS. Other ordinary files can stay Blob-backed without pretending to be a
live filesystem.

## Current Gap

The current code is close to the target shape, with remaining cleanup focused on
service extraction and hardening:

- Blob Service exposes base byte/blob APIs and object-tree primitives used by
  infrastructure.
- LogicalFS owns the live file authority and may use a Blob object adapter below
  that boundary.
- Cortex owns Workspace semantics and uses a LogicalFS authority for file-like
  Workspace APIs.
- Cortex uses the `novaic-logicalfs` substrate as a library to materialize a
  snapshot into a sandbox mount, then writes RW patches back to Workspace.
- Sandboxd only executes a process over the materialized view.

This is improved, but LogicalFS is not yet a standalone service boundary. The
remaining risk is future code reintroducing direct live workspace paths around
Blob object APIs, Cortex-local file authority wrappers, or sandbox scratch state.

## Target Model

```text
Cortex / Runtime shell pipeline
  scope/workspace semantics and shell execution
        |
        | live RO/RW file intents
        v
LogicalFS
  realtime RO/RW namespace authority
  RO/RW layout, subagent RW layout, snapshots, patches, stable shell paths
        |
        | store/load RO/RW bytes
        v
Blob Service
  cheap file server: blob refs, object primitives, multipart, physical backend
        |
        v
Disk / OSS / S3

App, CLI, display, and artifact byte delivery can use Blob directly when they
do not need live Cortex/shell RO/RW semantics.
```

Sandboxd is not a file authority. It receives an executable filesystem view
from LogicalFS, runs a process, and reports process output. It may host the
local mount implementation, but it should not own workspace semantics,
subagent semantics, display semantics, or blob persistence rules.

## Responsibilities

### Blob Service

Owns:

- raw bytes and blob references
- object storage primitives used by infrastructure
- multipart upload lifecycle
- physical backend choice such as disk, OSS, or S3
- byte-level metadata and limits
- display bytes and artifact bytes, when represented as Blob objects

Does not own:

- `/ro` and `/rw` semantics
- agent, wake, scope, or subagent layout
- shell cwd/env rules
- realtime RO/RW diffs or file-view authorization
- display/artifact product meaning

### LogicalFS Service

Owns:

- realtime `RO` / `RW` namespace APIs
- stable path layout such as `/cortex/ro`, `/cortex/rw`, and subagent RW dirs
- RO/RW projection rules
- file read/write/list/delete/move semantics above Blob
- snapshot and patch semantics for shell execution
- live update propagation for RO/RW state when needed
- mapping between semantic owners and durable Blob object keys

Does not own:

- LLM prompt/context decisions
- Cortex scope lifecycle state
- Queue/session FSM state
- raw Blob physical backend configuration
- process execution

### Cortex

Owns:

- agent-root, wake, skill, scope, step, payload, and context semantics
- which files belong in RO vs RW for a given agent/session/subagent
- context assembly and tool contract semantics

Uses LogicalFS for:

- Workspace file persistence and listing
- exposing `/ro` and `/rw` to shell
- reading payload-backed files when they are filesystem entries

### Sandboxd

Owns:

- process execution
- timeout and process result capture
- bind/mount namespace mechanics when needed

Uses LogicalFS for:

- executable filesystem views
- stable path mount inputs
- post-run patch collection, if the local implementation stays colocated

## Hard Boundary Rules

1. Cortex-related shell/workspace files must not bypass LogicalFS for live
   `RO` / `RW` semantics.
2. Direct Blob access is allowed for cheap file serving, base byte/blob
   functions, attachments, display bytes, artifact bytes, and LogicalFS
   internals implementing persistence.
3. Cortex may own semantic names, but LogicalFS must own file operations.
4. Sandboxd may execute over a view, but LogicalFS must own the view contract.
5. Display and artifacts use Blob for bytes. If the source starts as an RO/RW
   file, that file must be copied/exported through Blob before display or
   download. LogicalFS does not provide display/download handles.
6. Tests and CI should reject new direct Blob-object live RO/RW file paths
   outside LogicalFS internals.

## Migration Shape

### Phase 1: Declare the Boundary

- Keep Blob Service as the cheap bytes/object file server.
- Document LogicalFS as the Cortex/shell RO/RW authority.
- Add guardrails that distinguish base Blob use from live RO/RW semantics.

### Phase 2: Promote LogicalFS From Library To Service Boundary

- Define LogicalFS HTTP or RPC contracts focused on RO/RW:
  - `read`, `write`, `list`, `delete`, `move`
  - `snapshot`
  - `apply_patch`
- Keep the existing local substrate as an implementation detail behind the
  service boundary.

### Phase 3: Move Cortex Workspace Persistence Behind LogicalFS

- Replace Cortex-local persistence adapters with a LogicalFS-backed workspace
  adapter.
- Cortex passes semantic owner/scope/session/subagent metadata explicitly.
- LogicalFS maps those semantics to durable Blob object keys.

### Phase 4: Move Shell/Sandbox Mounts Behind LogicalFS

- Sandbox execution requests reference a LogicalFS view id or snapshot id.
- Sandboxd no longer receives Cortex Workspace details.
- Patch application becomes LogicalFS responsibility, not Cortex glue code.

### Phase 5: Display And Artifact Bytes

- `display` keeps direct Blob support for base `blob://...` files.
- Artifacts are Blob-backed when they need display/download.
- RO/RW files that need display/download are first exported or copied to Blob.
- LogicalFS remains a live RO/RW layer; it does not become a display/download
  handle service.

### Phase 6: Delete Old Paths

- Remove direct live RO/RW file use of Blob object APIs outside LogicalFS.
- Remove Cortex-local live RO/RW file service wrappers.
- Remove sandbox-local file semantics.
- Add residue scans for accidental direct Blob object calls in live RO/RW paths.

## Acceptance Criteria

- There is exactly one Cortex/shell live `RO` / `RW` authority: LogicalFS.
- Blob remains a cheap file server and byte/object infrastructure.
- Cortex no longer directly implements live RO/RW persistence over Blob.
- Sandboxd no longer owns workspace or subagent file semantics.
- Display path is split cleanly:
  - base attachments: direct Blob
  - ordinary artifact/display bytes: direct Blob
  - RO/RW file semantics: LogicalFS only
  - RO/RW file display/download: export/copy to Blob, then display/download
- New tests prove Cortex/shell live RO/RW paths cannot bypass LogicalFS.
