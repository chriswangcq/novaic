def _wrap_hd_file_pull(result: dict, request_body: dict) -> dict:
    if result.get("success") is False or not result.get("data") or result.get("is_base64") is not True:
        return result
    path = str(request_body.get("path") or result.get("path") or "")
    data = _decode_b64_field(str(result.get("data") or ""), field="data")
    name = Path(path).name or "pulled-file"
    mime_type = mimetypes.guess_type(name)[0] or "application/octet-stream"
    blob = _put_blob(
        data,
        mime_type=mime_type,
        filename=name,
        source_ref=f"devicectl hd file-pull {{path}}".strip(),
    )
    blob_ref = str(blob["blob_ref"])
    size = int(blob.get("size") or len(data))
    sha = str(blob.get("sha256") or hashlib.sha256(data).hexdigest())
    artifact = _artifact_manifest(
        uri=blob_ref,
        modality="binary",
        name=name,
        mime_type=mime_type,
        size_bytes=size,
        sha256=sha,
        summary=f"Host desktop file pulled from {{path}}" if path else "Host desktop file pulled",
        source={{"tool": "devicectl hd file-pull", "path": path}},
        projection={{"history": "manifest_only"}},
    )
    return _tool_output(
        ok=True,
        text=f"Pulled host desktop file {{path or name}} into runtime artifact {{blob_ref}}.",
        artifacts=[artifact],
        events=[{{"type": "device.file.pulled", "summary": "File stored as runtime artifact."}}],
        diagnostics={{
            "device_result": {{k: v for k, v in result.items() if k != "data"}},
        }},
    )

def _flag_values(args: list[str], flag: str) -> list[str]:
    out: list[str] = []
    i = 0
    while i < len(args):
        if args[i] == flag and i + 1 < len(args):
            out.append(args[i + 1])
            i += 2
            continue
        i += 1
    return out


def _flag_value(args: list[str], flag: str, default: str = "") -> str:
    values = _flag_values(args, flag)
    return values[-1] if values else default


def _flag_int(args: list[str], flag: str, default: int, low: int, high: int) -> int:
    try:
        value = int(_flag_value(args, flag, str(default)))
    except ValueError:
        value = default
    return max(low, min(value, high))

def _format_mcp_content(
    result_data: Dict[str, Any],
    *,
    include_display: bool,
    max_text_len: Optional[int] = None,
    empty_text: str = "OK",
) -> str:
    converted = []

    text = result_data.get("text", "")
    if max_text_len is not None and len(text) > max_text_len:
        text = text[:max_text_len] + f"\n\n... [Truncated due to context budget length > {max_text_len}]"

    if text:
        converted.append({"type": "text", "text": text})

    if include_display:
        display_files = result_data.get("display_files", [])
        for f in display_files:
            url = f.get("url", "")
            if not url:
                continue
            if url.startswith("data:"):
                # Data URL
                parts = url.split(",", 1)
                mime = "image/png"
                if len(parts) >= 1 and ";" in parts[0]:
                    mime = parts[0].split(";")[0].replace("data:", "")
                data_b64 = parts[1] if len(parts) > 1 else ""

                converted.append({
                    "type": "image",
                    "data": data_b64,
                    "mimeType": mime,
                })
            else:
                # Normal URL: represent unresolved display media as text.
                converted.append({"type": "text", "text": f"[Image: {url}]"})
                
    if not converted:
        converted.append({"type": "text", "text": empty_text})

    return json.dumps({"_mcp_content": converted}, ensure_ascii=False)

def _display_public_output(result: dict[str, Any]) -> dict[str, Any]:
    """Return display output safe for public tool history."""

    def sanitize_item(item: Any) -> Any:
        if not isinstance(item, dict) or item.get("type") != "image":
            return item
        return {
            "type": "image",
            "_placeholder": True,
            "mimeType": item.get("mimeType") or item.get("mime_type") or "image/png",
            "_note": "Image data provided separately to LLM",
        }

    public: dict[str, Any] = {}
    for key, value in result.items():
        if key == _DURABLE_PAYLOAD_KEY:
            continue
        if key == "_mcp_content" and isinstance(value, list):
            public[key] = [sanitize_item(item) for item in value]
        else:
            public[key] = value
    return public
