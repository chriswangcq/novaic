"""Pure ContextEvent projection to LLM-facing snapshot state."""

from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any, Iterable

from novaic_cortex.context_events import ContextEvent, stable_json
from novaic_cortex.types import estimate_tokens


class ContextEventProjectionError(ValueError):
    """Raised when ordered events cannot be projected consistently."""


@dataclass
class ContextProjectionSnapshot:
    stream_id: str | None = None
    root_scope_id: str | None = None
    first_seq: int = 0
    applied_seq: int = 0
    messages: list[dict[str, Any]] = field(default_factory=list)
    stack: list[dict[str, Any]] = field(default_factory=list)
    estimated_tokens: int = 0

    def to_dict(self) -> dict[str, Any]:
        return {
            "stream_id": self.stream_id,
            "root_scope_id": self.root_scope_id,
            "first_seq": self.first_seq,
            "applied_seq": self.applied_seq,
            "messages": self.messages,
            "stack": self.stack,
            "estimated_tokens": self.estimated_tokens,
        }


@dataclass
class _ScopeProjection:
    frame: dict[str, Any]
    parent_scope_id: str
    messages: list[dict[str, Any]] = field(default_factory=list)


@dataclass
class _ProjectionState:
    snapshot: ContextProjectionSnapshot = field(default_factory=ContextProjectionSnapshot)
    scopes: dict[str, _ScopeProjection] = field(default_factory=dict)
    observed_tool_call_ids: set[str] = field(default_factory=set)
    active_wake_scope_id: str | None = None
    active_wake_message_start_idx: int = 0
    discarded_messages: list[dict[str, Any]] = field(default_factory=list)


def project_context_events(events: Iterable[ContextEvent]) -> ContextProjectionSnapshot:
    state = _ProjectionState()
    for event in events:
        _validate_replay_order(state, event)
        _apply_event(state, event)
        state.snapshot.applied_seq = event.seq
    snapshot = state.snapshot
    snapshot.estimated_tokens = sum(
        estimate_tokens(str(message.get("content", ""))) for message in snapshot.messages
    )
    return snapshot


def _validate_replay_order(state: _ProjectionState, event: ContextEvent) -> None:
    snapshot = state.snapshot
    if snapshot.stream_id is None:
        if event.seq != 1:
            raise ContextEventProjectionError(
                f"context event sequence must start at 1, got {event.seq}"
            )
        snapshot.stream_id = event.stream_id
        snapshot.root_scope_id = event.root_scope_id
        snapshot.first_seq = event.seq
        return

    if event.stream_id != snapshot.stream_id:
        raise ContextEventProjectionError("cannot project events from multiple streams")
    if event.root_scope_id != snapshot.root_scope_id:
        raise ContextEventProjectionError("cannot project events from multiple root scopes")

    expected_seq = snapshot.applied_seq + 1
    if event.seq != expected_seq:
        raise ContextEventProjectionError(
            f"context event sequence must be contiguous: expected seq {expected_seq}, got {event.seq}"
        )


def _apply_event(state: _ProjectionState, event: ContextEvent) -> None:
    snapshot = state.snapshot
    if snapshot.root_scope_id is None:
        snapshot.root_scope_id = event.root_scope_id
    elif snapshot.root_scope_id != event.root_scope_id:
        raise ContextEventProjectionError("cannot project events from multiple root scopes")

    if event.type == "RootInitialized":
        snapshot.root_scope_id = event.payload.get("root_scope_id", event.root_scope_id)
    elif event.type == "WakeStarted":
        wake_scope_id = _required_payload_str(event, "wake_scope_id")
        _suppress_active_wake(state)
        state.active_wake_scope_id = wake_scope_id
        state.active_wake_message_start_idx = len(snapshot.messages)
        snapshot.stack.append(
            {
                "scope_id": wake_scope_id,
                "name": event.payload.get("scope_name", "wake"),
                "kind": "wake",
            }
        )
    elif event.type == "WakeArchived":
        remaining = event.payload.get("remaining_stack")
        if isinstance(remaining, list):
            snapshot.stack = [
                {"scope_id": str(scope_id), "name": str(scope_id), "kind": "scope"}
                for scope_id in remaining
            ]
        else:
            wake_scope_id = event.payload.get("wake_scope_id")
            if wake_scope_id is not None:
                snapshot.stack = [
                    frame for frame in snapshot.stack if frame.get("scope_id") != wake_scope_id
                ]
                if state.active_wake_scope_id == wake_scope_id:
                    state.active_wake_scope_id = None
    elif event.type == "SystemPromptAdded":
        _append_projected_message(snapshot.messages, _required_message(event))
    elif event.type == "ContextMessageAppended":
        _append_projected_message(
            _message_container(state, event.payload.get("scope_id")),
            _required_message(event),
        )
    elif event.type == "InputNotificationAttached":
        _message_container(state, event.payload.get("scope_id")).append(_notification_hint(event))
    elif event.type == "SkillScopeOpened":
        scope_id = _required_payload_str(event, "scope_id")
        parent_scope_id = _required_payload_str(event, "parent_scope_id")
        _suppress_open_siblings(state, parent_scope_id)
        frame = {
            "scope_id": scope_id,
            "parent_scope_id": parent_scope_id,
            "skill_name": _required_payload_str(event, "skill_name"),
            "name": event.payload.get("name", event.payload.get("skill_name", "skill")),
            "kind": "skill",
        }
        snapshot.stack.append(frame)
        state.scopes[scope_id] = _ScopeProjection(
            frame=frame,
            parent_scope_id=parent_scope_id,
        )
    elif event.type == "SkillScopeClosed":
        scope_id = _required_payload_str(event, "scope_id")
        if not snapshot.stack or snapshot.stack[-1].get("scope_id") != scope_id:
            raise ContextEventProjectionError(
                f"SkillScopeClosed violates LIFO: attempted to close {scope_id!r}"
            )
        frame = snapshot.stack.pop()
        scope = state.scopes.get(scope_id)
        if scope is None and frame.get("kind") == "wake":
            _close_active_wake(state, event, frame)
            return
        if scope is None:
            raise ContextEventProjectionError(f"SkillScopeClosed missing opened scope {scope_id!r}")
        report = str(event.payload.get("report", ""))
        parent_messages = _message_container(state, scope.parent_scope_id)
        if report.strip():
            parent_messages.append(
                {
                    "role": "system",
                    "content": f"[Skill '{scope.frame['skill_name']}' completed]\n{report}",
                    "_message_type": "SKILL_SUMMARY",
                    "_metadata": {
                        "scope_id": scope_id,
                        "skill_name": scope.frame["skill_name"],
                    },
                }
            )
        else:
            parent_messages.extend(scope.messages)
    elif event.type == "AssistantToolCallRecorded":
        message = _required_message(event)
        for call_id in _tool_call_ids(message):
            state.observed_tool_call_ids.add(call_id)
        _message_container(state, event.payload.get("scope_id")).append(message)
    elif event.type == "ToolStepRecorded":
        call_id = _required_payload_str(event, "call_id")
        _message_container(state, event.payload.get("scope_id")).append(
            _tool_result_message(
                event,
                call_id=call_id,
                orphan=call_id not in state.observed_tool_call_ids,
            )
        )


def _required_message(event: ContextEvent) -> dict[str, Any]:
    message = event.payload.get("message")
    if not isinstance(message, dict):
        raise ContextEventProjectionError(f"{event.type} requires payload.message object")
    return dict(message)


def _message_container(
    state: _ProjectionState,
    scope_id: Any,
) -> list[dict[str, Any]]:
    if isinstance(scope_id, str):
        if scope_id == state.active_wake_scope_id:
            return state.snapshot.messages
        if scope_id in state.scopes:
            if _is_scope_on_stack(state, scope_id):
                return state.scopes[scope_id].messages
            return state.discarded_messages
        if state.active_wake_scope_id is not None:
            return state.discarded_messages
    return state.snapshot.messages


def _append_projected_message(
    container: list[dict[str, Any]],
    message: dict[str, Any],
) -> None:
    if not _is_system_prompt(message):
        container.append(message)
        return

    insert_at = 0
    while insert_at < len(container) and _is_system_prompt(container[insert_at]):
        insert_at += 1
    container.insert(insert_at, message)


def _is_system_prompt(message: dict[str, Any]) -> bool:
    return (
        message.get("role") == "system"
        and message.get("_message_type") in {None, "SYSTEM_MESSAGE"}
    )


def _close_active_wake(
    state: _ProjectionState,
    event: ContextEvent,
    frame: dict[str, Any],
) -> None:
    scope_id = _required_payload_str(event, "scope_id")
    report = str(event.payload.get("report", ""))
    if state.active_wake_scope_id == scope_id:
        del state.snapshot.messages[state.active_wake_message_start_idx :]
        if report.strip():
            state.snapshot.messages.append(
                {
                    "role": "system",
                    "content": f"[Skill '{frame.get('name', 'wake')}' completed]\n{report}",
                    "_message_type": "SKILL_SUMMARY",
                    "_metadata": {
                        "scope_id": scope_id,
                        "skill_name": frame.get("name", "wake"),
                    },
                }
            )
        state.active_wake_scope_id = None
        state.active_wake_message_start_idx = len(state.snapshot.messages)
        return

    if report.strip():
        state.snapshot.messages.append(
            {
                "role": "system",
                "content": f"[Skill '{frame.get('name', 'wake')}' completed]\n{report}",
                "_message_type": "SKILL_SUMMARY",
                "_metadata": {
                    "scope_id": scope_id,
                    "skill_name": frame.get("name", "wake"),
                },
            }
        )
