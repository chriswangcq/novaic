    if payload.get("write_as_step") and payload.get("step"):
        step = payload["step"]
        result = bridge.write_step(cortex_scope_id, step)
        if result is None:
            raise RuntimeError(
                f"write_step failed (cortex unavailable or rejected) for scope={scope_id}"
            )
        project_tool_step(payload, activity_ctx, step)
        return {
            "success": True,
            "scope_id": scope_id,
            "wrote_step": True,
            "step_path": result.path,
        }

    if not payload.get("message"):
        raise ValidationError("Missing required field: message")

    message = dict(payload["message"])
    if payload.get("idempotency_key"):
        message["_idempotency_key"] = payload["idempotency_key"]
    if payload.get("message_type"):
        message["_message_type"] = payload["message_type"]

    bridge.append_context(cortex_scope_id, message, scope_path=scope_path)
    if message.get("role") == "assistant":
        project_assistant_message(payload, activity_ctx, message)

    return {
        "success": True,
        "scope_id": scope_id,
        "appended": True,
        "context_length": -1,
    }
