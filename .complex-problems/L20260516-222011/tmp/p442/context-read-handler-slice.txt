    """
    scope_id = payload.get("scope_id")
    cortex_scope_id = payload.get("agent_root_scope_id") or scope_id
    scope_path = payload.get("wake_scope_path") or payload.get("scope_path")
    agent_id = payload.get("agent_id")
    subagent_id = payload.get("subagent_id", "")
    user_id = payload.get("user_id", "")

    if not scope_id:
        raise ValidationError("Missing required field: scope_id")
    if not agent_id:
        raise ValidationError("Missing required field: agent_id")

    bridge = get_cortex_bridge(cortex_scope_id, user_id, agent_id)
    context = bridge.read_context(cortex_scope_id, scope_path=scope_path)

    # Idempotency set: collect _idempotency_key of every message already in
    # Cortex context. Used to skip double-append if a previous context.read
    # crashed after bridge.append_context.
    seen_keys: set[str] = set()
    for m in context:
        key = m.get("_idempotency_key") if isinstance(m, dict) else None
        if key:
            seen_keys.add(key)

    new_messages_list = []

    # PR-46/PR-113 — single ``read_session_meta`` call for by-id input
    # assembly. Runtime no longer replays already-consumed chat history.
    # PR-165A further narrows this path: ids become Environment
    # notifications, not raw message-body context.
    try:
        session_meta = bridge.read_session_meta(cortex_scope_id, scope_path=scope_path)
    except Exception as e:
        logger.warning("[context.read] read_session_meta soft-fail: %s", e)
        session_meta = None
    if not isinstance(session_meta, dict):
        session_meta = {}

    # PR-165A/PR-165C — prompt receives Environment notifications only. We keep
    # the PR-46 by-id SSOT, but we do not fetch Business message rows here and
    # do not consult UI read/unread receipt state.
    raw_input_ids = session_meta.get("input_message_ids") or []
    if not isinstance(raw_input_ids, list):
        raw_input_ids = []
    unseen_ids = []
    for mid in raw_input_ids:
        if not isinstance(mid, str) or not mid:
            continue
        if _notification_key(mid) in seen_keys:
            continue
        unseen_ids.append(mid)

    logger.info(
        "[context.read] event=context_read kind=environment_notifications scope=%s agent=%s "
        "input_ids=%d unseen=%d",
        scope_id, agent_id, len(raw_input_ids),
        len(unseen_ids),
    )
    metric_inc("context_read_assembly_total", kind="environment_notifications")

    for notification_id in unseen_ids:
        ctx_msg = _notification_context_message(notification_id)
        try:
            bridge.append_context(cortex_scope_id, ctx_msg, scope_path=scope_path)
            seen_keys.add(ctx_msg["_idempotency_key"])
            context.append(ctx_msg)
            new_messages_list.append({
                "notification_id": notification_id,
                "source_kind": "im_message",
            })
        except Exception as e:
            logger.warning(
                "[context.read] failed to append notification id=%s err=%s",
                notification_id, e,
            )
