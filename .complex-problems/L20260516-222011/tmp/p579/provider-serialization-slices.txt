   183	def process_multimodal_messages(
   184	    messages: List[Dict[str, Any]], 
   185	    provider: str = "openai"
   186	) -> List[Dict[str, Any]]:
   187	    """
   188	    处理消息中的多模态内容
   189	    
   190	    将 tool_result 中的图片提取出来，作为单独的 user message 添加
   191	    （因为 LLM API 不支持在 tool result 中放图片）
   192	    
   193	    关键：保证 assistant+tool_calls 后的所有 tool results 连续，
   194	    图片 user messages 添加到所有 tool results 之后。
   195	    
   196	    Args:
   197	        messages: 原始消息列表
   198	        provider: LLM 提供商 ("openai" 或 "anthropic")
   199	        
   200	    Returns:
   201	        处理后的消息列表
   202	        
   203	    Only current-turn ``display_perception`` tool messages may be converted to
   204	    provider-native image content. Shell and historical tool messages stay as
   205	    terminal-style text receipts.
   206	    """
   207	    if not HAS_MULTIMODAL:
   208	        return messages
   209	    
   210	    processed = []
   211	    pending_images = []  # 收集待添加的图片 user messages
   212	    
   213	    for i, msg in enumerate(messages):
   214	        role = msg.get("role")
   215	        content = msg.get("content", "")
   216	        
   217	        if role in ("tool_result", "tool"):
   218	            projection = msg.get("_projection")
   219	            tool_msg = {k: v for k, v in msg.items() if k != "_projection"}
   220	            if projection != "display_perception":
   221	                processed.append(tool_msg)
   222	                continue
   223	
   224	            try:
   225	                result = json.loads(content) if isinstance(content, str) else content
   226	            except (json.JSONDecodeError, TypeError):
   227	                result = {"content": content}
   228	
   229	            if isinstance(result, dict) and multimodal.has_images(result):
   230	                _, images = multimodal.extract_from_result(result)
   231	                text_only = multimodal.result_to_text_only(result)
   232	
   233	                # 添加文本版本的 tool result
   234	                processed.append({
   235	                    **tool_msg,
   236	                    "content": text_only,
   237	                })
   238	                
   239	                # 收集图片，但不立即添加
   240	                if images:
   241	                    tool_name = msg.get("name", "tool")
   242	                    description = f"[Image from {tool_name}]"
   243	                    
   244	                    if provider == "anthropic":
   245	                        img_content = multimodal.to_anthropic_content(images, description)
   246	                    else:
   247	                        img_content = multimodal.to_openai_content(images, description)
   248	                    
   249	                    pending_images.append({
   250	                        "role": "user",
   251	                        "content": img_content,
   252	                    })
   253	            else:
   254	                processed.append(tool_msg)
   255	        elif role == "user":
   256	            # User message: 若有 {text, attachments} 格式，转为 LLM content 数组
   257	            # 注意：Repository 层已统一将 content 解析为 dict
   258	            if pending_images:
   259	                processed.extend(pending_images)
   260	                pending_images = []
   261	            
   262	            if isinstance(content, dict) and "attachments" in content and content["attachments"]:
   263	                # 有附件，转为 LLM 格式（文本 + 附件描述）
   264	                from .user_content import user_content_to_llm
   265	                llm_content = user_content_to_llm(content, provider=provider)
   266	                processed.append({"role": "user", "content": llm_content})
   267	            elif isinstance(content, dict) and "text" in content:
   268	                # 无附件，直接使用 text
   269	                processed.append({"role": "user", "content": content.get("text", "")})
   270	            else:
   271	                # 兼容旧格式（纯字符串）
   272	                processed.append(msg)
   273	        else:
   274	            # 其他消息
   275	            if pending_images:
   276	                processed.extend(pending_images)
   277	                pending_images = []
   278	            processed.append(msg)
   279	    
   280	    # 处理末尾剩余的图片
   281	    if pending_images:
   282	        processed.extend(pending_images)
   283	    
   284	    return processed

    14	def _bounded_text(value: Any, *, limit: int = _TEXT_ONLY_CONTENT_LIMIT_CHARS) -> str:
    15	    text = "" if value is None else str(value)
    16	    if len(text) <= limit:
    17	        return text
    18	    return text[:limit] + f"\n\n... [truncated display text to {limit} chars]"

    21	def extract_from_result(result: Dict[str, Any]) -> Tuple[str, List[Dict[str, str]]]:
    22	    """Extract text/images from the current display ``_mcp_content`` contract."""
    23	    text_parts = []
    24	    images = []
    25	    mcp_content = result.get("_mcp_content")
    26	    if not isinstance(mcp_content, list):
    27	        return "", []
    28	
    29	    for item in mcp_content:
    30	        if not isinstance(item, dict):
    31	            continue
    32	        item_type = item.get("type", "")
    33	        if item_type == "text":
    34	            text = item.get("text", "")
    35	            if text:
    36	                text_parts.append(text)
    37	        elif item_type == "image":
    38	            data = item.get("data", "")
    39	            if data:
    40	                images.append({
    41	                    "data": data,
    42	                    "mime_type": item.get("mimeType", "image/png"),
    43	                })
    44	
    45	    return "\n".join(text_parts), images

    48	def has_images(result: Dict[str, Any]) -> bool:
    49	    """Return whether a display ``_mcp_content`` result contains image data."""
    50	    mcp_content = result.get("_mcp_content", [])
    51	    for item in mcp_content:
    52	        if isinstance(item, dict) and item.get("type") == "image" and item.get("data"):
    53	            return True
    54	
    55	    return False

    58	def to_openai_content(images: List[Dict[str, str]], text: str = "") -> List[Dict[str, Any]]:
    59	    """
    60	    转换为 OpenAI 多模态 content 格式
    61	    
    62	    Args:
    63	        images: 图片列表
    64	        text: 附加的文本说明
    65	        
    66	    Returns:
    67	        OpenAI content 数组 [{"type": "image_url", ...}, {"type": "text", ...}]
    68	    """
    69	    content = []
    70	    
    71	    for img in images:
    72	        data_url = f"data:{img['mime_type']};base64,{img['data']}"
    73	        content.append({
    74	            "type": "image_url",
    75	            "image_url": {"url": data_url}
    76	        })
    77	    
    78	    if text:
    79	        content.append({"type": "text", "text": text})
    80	    
    81	    return content

    84	def to_anthropic_content(images: List[Dict[str, str]], text: str = "") -> List[Dict[str, Any]]:
    85	    """
    86	    转换为 Anthropic 多模态 content 格式
    87	    
    88	    Args:
    89	        images: 图片列表
    90	        text: 附加的文本说明
    91	        
    92	    Returns:
    93	        Anthropic content 数组 [{"type": "image", ...}, {"type": "text", ...}]
    94	    """
    95	    content = []
    96	    
    97	    for img in images:
    98	        content.append({
    99	            "type": "image",
   100	            "source": {
   101	                "type": "base64",
   102	                "media_type": img["mime_type"],
   103	                "data": img["data"]
   104	            }
   105	        })
   106	    
   107	    if text:
   108	        content.append({"type": "text", "text": text})
   109	    
   110	    return content

   113	def result_to_text_only(result: Dict[str, Any]) -> str:
   114	    """Return JSON with display image payloads replaced by placeholders."""
   115	    output = {}
   116	    
   117	    def sanitize_content_array(content: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
   118	        sanitized = []
   119	        for item in content:
   120	            if item.get("type") == "image":
   121	                sanitized.append({
   122	                    "type": "image",
   123	                    "_placeholder": True,
   124	                    "mimeType": item.get("mimeType", item.get("mime_type", "image/png")),
   125	                    "_note": "Image data provided separately to LLM"
   126	                })
   127	            elif item.get("type") == "text":
   128	                sanitized.append({
   129	                    **item,
   130	                    "text": _bounded_text(item.get("text", "")),
   131	                })
   132	            else:
   133	                sanitized.append(item)
   134	        return sanitized
   135	    
   136	    for key, value in result.items():
   137	        if key == "_mcp_content":
   138	            if isinstance(value, list):
   139	                output[key] = sanitize_content_array(value)
   140	            else:
   141	                output[key] = value
   142	        else:
   143	            output[key] = value
   144	    
   145	    return json.dumps(output)

   142	        model=config.model_id,
   143	        tools=deepcopy(source.tools),
   144	        provider=source.provider,
   145	        agent_id=agent_id,
   146	    )

     4	def _text_values(value):
     5	    if isinstance(value, dict):
     6	        for key, item in value.items():
     7	            if key == "text" and isinstance(item, str):
