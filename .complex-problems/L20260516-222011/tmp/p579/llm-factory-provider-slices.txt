    46	    def _classify_error(self, status_code: int, body: str) -> ProviderError:
    47	        """分类错误，判断是否可重试"""
    48	        retryable_codes = {429, 500, 502, 503, 529}
    49	        retryable = status_code in retryable_codes
    50	
    51	        # 特殊情况：context overflow 不应重试
    52	        overflow_keywords = [
    53	            "context_length_exceeded",
    54	            "max_tokens",
    55	            "too many tokens",
    56	            "context window",
    57	            "prompt is too long",
    58	        ]
    59	        body_lower = body.lower()
    60	        if any(kw in body_lower for kw in overflow_keywords):
    61	            retryable = False
    62	
    63	        return ProviderError(
    64	            f"API error {status_code}: {body[:500]}",
    65	            status_code=status_code,
    66	            retryable=retryable,
    67	        )

   121	        }

   123	    def _convert_messages(self, messages: List[Dict[str, Any]]) -> tuple:
   124	        """OpenAI messages → Anthropic (system, messages)"""
   125	        system_prompt = None
   126	        converted = []
   127	        pending_tool_results = []
   128	
   129	        for msg in messages:
   130	            role = msg.get("role")
   131	            content = msg.get("content", "")
   132	
   133	            if role == "system":
   134	                system_prompt = content
   135	            elif role == "user":
   136	                if pending_tool_results:
   137	                    converted.append({"role": "user", "content": pending_tool_results})
   138	                    pending_tool_results = []
   139	                converted.append(
   140	                    {"role": "user", "content": self._convert_content(content)}
   141	                )
   142	            elif role == "assistant":
   143	                if pending_tool_results:
   144	                    converted.append({"role": "user", "content": pending_tool_results})
   145	                    pending_tool_results = []
   146	                if "tool_calls" in msg:
   147	                    blocks = []
   148	                    if content:
   149	                        blocks.append({"type": "text", "text": content})
   150	                    for tc in msg["tool_calls"]:
   151	                        func = tc.get("function", {})
   152	                        blocks.append(
   153	                            {
   154	                                "type": "tool_use",
   155	                                "id": tc.get("id"),
   156	                                "name": func.get("name"),
   157	                                "input": json.loads(
   158	                                    func.get("arguments", "{}")
   159	                                ),
   160	                            }
   161	                        )
   162	                    converted.append({"role": "assistant", "content": blocks})
   163	                else:
   164	                    converted.append({"role": "assistant", "content": content})
   165	            elif role == "tool":
   166	                tool_content = self._convert_content(content)
   167	                pending_tool_results.append(
   168	                    {
   169	                        "type": "tool_result",
   170	                        "tool_use_id": msg.get("tool_call_id"),
   171	                        "content": tool_content,
   172	                    }
   173	                )
   174	
   175	        if pending_tool_results:
   176	            converted.append({"role": "user", "content": pending_tool_results})
   177	
   178	        # 合并连续的相同 role 消息
   179	        merged = []
   180	        for msg in converted:
   181	            if merged and merged[-1]["role"] == msg["role"]:
   182	                prev = merged[-1]["content"]
   183	                curr = msg["content"]
   184	                if isinstance(prev, str) and isinstance(curr, str):
   185	                    merged[-1]["content"] = prev + "\n" + curr
   186	                elif isinstance(prev, list) and isinstance(curr, list):
   187	                    merged[-1]["content"] = prev + curr
   188	                elif isinstance(prev, str):
   189	                    merged[-1]["content"] = [{"type": "text", "text": prev}] + (
   190	                        curr if isinstance(curr, list) else [{"type": "text", "text": curr}]
   191	                    )
   192	                else:
   193	                    merged[-1]["content"] = prev + (
   194	                        [{"type": "text", "text": curr}] if isinstance(curr, str) else curr
   195	                    )
   196	            else:
   197	                merged.append(msg)
   198	
   199	        return system_prompt, merged

   201	    def _convert_content(self, content) -> Any:

   212	            elif isinstance(item, dict):
   213	                item_type = item.get("type")
   214	                if item_type == "text":
   215	                    converted.append({"type": "text", "text": item.get("text", "")})
   216	                elif item_type == "image_url":
   217	                    image_url = item.get("image_url", {})
   218	                    url = (
   219	                        image_url.get("url", "")
   220	                        if isinstance(image_url, dict)
   221	                        else str(image_url)
   222	                    )
   223	                    if url.startswith("data:"):
   224	                        try:
   225	                            header, data = url.split(",", 1)
   226	                            media_type = header.split(":")[1].split(";")[0]
   227	                            converted.append(
   228	                                {
   229	                                    "type": "image",
   230	                                    "source": {
   231	                                        "type": "base64",
   232	                                        "media_type": media_type,
   233	                                        "data": data,
   234	                                    },
   235	                                }
   236	                            )
   237	                        except Exception:
   238	                            converted.append(
   239	                                {"type": "text", "text": "[Image parse error]"}
   240	                            )
   241	                    else:
   242	                        converted.append(
   243	                            {"type": "text", "text": f"[Image URL: {url}]"}
   244	                        )
   245	                elif item_type == "image":
   246	                    converted.append(item)
   247	                else:
   248	                    converted.append({"type": "text", "text": str(item)})
   249	            else:
   250	                converted.append({"type": "text", "text": str(item)})

   252	        return converted if converted else ""

   154	    def __init__(self, db_path: str = "llm_factory.db"):
   155	        self.db_path = db_path
   156	        self._local = threading.local()
   157	        self._init_db()
   158	
   159	    def _get_conn(self) -> sqlite3.Connection:
   160	        if not hasattr(self._local, "conn") or self._local.conn is None:
   161	            self._local.conn = sqlite3.connect(self.db_path)
   162	            self._local.conn.row_factory = sqlite3.Row
   163	            self._local.conn.execute("PRAGMA journal_mode=WAL")
   164	            self._local.conn.execute("PRAGMA synchronous=NORMAL")
   165	            self._local.conn.execute("PRAGMA foreign_keys=ON")
   166	        return self._local.conn
   167	
   168	    def _init_db(self):
   169	        conn = self._get_conn()
   170	        # Deduplicate models before creating unique index
   171	        try:
   172	            conn.execute("""
   173	                DELETE FROM models WHERE id NOT IN (
   174	                    SELECT id FROM (
   175	                        SELECT id, ROW_NUMBER() OVER (
   176	                            PARTITION BY api_key_id, model_name
