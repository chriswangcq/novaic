     1	"""Tests for Workspace.materialize()."""
     2	
     3	import asyncio
     4	import pytest
     5	
     6	from novaic_cortex.store import MemoryStore
     7	from novaic_cortex.workspace import Workspace
     8	from cortex_tests.workspace_test_utils import make_workspace_with_store
     9	
    10	
    11	@pytest.fixture
    12	def ws():
    13	    workspace, store = make_workspace_with_store("agt-1")
    14	    asyncio.run(workspace.initialize())
    15	    asyncio.run(workspace.create_scope("scope-1", name="test"))

    17	
    18	
    19	class TestMaterialize:
    20	    def test_basic(self, ws):
    21	        path = asyncio.run(ws.materialize("scope-1", "image.png", b"\x89PNGfake"))
    22	        assert path == "/rw/scratch/image.png"
    23	        data = asyncio.run(ws.read_bytes(path))
    24	        assert data == b"\x89PNGfake"
    25	
    26	    def test_subdir_parameter_not_accepted(self, ws):
    27	        with pytest.raises(TypeError):
    28	            asyncio.run(ws.materialize("scope-1", "data.csv", b"a,b\n1,2", subdir="inputs"))
    29	
    30	    def test_rejects_slash_in_filename(self, ws):
    31	        with pytest.raises(ValueError, match="invalid filename"):
    32	            asyncio.run(ws.materialize("scope-1", "../etc/passwd", b"bad"))
    33	
    34	    def test_rejects_null_in_filename(self, ws):
    35	        with pytest.raises(ValueError, match="invalid filename"):
    36	            asyncio.run(ws.materialize("scope-1", "foo\x00bar", b"bad"))
    37	
    38	    def test_dotdot_subdir_parameter_not_accepted(self, ws):
    39	        with pytest.raises(TypeError):
    40	            asyncio.run(ws.materialize("scope-1", "ok.txt", b"data", subdir=".."))
