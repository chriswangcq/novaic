    40	def _bounded_body(value: object, *, limit: int) -> tuple[object, int, bool]:
    41	    if not isinstance(value, str):
    42	        return value, 0, False
    43	    total = len(value)
    44	    if total <= limit:
    45	        return value, total, False
    46	    return (
    47	        value[:limit] + f"\n\n... [truncated log body to {limit} chars; original chars={total}]",
    48	        total,

    50	    )

    53	def _with_bounded_bodies(log: dict, *, body_limit: int) -> dict:
    54	    bounded = dict(log)

   548	    }
   549	
   550	    // Choices
   551	    if (resObj.choices) {
   552	      html += renderResponseChoices(resObj.choices);
   553	    }
   554	
   555	    // Usage
   556	    if (resObj.usage) {
   557	      html += renderUsage(resObj.usage);
   558	    }
   559	  } else {

   561	  }
   562	  html += `</div>`;
   563	
   564	  return html;
