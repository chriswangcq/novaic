                    "mimeType": mime,
                })
            else:
                # Normal URL: represent unresolved display media as text.
                converted.append({"type": "text", "text": f"[Image: {url}]"})
                
    if not converted:
        converted.append({"type": "text", "text": empty_text})

    return json.dumps({"_mcp_content": converted}, ensure_ascii=False)
