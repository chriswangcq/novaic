   296	@register_handler(TaskTopics.CORTEX_PREPARE_LLM_CONTEXT)
   297	def handle_cortex_prepare_llm_context(payload: Dict[str, Any], ctx: dict) -> Dict[str, Any]:
   298	    """Collect explicit LLM assembly inputs and invoke the pure assembler.
   299	
   300	    Boundary responsibilities:
   301	    1. Read current Environment notification hints into Cortex.
   302	    2. Ask Cortex for the prepared context snapshot.
   303	    3. Ask Cortex/Common for active tool schemas.
   304	
   305	    Pure assembly responsibilities live in ``common.contracts.llm_assembly``.
   306	    """
   307	    try:
   308	        prepare_input = LLMPrepareContextInput.from_mapping(payload)
   309	    except ValueError as exc:
   310	        raise ValidationError(str(exc)) from exc
   311	
   312	    # Step 1: merge current Environment notifications into Cortex context.
   313	    from .context_handlers import handle_context_read
   314	    read_result = handle_context_read({
   315	        "scope_id": prepare_input.scope_id,
   316	        "agent_root_scope_id": prepare_input.agent_root_scope_id,
   317	        "wake_scope_path": prepare_input.wake_scope_path,
   318	        "agent_id": prepare_input.agent_id,
   319	        "subagent_id": prepare_input.subagent_id,
   320	        "user_id": prepare_input.user_id,
   321	    }, ctx)
   322	
   323	    if not read_result.get("success"):
   324	        return read_result
   325	
   326	    # Step 2: ContextEvent-backed context preparation. `get_cortex_bridge` is
   327	    # already imported at module top; the redundant local re-import that
   328	    # used to sit here defeated `patch("…cortex_handlers.get_cortex_bridge")`
   329	    # in tests (the local import would re-bind to the source module's
   330	    # reference, bypassing the patch).
   331	    bridge = get_cortex_bridge(
   332	        prepare_input.agent_root_scope_id,
   333	        prepare_input.user_id,
   334	        prepare_input.agent_id,
   335	    )
   336	    prepare_result = bridge.prepare_for_llm(prepare_input.agent_root_scope_id)
   337	    try:
   338	        cortex_context = CortexPreparedContext.from_mapping(prepare_result)
   339	    except ValueError as exc:
   340	        raise ValidationError(str(exc)) from exc
   341	
   342	    raw_tool_schemas = bridge.load_tool_schemas(
   343	        subagent_id=prepare_input.subagent_id,
   344	    )
   345	    if not raw_tool_schemas:
   346	        logger.warning("[cortex] No tool schemas returned from Cortex")
   347	
   348	    # Transient NO_TOOL_WARNING is driven only by the explicit saga signal.
   349	    # The pure assembler decides where to place it in the final request.
   350	    no_tool_retry = prepare_input.no_tool_retry_count
   351	    if no_tool_retry > 0:
   352	        logger.info(
   353	            "[cortex.prepare_llm_context] Injecting NO_TOOL_WARNING "
   354	            "(no_tool_retry_count=%d, scope=%s)",
   355	            no_tool_retry, prepare_input.scope_id,
   356	        )
   357	
   358	    assembly_output = assemble_llm_request_from_snapshot(
   359	        LLMContextAssemblySnapshot(
   360	            prepare_input=prepare_input,
   361	            cortex_context=cortex_context,
   362	            raw_tool_schemas=raw_tool_schemas,
   363	            new_messages=read_result.get("new_messages", []),
   364	        ),
   365	        no_tool_warning=NO_TOOL_WARNING,
   366	    )
   367	    return assembly_output.to_dict()

   142	def adapt_function_tools(raw_schemas: Iterable[dict[str, Any]]) -> tuple[list[dict[str, Any]], list[str]]:
   143	    """Adapt canonical tool schemas to OpenAI function-calling shape."""
   144	
   145	    tools: list[dict[str, Any]] = []
   146	    names: set[str] = set()
   147	    for schema in raw_schemas:
   148	        if not isinstance(schema, dict):
   149	            continue
   150	        name = schema.get("name", "")
   151	        if not name:
   152	            continue
   153	        tools.append({
   154	            "type": "function",
   155	            "function": {
   156	                "name": name,
   157	                "description": schema.get("description", ""),
   158	                "parameters": schema.get("parameters", {}),
   159	            },
   160	        })
   161	        names.add(str(name))
   162	    return tools, sorted(names)

   165	def format_active_skill_stack_message(
   166	    stack: list[dict[str, Any]],
   167	    *,
   168	    hidden_scope_ids: set[str] | None = None,
   169	) -> str:
   170	    """Render the active skill stack as a transient LLM system message."""
   171	
   172	    if not stack:
   173	        return ""
   174	    hidden_scope_ids = hidden_scope_ids or set()
   175	    frames = [
   176	        frame for frame in stack
   177	        if isinstance(frame, dict) and frame.get("scope_id") not in hidden_scope_ids
   178	    ]
   179	    if not frames:
   180	        return ""
   181	    frames = sorted(frames, key=lambda frame: frame.get("depth", 0))
   182	    lines = ["[Active skill stack (LIFO — close innermost skill first)]"]
   183	    for visible_depth, frame in enumerate(frames):
   184	        lines.append(
   185	            f"  depth {visible_depth}: {frame.get('skill_name', '?')} "
   186	            f"(scope_id={frame.get('scope_id', '?')})"
   187	        )
   188	    top = frames[-1]
   189	    top_scope = top.get("scope_id", "")
   190	    lines.append(
   191	        f"→ Stack top: scope_id={top_scope or '?'} "
   192	        f"(skill={top.get('skill_name', '?')}). "
   193	        f"To close this skill call: skill_end(scope_id='{top_scope}', report='...')"
   194	    )
   195	    return "\n".join(lines)

   198	def build_active_skill_stack_message(
   199	    stack: list[dict[str, Any]],
   200	    *,
   201	    hidden_scope_ids: set[str] | None = None,
   202	) -> dict[str, Any] | None:
   203	    """Build the transient control-plane stack snapshot system message."""
   204	
   205	    content = format_active_skill_stack_message(
   206	        stack,
   207	        hidden_scope_ids=hidden_scope_ids,

   209	    if not content:
   210	        return None
   211	    return {
   212	        "role": "system",
   213	        "content": content,
   214	        "_message_type": "SYSTEM_MESSAGE",
   215	        "_metadata": {
   216	            "skill_stack_snapshot": True,
   217	            "transient": True,
   218	            "source": "control_stack",
   219	        },

   223	def _is_active_skill_stack_snapshot(message: dict[str, Any]) -> bool:
   224	    if not isinstance(message, dict) or message.get("role") != "system":
   225	        return False
   226	    metadata = message.get("_metadata")
   227	    if isinstance(metadata, dict) and metadata.get("skill_stack_snapshot") is True:
   228	        return True
   229	    content = message.get("content")
   230	    return isinstance(content, str) and content.startswith("[Active skill stack")

   233	def assemble_llm_request(
   234	    assembly_input: LLMAssemblyInput,
   235	    *,
   236	    no_tool_warning: str = "",
   237	) -> LLMAssemblyOutput:
   238	    """Pure assembly of the final LLM request.
   239	
   240	    The caller is responsible for all IO. This function only copies explicit
   241	    inputs, appends transient system guidance, and adapts tool schemas.
   242	    """
   243	
   244	    if not assembly_input.scope_id:
   245	        raise ValueError("LLMAssemblyInput.scope_id is required")
   246	    if not assembly_input.agent_root_scope_id:
   247	        raise ValueError("LLMAssemblyInput.agent_root_scope_id is required")
   248	
   249	    messages = [
   250	        deepcopy(message)
   251	        for message in assembly_input.cortex_context.messages
   252	        if not _is_active_skill_stack_snapshot(message)
   253	    ]
   254	    stack = deepcopy(assembly_input.cortex_context.stack)
   255	
   256	    stack_message = build_active_skill_stack_message(stack)
   257	    if stack_message:
   258	        messages.append(stack_message)
   259	
   260	    if int(assembly_input.no_tool_retry_count or 0) > 0 and no_tool_warning:
   261	        messages.append({"role": "system", "content": no_tool_warning})
   262	
   263	    tools, tool_names = adapt_function_tools(assembly_input.raw_tool_schemas)
   264	    return LLMAssemblyOutput(
   265	        scope_id=assembly_input.scope_id,
   266	        agent_root_scope_id=assembly_input.agent_root_scope_id,
   267	        wake_scope_path=assembly_input.wake_scope_path,
   268	        messages=messages,
   269	        tools=tools,
   270	        tool_names=tool_names,
   271	        new_messages=deepcopy(assembly_input.new_messages),
   272	        stack=stack,
   273	    )
   274

   276	def assemble_llm_request_from_snapshot(
   277	    snapshot: LLMContextAssemblySnapshot,
   278	    *,
   279	    no_tool_warning: str = "",
   280	) -> LLMAssemblyOutput:
   281	    """Pure wrapper from collected context/tool snapshot to LLM request."""
   282	
   283	    return assemble_llm_request(
   284	        snapshot.to_assembly_input(),
   285	        no_tool_warning=no_tool_warning,
   286	    )

   120	    msg: Dict[str, Any],
   121	    current_round_id: Optional[str],
   122	    *,

   128	    is_current = (
   129	        current_round_id is None
   130	        or msg_round_id == current_round_id
   131	        or (not msg_round_id and tool_call_id in latest_tool_call_ids)
   132	    )

    17	def sanitize_context(context: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    18	    """
    19	    清理 context，确保 tool_results 紧跟在对应的 assistant+tool_calls 之后
    20	    
    21	    这对 LLM 调用很重要，否则 LLM 可能报错。
    22	    
    23	    处理：
    24	    1. 过滤无效消息（空 role、空 content）
    25	    2. 移除内部元数据字段（_round_id、_message_type 等）
    26	    3. assistant message with tool_calls 后面必须紧跟对应的 tool_result
    27	    4. 为缺失的 tool_result 添加占位符
    28	    5. 移除孤立的 tool_result
    29	    
    30	    Args:
    31	        context: 原始消息列表
    32	        
    33	    Returns:
    34	        清理后的消息列表
    35	        
    36	    Example:
    37	        >>> from task_queue.utils import sanitize_context
    38	        >>> messages = [
    39	        ...     {"role": "user", "content": "Hello"},
    40	        ...     {"role": "assistant", "tool_calls": [{"id": "call_1", "function": {"name": "test"}}]},
    41	        ...     {"role": "user", "content": "OK"},  # 乱序
    42	        ...     {"role": "tool", "tool_call_id": "call_1", "content": "result"},
    43	        ... ]
    44	        >>> sanitized = sanitize_context(messages)
    45	        >>> # tool_result 现在紧跟 assistant message
    46	    """
    47	    if not context:
    48	        return []
    49	    
    50	    # 预处理：过滤和清理消息
    51	    cleaned_context = []
    52	    for msg in context:
    53	        if not isinstance(msg, dict):
    54	            continue
    55	        
    56	        role = msg.get("role", "")
    57	        
    58	        # 跳过内部元数据消息（没有 role 但有 _round_id 等字段）
    59	        if not role and (msg.get("_round_id") or msg.get("_message_type") or msg.get("_idempotency_key")):
    60	            continue
    61	        
    62	        # 跳过空 role 的消息
    63	        if not role:
    64	            continue
    65	        
    66	        # 对于非 tool 消息，检查 content 是否有效
    67	        if role not in ("tool", "tool_result"):
    68	            content = msg.get("content")
    69	            tool_calls = msg.get("tool_calls")
    70	            # assistant 可以只有 tool_calls 没有 content
    71	            if role == "assistant" and tool_calls:
    72	                pass
    73	            elif not content and content != "":  # 允许空字符串
    74	                continue
    75	        
    76	        # Keep tool _projection until process_multimodal_messages runs. It is
    77	        # an internal handoff marker that decides whether current display()
    78	        # output may be converted into a provider-native image message; it is
    79	        # stripped before the final request leaves this module.
    80	        cleaned_msg = {}
    81	        for key in (
    82	            "role",
    83	            "content",
    84	            "reasoning_content",
    85	            "tool_calls",
    86	            "tool_call_id",
    87	            "name",
    88	        ):
    89	            if key in msg:
    90	                cleaned_msg[key] = msg[key]
    91	        if role in ("tool", "tool_result") and "_projection" in msg:
    92	            cleaned_msg["_projection"] = msg["_projection"]
    93	
    94	        cleaned_context.append(cleaned_msg)
    95	    
    96	    context = cleaned_context
    97	    
    98	    # 第一遍：收集所有消息并跟踪 tool_calls
    99	    assistant_tool_calls = {}  # tc_id -> {"assistant_idx": int, "name": str}
   100	    tool_results = {}  # tc_id -> msg
   101	    other_messages = []  # (original_idx, msg)
   102	    assistant_messages_with_tools = {}  # idx -> msg
   103	    
   104	    for idx, msg in enumerate(context):
   105	        role = msg.get("role")
   106	        
   107	        if role == "assistant" and msg.get("tool_calls"):
   108	            assistant_messages_with_tools[idx] = msg
   109	            for tc in msg["tool_calls"]:
   110	                tc_id = tc.get("id")
   111	                if tc_id:
   112	                    func = tc.get("function", {})
   113	                    tool_name = func.get("name", "unknown")
   114	                    assistant_tool_calls[tc_id] = {"assistant_idx": idx, "name": tool_name}
   115	                    
   116	        elif role == "tool_result" or role == "tool":
   117	            tc_id = msg.get("tool_call_id")
   118	            if tc_id and tc_id in assistant_tool_calls:
   119	                tool_results[tc_id] = msg
   120	            elif tc_id:
   121	                # 孤立的 tool_result，忽略
   122	                pass
   123	            else:
   124	                other_messages.append((idx, msg))
   125	        else:
   126	            other_messages.append((idx, msg))
   127	    
   128	    # 第二遍：按正确顺序构建结果
   129	    result = []
   130	    processed_assistant_indices = set()
   131	    
   132	    for orig_idx, msg in other_messages:
   133	        # 插入之前的 assistant+tool_calls+tool_results
   134	        for asst_idx in sorted(assistant_messages_with_tools.keys()):
   135	            if asst_idx < orig_idx and asst_idx not in processed_assistant_indices:
   136	                asst_msg = assistant_messages_with_tools[asst_idx]
   137	                result.append(asst_msg)
   138	                processed_assistant_indices.add(asst_idx)
   139	                
   140	                # 添加对应的 tool_results
   141	                for tc in asst_msg.get("tool_calls", []):
   142	                    tc_id = tc.get("id")
   143	                    if tc_id:
   144	                        if tc_id in tool_results:
   145	                            result.append(tool_results[tc_id])
   146	                        else:
   147	                            # 缺失的 tool_result，添加占位符
   148	                            func = tc.get("function", {})
   149	                            tool_name = func.get("name", "unknown")
   150	                            result.append({
   151	                                "role": "tool",
   152	                                "tool_call_id": tc_id,
   153	                                "name": tool_name,
   154	                                "content": "[Execution was interrupted. Please retry if needed.]"
   155	                            })
   156	        
   157	        result.append(msg)
   158	    
   159	    # 添加末尾剩余的 assistant+tool_results
   160	    for asst_idx in sorted(assistant_messages_with_tools.keys()):
   161	        if asst_idx not in processed_assistant_indices:
   162	            asst_msg = assistant_messages_with_tools[asst_idx]
   163	            result.append(asst_msg)
   164	            
   165	            for tc in asst_msg.get("tool_calls", []):
   166	                tc_id = tc.get("id")
   167	                if tc_id:
   168	                    if tc_id in tool_results:
   169	                        result.append(tool_results[tc_id])
   170	                    else:
   171	                        func = tc.get("function", {})
   172	                        tool_name = func.get("name", "unknown")
   173	                        result.append({
   174	                            "role": "tool",
   175	                            "tool_call_id": tc_id,
   176	                            "name": tool_name,
   177	                            "content": "[Execution was interrupted. Please retry if needed.]"
   178	                        })
   179	    
   180	    return result
