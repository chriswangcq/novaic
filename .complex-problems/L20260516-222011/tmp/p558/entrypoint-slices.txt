    64	    """Wire registry / internal-key into the app synchronously.

    66	    Previously done in an ``@app.on_event("startup")`` hook, but all the
    67	    work here is pure object construction (WorkspaceRegistry, ``set_*``

    78	    # the process dies before ``uvicorn.run`` — no silent downgrade to an
    79	    # in-process lock. Multi-worker / multi-replica Cortex is therefore
    80	    # always safe because every endpoint serializes through Redis.

     1	"""Cortex shell execution boundary.
     2	
     3	The public ``Sandbox`` facade composes two explicit substrates:
     4	
     5	* ``MountNamespaceLogicalFS`` owns Cortex filesystem semantics and the stable
     6	  private ``/cortex`` mount namespace.
     7	* A sandboxd SDK client owns remote process execution.
     8	
     9	There is deliberately no alternate local path adapter. If a host cannot provide
    10	the sandboxd service boundary, shell execution fails explicitly instead of
    11	silently running with different path semantics.
    12	"""
    53	class ShellExecutionOrchestrator:
    54	    """Acquire LogicalFS view, run a process, release view, assemble ShellResult."""
    55	
    56	    def __init__(
    57	        self,
    58	        logical_fs: MountNamespaceLogicalFS,
    59	        sandbox_executor: SandboxExecutor | None,
    60	        *,
    61	        max_wall_timeout: int | None = None,
    62	    ) -> None:
    63	        self.logical_fs = logical_fs
    64	        self.sandbox_executor = sandbox_executor
    65	        self._max_wall_timeout = max_wall_timeout
    66	
    67	    async def exec(
    68	        self,
    69	        command: str,
    70	        timeout: int = 30,
    71	        cwd: str | None = None,
    72	        extra_env: dict[str, str] | None = None,
    73	    ) -> ShellResult:
    74	        rejected = _reject_ephemeral_sandbox_paths(command)
    75	        if rejected is not None:
    76	            return rejected
    77	
    78	        if self._max_wall_timeout is not None:
    79	            timeout = min(timeout, self._max_wall_timeout)
    80	
    81	        if self.sandbox_executor is None:
    82	            return ShellResult(
    83	                stdout="",
    84	                stderr=(
    85	                    "Sandbox executor is not configured. Cortex shell execution "
    86	                    "must go through sandboxd via sandbox_sdk."
    87	                ),
    88	                exit_code=-2,
    89	                files_changed=[],
    90	            )
    91	
    92	        with tempfile.TemporaryDirectory(prefix="novaic-cortex-logicalfs-") as tmp:
    93	            root = Path(tmp)
    94	            try:
    95	                view = await self.logical_fs.acquire_view(root, cwd=cwd, extra_env=extra_env)
    96	            except ValueError as exc:
    97	                return ShellResult(
    98	                    stdout="",
    99	                    stderr=str(exc),
   100	                    exit_code=-2,
   101	                    files_changed=[],
   102	                )
   103	            if not view.cwd.is_dir():
   104	                stderr = self.logical_fs.sanitize_output(
   105	                    f"cwd is not a directory: {view.cwd}",
   106	                    view,
   107	                )
   108	                return ShellResult(stdout="", stderr=stderr, exit_code=-2, files_changed=[])
   109	
   110	            try:
   111	                mount_plan = self.logical_fs.process_mount_plan(view)
   112	            except RuntimeError as exc:
   113	                return ShellResult(stdout="", stderr=str(exc), exit_code=-2, files_changed=[])
   114	            process_result = await self.sandbox_executor.execute(
   115	                SandboxExecSpec(
   116	                    command=command,
   117	                    cwd=str(self.logical_fs.process_cwd(view)),
   118	                    env=view.env,
   119	                    timeout=timeout,
   120	                    display_command=command,
   121	                    mount=mount_plan,
   122	                )
   123	            )
   124	
   125	            try:
   126	                from novaic_cortex.observability import log_cortex
   127	
   128	                dur_ms = int(process_result.duration_s * 1000)
   129	                log_cortex(
   130	                    "sandbox.exec",
   131	                    cmd=command[:200],
   132	                    exit_code=process_result.exit_code,
   133	                    duration_ms=dur_ms,
   134	                    duration_s=round(process_result.duration_s, 3),
   135	                    fs_provider=view.capabilities.provider,
   136	                )
   137	            except ImportError:
   138	                pass
   139	
   140	            files_changed = await self.logical_fs.release_view(view)
   141	
   142	            stdout = process_result.stdout.decode("utf-8", errors="replace")
   143	            stderr = process_result.stderr.decode("utf-8", errors="replace")
   144	            stdout = self.logical_fs.sanitize_output(stdout, view)
   145	            stderr = self.logical_fs.sanitize_output(stderr, view)
   146	
   147	            return ShellResult(
   148	                stdout=stdout,
   149	                stderr=stderr,
   150	                exit_code=process_result.exit_code,
   151	                files_changed=files_changed,
   152	            )

   216	    async def _workspace_snapshot(
   217	        self,
   218	        *,
   219	        generated_rw_files: Mapping[str, bytes],
   220	        extra_env: Mapping[str, str] | None,
   221	    ) -> LogicalFSSnapshot:
   222	        files: dict[tuple[bool, str], LogicalFSFile] = {}
   223	
   224	        # Keep shell startup bounded. The previous implementation recursively
   225	        # materialized all of /ro, which made every shell call proportional to
   226	        # total agent history. Shell does not need the runtime tool-schema
   227	        # store or the installed skill library as local files; capability CLIs
   228	        # are generated into /cortex/bin and call Cortex/Business APIs when
   229	        # they need agent-harness data. Historical or broad reads must be
   230	        # explicit API reads, not implicit full mounts.
   231	        await self._add_file_if_exists(files, "/ro/config/engine.json")
   232	        await self._add_file_if_exists(files, "/ro/config/tools/_index.json")
   233	
   234	        scope_id = str((extra_env or {}).get("NOVAIC_SCOPE_ID") or "").strip()
   235	        wake_scope_path = str((extra_env or {}).get("NOVAIC_WAKE_SCOPE_PATH") or "").strip().rstrip("/")
   236	        if scope_id:
   237	            root_path = f"/ro/active/{scope_id}"
   238	            await self._add_file_if_exists(files, f"{root_path}/meta.json")
   239	            await self._add_file_if_exists(files, f"{root_path}/steps/_index.jsonl")
   240	            await self._add_file_if_exists(files, f"{root_path}/context_events/events.jsonl")
   241	            if wake_scope_path and (
   242	                wake_scope_path == root_path or wake_scope_path.startswith(root_path + "/")
   243	            ):
   244	                await self._add_tree(files, wake_scope_path + "/")
   245	
   246	        await self._add_rw_working_set(files, extra_env=extra_env)
   247	        for relpath, data in generated_rw_files.items():
   248	            files[(True, relpath.lstrip("/"))] = LogicalFSFile(
   249	                path=relpath.lstrip("/"),
   250	                data=data,
   251	                writable=True,
   252	            )
   253	        return LogicalFSSnapshot(files=tuple(files.values()))

   300	        self,
   301	        root: Path,
   302	        *,

   304	        extra_env: dict[str, str] | None,
   305	    ) -> LogicalFSView:

   171	def _preview_shell_stream(value: Any, *, limit: int = _SHELL_STREAM_PREVIEW_CHARS) -> tuple[str, bool, int]:
   172	    text = "" if value is None else str(value)
   173	    if len(text) <= limit:

   178	def _shell_result_text(result: dict[str, Any]) -> tuple[str, dict[str, Any]]:
   179	    exit_code = result.get("exit_code")
   180	    stdout_preview, stdout_truncated, stdout_chars = _preview_shell_stream(result.get("stdout", ""))
   181	    stderr_preview, stderr_truncated, stderr_chars = _preview_shell_stream(result.get("stderr", ""))
   182	    files_changed = result.get("files_changed") if isinstance(result.get("files_changed"), list) else []
   183	
   184	    lines = [f"exit_code={exit_code}"]
   185	    if stdout_preview:
   186	        lines.extend(["stdout:", stdout_preview])
   187	    if stderr_preview:
   188	        lines.extend(["stderr:", stderr_preview])
   189	    if files_changed:
   190	        lines.append("files_changed:")
   191	        lines.extend(f"- {path}" for path in files_changed[:40])
   192	        if len(files_changed) > 40:
   193	            lines.append(f"... [truncated {len(files_changed) - 40} changed files]")
   194	    if len(lines) == 1:
   195	        lines.append("shell completed with no stdout/stderr")

   362	def _display_content_for_llm(file_url: str, data: bytes, mime_type: str) -> dict:
   363	    filename = file_url.rstrip("/").rsplit("/", 1)[-1] or "file"
   364	    size = len(data)
   365	    content: list[dict[str, str]] = []
   366	
   367	    if mime_type.startswith("image/"):
   368	        if size <= _DISPLAY_INLINE_BUDGET_BYTES:
   369	            content.append({
   370	                "type": "image",
   371	                "data": base64.b64encode(data).decode("ascii"),
   372	                "mimeType": mime_type,
   373	            })
   374	        else:
   375	            content.append({
   376	                "type": "text",
   377	                "text": f"[Image: {filename} ({_human_size(size)}) — too large for context]",
   378	            })
   379	    elif mime_type.startswith("text/") or mime_type in _TEXT_MIME_TYPES or filename.rsplit(".", 1)[-1].lower() in _TEXT_EXTS:
   380	        text = data.decode("utf-8", errors="replace")
   381	        if len(text) > _DISPLAY_INLINE_BUDGET_BYTES:
   382	            text = text[:_DISPLAY_INLINE_BUDGET_BYTES] + f"\n\n... [Truncated — full file is {_human_size(size)}]"
   383	        content.append({"type": "text", "text": text})
   384	    else:
   385	        content.append({
   386	            "type": "text",
   387	            "text": f"[Binary file: {filename} ({_human_size(size)}, {mime_type})]",
   388	        })
   389	
   390	    return {
   391	        "success": True,
   392	        "message": f"Loaded {filename} ({_human_size(size)}, {mime_type})",
   393	        "file_url": file_url,
   394	        "mime_type": mime_type,
   395	        "size": size,
   396	        "_mcp_content": content,
   397	    }

   529	    """Execute tool — unified dispatch, unified content return."""
   530	    scope_id = payload.get("scope_id") or payload.get("runtime_id", "")

   532	    for field in ("round_id", "tool_call_id", "tool_name"):
   533	        if not payload.get(field):
   534	            raise ValidationError(f"Missing required field: {field}")
   535	    if not scope_id:
   536	        raise ValidationError("Missing required field: scope_id")
   537	
   538	    round_id = payload["round_id"]
   539	    tool_call_id = payload["tool_call_id"]
   540	    tool_name = payload["tool_name"]
   541	    arguments = payload.get("arguments", {})
   542	
   543	    if isinstance(arguments, str):
   544	        try:
   545	            arguments = json.loads(arguments)
   546	        except json.JSONDecodeError:
   547	            pass
   548	
   549	    agent_id = payload.get("agent_id", "")
   550	    subagent_id = payload.get("subagent_id", "main")
   551	    user_id = payload.get("user_id") or ctx.get("user_id") or ""
   552	
   553	    executor = _EXECUTORS.get(tool_name)
   554	    if not executor:
   555	        logger.warning(f"[tool] Unknown tool: {tool_name} ({tool_call_id})")
   556	        err_msg = f"Unknown tool: {tool_name}. Available: {', '.join(sorted(KNOWN_TOOLS))}"
   557	        step_ref = _tool_step_ref(scope_id, round_id, tool_call_id)
   558	        tool_result = _err(scope_id, round_id, tool_call_id, tool_name, err_msg)
   559	        tool_result["arguments"] = arguments if isinstance(arguments, dict) else {}
   560	        tool_result["step_ref"] = step_ref
   561	        return tool_result
   562	
   563	    deps = {
   564	        "gw_client": ctx.get("business_client"),
   565	        "agent_id": agent_id,
   566	        "subagent_id": subagent_id,
   567	        "tool_call_id": tool_call_id,
   568	        "tool_name": tool_name,
   569	        "user_id": user_id,
   570	    }
   571	    if tool_name in (
   572	        "shell",
   573	        "skill_begin",
   574	        "skill_end",
   575	    ):
   576	        from ..utils.cortex_bridge import get_cortex_bridge
   577	        deps["bridge"] = get_cortex_bridge(cortex_scope_id, user_id, agent_id)
   578	        deps["scope_id"] = cortex_scope_id
   579	        deps["wake_scope_path"] = payload.get("wake_scope_path") or payload.get("scope_path")
   580	
   581	    try:
   582	        result = executor(arguments if isinstance(arguments, dict) else {}, deps)
   583	        step_ref = _tool_step_ref(scope_id, round_id, tool_call_id)
   584	        tool_result = _ok(scope_id, round_id, tool_call_id, tool_name, result)

   586	        tool_result["step_ref"] = step_ref

   590	        # PR-17 bake-snapshot runner. The error is otherwise swallowed into

   592	        # exactly the silent-failure pattern PR-33 §"no silent failure"
   593	        # requires us to surface at the service boundary.
   594	        reason_type = type(e).__name__
   595	        reason_msg = str(e)[:160].replace("\n", " ")

   597	            "event=tool_call_failed tool=%s agent=%s subagent=%s "
   598	            "tool_call_id=%s reason_type=%s reason=%s",

   600	            tool_call_id, reason_type, reason_msg,
   601	        )
   602	        step_ref = _tool_step_ref(scope_id, round_id, tool_call_id)
   603	        tool_result = _err(scope_id, round_id, tool_call_id, tool_name, e)
   604	        tool_result["arguments"] = arguments if isinstance(arguments, dict) else {}
   605	        tool_result["step_ref"] = step_ref
   606	        return tool_result
   183	def process_multimodal_messages(
   184	    messages: List[Dict[str, Any]], 
   185	    provider: str = "openai"
   186	) -> List[Dict[str, Any]]:
   187	    """
   188	    处理消息中的多模态内容
   189	    
   190	    将 tool_result 中的图片提取出来，作为单独的 user message 添加
   191	    （因为 LLM API 不支持在 tool result 中放图片）
   192	    
   193	    关键：保证 assistant+tool_calls 后的所有 tool results 连续，
   194	    图片 user messages 添加到所有 tool results 之后。
   195	    
   196	    Args:
   197	        messages: 原始消息列表
   198	        provider: LLM 提供商 ("openai" 或 "anthropic")
   199	        
   200	    Returns:
   201	        处理后的消息列表
   202	        
   203	    Only current-turn ``display_perception`` tool messages may be converted to
   204	    provider-native image content. Shell and historical tool messages stay as
   205	    terminal-style text receipts.
   206	    """
   207	    if not HAS_MULTIMODAL:
   208	        return messages
   209	    
   210	    processed = []
   211	    pending_images = []  # 收集待添加的图片 user messages
   212	    
   213	    for i, msg in enumerate(messages):
   214	        role = msg.get("role")
   215	        content = msg.get("content", "")
   216	        
   217	        if role in ("tool_result", "tool"):
   218	            projection = msg.get("_projection")
   219	            tool_msg = {k: v for k, v in msg.items() if k != "_projection"}
   220	            if projection != "display_perception":
   221	                processed.append(tool_msg)
   222	                continue
   223	
   224	            try:
   225	                result = json.loads(content) if isinstance(content, str) else content
   226	            except (json.JSONDecodeError, TypeError):
   227	                result = {"content": content}
   228	
   229	            if isinstance(result, dict) and multimodal.has_images(result):
   230	                _, images = multimodal.extract_from_result(result)
   231	                text_only = multimodal.result_to_text_only(result)
   232	
   233	                # 添加文本版本的 tool result
   234	                processed.append({
   235	                    **tool_msg,
   236	                    "content": text_only,
   237	                })
   238	                
   239	                # 收集图片，但不立即添加
   240	                if images:
   241	                    tool_name = msg.get("name", "tool")
   242	                    description = f"[Image from {tool_name}]"
   243	                    
   244	                    if provider == "anthropic":
   245	                        img_content = multimodal.to_anthropic_content(images, description)
   246	                    else:
   247	                        img_content = multimodal.to_openai_content(images, description)
   248	                    
   249	                    pending_images.append({
   250	                        "role": "user",
   251	                        "content": img_content,
   252	                    })
   253	            else:
   254	                processed.append(tool_msg)
   255	        elif role == "user":
   256	            # User message: 若有 {text, attachments} 格式，转为 LLM content 数组
   257	            # 注意：Repository 层已统一将 content 解析为 dict
   258	            if pending_images:
   259	                processed.extend(pending_images)
   260	                pending_images = []
   261	            
   262	            if isinstance(content, dict) and "attachments" in content and content["attachments"]:
   263	                # 有附件，转为 LLM 格式（文本 + 附件描述）
   264	                from .user_content import user_content_to_llm
   265	                llm_content = user_content_to_llm(content, provider=provider)
   266	                processed.append({"role": "user", "content": llm_content})
   267	            elif isinstance(content, dict) and "text" in content:
   268	                # 无附件，直接使用 text
   269	                processed.append({"role": "user", "content": content.get("text", "")})
   270	            else:
   271	                # 兼容旧格式（纯字符串）
   272	                processed.append(msg)
   273	        else:
   274	            # 其他消息
   275	            if pending_images:
   276	                processed.extend(pending_images)
   277	                pending_images = []
   278	            processed.append(msg)
   279	    
   280	    # 处理末尾剩余的图片
   281	    if pending_images:
   282	        processed.extend(pending_images)
   283	    
   284	    return processed

   211	    text = result_data.get("text", "")
   212	    if max_text_len is not None and len(text) > max_text_len:
   213	        text = text[:max_text_len] + f"\n\n... [Truncated due to context budget length > {max_text_len}]"
   214	
   215	    if text:
   216	        converted.append({"type": "text", "text": text})
   217	
   218	    if include_display:
   219	        display_files = result_data.get("display_files", [])
   220	        for f in display_files:
   221	            url = f.get("url", "")
   222	            if not url:
   223	                continue
   224	            if url.startswith("data:"):
   225	                # Data URL
   226	                parts = url.split(",", 1)
   227	                mime = "image/png"
   228	                if len(parts) >= 1 and ";" in parts[0]:
   229	                    mime = parts[0].split(";")[0].replace("data:", "")
   230	                data_b64 = parts[1] if len(parts) > 1 else ""
   231	
   232	                converted.append({
   233	                    "type": "image",
   234	                    "data": data_b64,
   235	                    "mimeType": mime,
   236	                })
   237	            else:
   238	                # Normal URL: represent unresolved display media as text.
   239	                converted.append({"type": "text", "text": f"[Image: {url}]"})
   240	                
   241	    if not converted:
   242	        converted.append({"type": "text", "text": empty_text})
   243	
   244	    return json.dumps({"_mcp_content": converted}, ensure_ascii=False)

    27	    def _execute_sync(self, spec: SandboxExecSpec) -> SandboxExecResult:
    28	        url = self.base_url.rstrip("/") + "/v1/execute"
    29	        payload = json.dumps(
    30	            SandboxdExecuteRequest.from_spec(spec).to_dict(),
    31	            ensure_ascii=False,
    32	        ).encode("utf-8")
    33	        request = urllib.request.Request(
    34	            url,
    35	            data=payload,
    36	            headers={"Content-Type": "application/json"},
    37	            method="POST",
    38	        )
    39	        try:
    40	            with urllib.request.urlopen(request, timeout=self.request_timeout) as response:
    41	                raw = response.read()
    42	        except urllib.error.HTTPError as exc:
    43	            body = exc.read().decode("utf-8", errors="replace")
    44	            raise SandboxdClientError(f"sandboxd returned HTTP {exc.code}: {body}") from exc
    45	        except urllib.error.URLError as exc:
    46	            raise SandboxdClientError(f"sandboxd request failed: {exc.reason}") from exc
    47	        except TimeoutError as exc:
    48	            raise SandboxdClientError("sandboxd request timed out") from exc
    49	
    50	        try:
    51	            data = json.loads(raw.decode("utf-8"))
    52	            return SandboxdExecuteResponse.from_dict(data).to_result()
    53	        except (json.JSONDecodeError, UnicodeDecodeError, ValueError) as exc:
    54	            raise SandboxdClientError("sandboxd returned an invalid response") from exc

    86	class LocalLogicalFSProvider:
    87	    """Materialize explicit snapshots into local RO/RW views and observe patches."""
    88	
    89	    def materialize(
    90	        self,
    91	        *,
    92	        snapshot: LogicalFSSnapshot,
    93	        root: Path,
    94	        layout: LogicalFSLayout,
    95	        env: LogicalFSEnv,
    96	        cwd: str | None = None,
    97	    ) -> LogicalFSView:
    98	        ro = root / "ro"
    99	        rw = root / "rw"
   100	        bin_dir = root / "bin"
   101	        ro.mkdir(parents=True)
   102	        rw.mkdir(parents=True)
   103	        bin_dir.mkdir(parents=True)
   104	
   105	        for item in snapshot.files:
   106	            clean = item.path.strip("/")
   107	            if not clean:
   108	                continue
   109	            base = rw if item.writable else ro
   110	            dest = base / clean
   111	            dest.parent.mkdir(parents=True, exist_ok=True)
   112	            dest.write_bytes(item.data)
   113	
   114	        for stable_path in [*layout.writable_env.values(), *layout.writable_dirs]:
   115	            if stable_path.startswith(layout.paths.rw + "/"):
   116	                rel = stable_path.removeprefix(layout.paths.rw + "/")
   117	                (rw / rel).mkdir(parents=True, exist_ok=True)
   118	
   119	        proc_cwd = _resolve_cwd_under_root(rw, cwd)
   120	        base_env = {
   121	            "PATH": f"{layout.paths.bin}:/usr/local/bin:/usr/bin:/bin",
   122	            "RO": layout.paths.ro,
   123	            "RW": layout.paths.rw,
   124	            "CORTEX_RO": layout.paths.ro,
   125	            "CORTEX_RW": layout.paths.rw,
   126	            "CORTEX_ROOT": layout.paths.root,
   127	            "CORTEX_FS_PROVIDER": "logicalfs",
   128	            "CORTEX_TRUE_STABLE_PATH_MOUNT": "1",
   129	            "HOME": layout.paths.rw,
   130	        }
   131	        base_env.update(dict(layout.writable_env))
   132	        base_env.update(dict(env.values))
   133	        return LogicalFSView(
   134	            root=root,
   135	            ro=ro,
   136	            rw=rw,
   137	            bin=bin_dir,
   138	            cwd=proc_cwd,
   139	            stable_cwd=_stable_path_for_actual(
   140	                proc_cwd,
   141	                view_root=root,
   142	                ro=ro,
   143	                rw=rw,
   144	                stable=layout.paths,
   145	            ),
   146	            env=base_env,
   147	            before_rw=_file_stats(rw),
   148	            paths=layout.paths,
   149	        )
   150	
   151	    def observe_patch(self, view: LogicalFSView) -> LogicalFSPatch:
   152	        after = _file_stats(view.rw)
   153	        upserts: list[LogicalFSPatchFile] = []
   154	        deletes: list[str] = []
   155	        for rel in _changed_relative_paths(view.before_rw, after):
   156	            logical = _logical_rw_path(rel, view.paths)
   157	            if rel not in after:
   158	                deletes.append(logical)
   159	            else:
   160	                upserts.append(LogicalFSPatchFile(path=logical, data=(view.rw / rel).read_bytes()))
   161	        return LogicalFSPatch(upserts=tuple(upserts), deletes=tuple(sorted(deletes)))
   162	
   163	    def sanitize_output(self, text: str, view: LogicalFSView) -> str:
   164	        return sanitize_paths(
   165	            text,
   166	            {
   167	                str(view.ro.resolve()): view.paths.ro,
   168	                str(view.rw.resolve()): view.paths.rw,
   169	                str(view.root.resolve()): view.paths.root,
   170	            },
   171	        )
     1	"""Blob Service object-store adapter below LogicalFS.
     2	
     3	Blob Service is a cheap byte/object server. LogicalFS owns realtime /ro and /rw
     4	file authority above this adapter; callers should not use this class as a
     5	semantic workspace API.
     6	"""
    18	class BlobObjectStore:
    19	    """Async object-store adapter over Blob Service object APIs."""
    20	
    21	    def __init__(
    22	        self,
    23	        *,
    24	        base_url: str,
    25	        tenant_id: str,
    26	        namespace: str = "cortex-store",
    27	        timeout: float = 10.0,
    28	        client_service_name: str = "logicalfs",
    29	    ) -> None:
    30	        if not base_url or not base_url.strip():
    31	            raise ValueError("base_url is required")
    32	        if not tenant_id or not tenant_id.strip():
    33	            raise ValueError("tenant_id is required")
    34	        self._base_url = base_url.rstrip("/")
    35	        self._tenant_id = tenant_id
    36	        self._namespace = namespace
    37	        self._timeout = timeout
    38	        self._client_service_name = client_service_name
    39	
    40	    def _headers(self) -> dict[str, str]:
    41	        return {"X-Tenant-ID": self._tenant_id}
    42	
    43	    def _object_url(self, key: str) -> str:
    44	        return f"{self._base_url}/v1/objects/{quote(self._namespace)}/{quote(key, safe='/')}"
    45	
    46	    async def put(self, key: str, data: bytes, content_type: str = "") -> None:
    47	        headers = self._headers()
    48	        if content_type:
    49	            headers["Content-Type"] = content_type
    50	        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
    51	            resp = await client.put(self._object_url(key), headers=headers, content=data)
    52	            resp.raise_for_status()
    53	
    54	    async def get(self, key: str) -> bytes:
    55	        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
    56	            resp = await client.get(self._object_url(key), headers=self._headers())
    57	        if resp.status_code == 404:
    58	            raise KeyError(key)
    59	        resp.raise_for_status()
    60	        return resp.content
    61	
    62	    async def exists(self, key: str) -> bool:
    63	        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
    64	            resp = await client.get(
    65	                f"{self._object_url(key)}/info",
    66	                headers=self._headers(),
    67	            )
    68	        if resp.status_code == 404:
    69	            return False
    70	        resp.raise_for_status()
    71	        return True
    72	
    73	    async def delete(self, key: str) -> None:
    74	        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
    75	            resp = await client.delete(self._object_url(key), headers=self._headers())
    76	        if resp.status_code == 404:
    77	            return
    78	        resp.raise_for_status()
    79	
    80	    async def list_objects(self, prefix: str, delimiter: str = "/") -> list[LogicalObjectInfo]:
    81	        if delimiter != "/":
    82	            raise NotImplementedError("BlobObjectStore only supports delimiter='/'")
    83	        data = await self._list(prefix=prefix, delimiter=delimiter, recursive=False)
    84	        return [
    85	            LogicalObjectInfo(
    86	                key=str(item["key"]),
    87	                size=int(item.get("size") or 0),
    88	                is_prefix=bool(item.get("is_prefix")),
    89	            )
    90	            for item in data.get("objects", [])
    91	        ]
    92	
    93	    async def list_recursive(self, prefix: str) -> list[str]:
    94	        data = await self._list(prefix=prefix, delimiter="/", recursive=True)
    95	        return sorted(str(item["key"]) for item in data.get("objects", []))
    96	
    97	    async def _list(self, *, prefix: str, delimiter: str, recursive: bool) -> dict[str, Any]:
    98	        params = {
    99	            "namespace": self._namespace,
   100	            "prefix": prefix,
   101	            "delimiter": delimiter,
   102	            "recursive": "true" if recursive else "false",
   103	        }
   104	        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
   105	            resp = await client.get(
   106	                f"{self._base_url}/v1/objects/_list",
   107	                params=params,
   108	                headers=self._headers(),
   109	            )
   110	            resp.raise_for_status()
   111	            return resp.json()
   112	
   113	    async def move(self, src: str, dst: str) -> None:
   114	        data = await self.get(src)
   115	        await self.put(dst, data)
   116	        await self.delete(src)
   117	
   118	    async def move_prefix(self, src_prefix: str, dst_prefix: str) -> int:
   119	        body = {
   120	            "tenant_id": self._tenant_id,
   121	            "namespace": self._namespace,
   122	            "src_prefix": src_prefix,
   123	            "dst_prefix": dst_prefix,
   124	        }
   125	        async with internal_async_client(self._client_service_name, timeout=self._timeout) as client:
   126	            resp = await client.post(f"{self._base_url}/v1/objects/_move-prefix", json=body)
   127	            resp.raise_for_status()
   128	            data = resp.json()
   129	        return int(data.get("moved") or 0)
