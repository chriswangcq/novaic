    35	def _text_values(value):
    36	    if isinstance(value, dict):
    37	        for key, item in value.items():
    38	            if key == "text" and isinstance(item, str):
    39	                yield item
    40	            yield from _text_values(item)
    41	    elif isinstance(value, list):
    42	        for item in value:
    43	            yield from _text_values(item)
